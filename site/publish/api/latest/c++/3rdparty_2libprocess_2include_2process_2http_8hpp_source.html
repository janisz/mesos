293 // Represents an asynchronous in-memory unbuffered Pipe, currently
294 // used for streaming HTTP responses via chunked encoding. Note that
295 // being an in-memory pipe means that this cannot be used across OS
296 // processes.
297 //
298 // Much like unix pipes, data is read until end-of-file is
299 // encountered; this occurs when the write-end of the pipe is
300 // closed and there is no outstanding data left to read.
301 //
302 // Unlike unix pipes, if the read-end of the pipe is closed before
303 // the write-end is closed, rather than receiving SIGPIPE or EPIPE
304 // during a write, the writer is notified via a future. Like unix
305 // pipes, we are not notified if the read-end is closed after the
306 // write-end is closed, even if data is remaining in the pipe!
307 //
308 // No buffering means that each non-empty write to the pipe will
309 // correspond to to an equivalent read from the pipe, and the
310 // reader must "keep up" with the writer in order to avoid
311 // unbounded memory growth.
312 //
313 // The writer can induce a failure on the reader in order to signal
314 // that an error has occurred. For example, if we are receiving a
315 // response but a disconnection occurs before the response is
316 // completed, we want the reader to detect that a disconnection
317 // occurred!
318 //
319 // TODO(bmahler): Consider aggregating writes into larger reads to
320 // help the reader keep up (a process::Stream abstraction with
321 // backpressure would obviate the need for this).
322 //
323 // TODO(bmahler): Add a more general process::Stream<T> abstraction
324 // to represent asynchronous finite/infinite streams (possibly
325 // with "backpressure" on the writer). This is broadly useful
326 // (e.g. allocator can expose Stream<Allocation>, http::Pipe
327 // becomes Stream<string>, process::Queue<T> is just an infinite
328 // Stream<T> (i.e. completion and error semantics hidden)).
329 class Pipe
330 {
331 private:
332  struct Data; // Forward declaration.
333 
334 public:
335  class Reader
336  {
337  public:
338  // Returns data written to the pipe.
339  // Returns an empty read when end-of-file is reached.
340  // Returns Failure if the writer failed, or the read-end
341  // is closed.
342  Future<std::string> read();
343 
344  // Closing the read-end of the pipe before the write-end closes
345  // or fails will notify the writer that the reader is no longer
346  // interested. Returns false if the read-end was already closed.
347  bool close();
348 
349  // Comparison operators useful for checking connection equality.
350  bool operator==(const Reader& other) const { return data == other.data; }
351  bool operator!=(const Reader& other) const { return !(*this == other); }
352 
353  private:
354  friend class Pipe;
355 
356  enum State
357  {
358  OPEN,
359  CLOSED,
360  };
361 
362  explicit Reader(const std::shared_ptr<Data>& _data) : data(_data) {}
363 
364  std::shared_ptr<Data> data;
365  };
366 
367  class Writer
368  {
369  public:
370  // Returns false if the data could not be written because
371  // either end of the pipe was already closed. Note that an
372  // empty write has no effect.
373  bool write(std::string s);
374 
375  // Closing the write-end of the pipe will send end-of-file
376  // to the reader. Returns false if the write-end of the pipe
377  // was already closed or failed.
378  bool close();
379 
380  // Closes the write-end of the pipe but sends a failure
381  // to the reader rather than end-of-file. Returns false
382  // if the write-end of the pipe was already closed or failed.
383  bool fail(const std::string& message);
384 
385  // Returns Nothing when the read-end of the pipe is closed
386  // before the write-end is closed, which means the reader
387  // was unable to continue reading!
388  Future<Nothing> readerClosed() const;
389 
390  // Comparison operators useful for checking connection equality.
391  bool operator==(const Writer& other) const { return data == other.data; }
392  bool operator!=(const Writer& other) const { return !(*this == other); }
393  private:
394  friend class Pipe;

396  enum State
397  {
398  OPEN,
399  CLOSED,
400  FAILED,
401  };
402 
403  explicit Writer(const std::shared_ptr<Data>& _data) : data(_data) {}
404 
405  std::shared_ptr<Data> data;
406  };

409 
410  Reader reader() const;
411  Writer writer() const;
412 
413  // Comparison operators useful for checking connection equality.
414  bool operator==(const Pipe& other) const { return data == other.data; }
415  bool operator!=(const Pipe& other) const { return !(*this == other); }
416 private:
417  struct Data
418  {
419  Data()
420  : readEnd(Reader::OPEN),
421  writeEnd(Writer::OPEN) {}
422 
423  // Rather than use a process to serialize access to the pipe's
424  // internal data we use a 'std::atomic_flag'.
425  std::atomic_flag lock = ATOMIC_FLAG_INIT;
426 
427  Reader::State readEnd;
428  Writer::State writeEnd;
429 
430  // Represents readers waiting for data from the pipe.
431  std::queue<Owned<Promise<std::string>>> reads;
432 
433  // Represents unread writes in the pipe. Note that we omit
434  // empty strings as they serve as a signal for end-of-file.
435  std::queue<std::string> writes;
436 
437  // Signals when the read-end is closed before the write-end.
438  Promise<Nothing> readerClosure;
439 
440  // Failure reason when the 'writeEnd' is FAILED.
441  Option<Failure> failure;
442  };
443 
444  std::shared_ptr<Data> data;

478  // Either provide a 'body', an absolute 'path' to a file, or a
479  // 'pipe' for streaming a response. Distinguish between the cases
480  // using 'type' below.
481  //
482  // BODY: Uses 'body' as the body of the response. These may be
483  // encoded using gzip for efficiency, if 'Content-Encoding' is not
484  // already specified.
485  //
486  // PATH: Attempts to perform a 'sendfile' operation on the file
487  // found at 'path'.
488  //
489  // PIPE: Splices data from the Pipe 'reader' using a "chunked"
490  // 'Transfer-Encoding'. The writer uses a Pipe::Writer to
491  // perform writes and to detect a closed read-end of the Pipe
492  // (i.e. nobody is listening any longer). Once the writer is
493  // finished, it will close its end of the pipe to signal end
494  // of file to the Reader.
495  //
496  // In all cases (BODY, PATH, PIPE), you are expected to properly
497  // specify the 'Content-Type' header, but the 'Content-Length' and
498  // or 'Transfer-Encoding' headers will be filled in for you.
499  enum
500  {
501  NONE,
502  BODY,
503  PATH,
504  PIPE
505  } type;
506 
507  std::string body;
508  std::string path;
509  Option<Pipe::Reader> reader;
510 
511  uint16_t code;
512 };

763 //
764 // decode("foo=1&bar=%20&baz&foo=3")
765 //
766 // Would return a map with the following:
767 // bar: " "
768 // baz: ""
769 // foo: "3"
770 //
771 // We use the last value for a key for simplicity, since the RFC does not
772 // specify how to handle duplicate keys:
773 // http://en.wikipedia.org/wiki/Query_string
