94  // Returns a string describing the flags, preceded by a "usage
95  // message" that will be prepended to that description (see
96  // 'FlagsBase::usageMessage_').
97  //
98  // The optional 'message' passed to this function will be prepended
99  // to the generated string returned from this function.
100  //
101  // Derived classes and clients can modify the standard usage message
102  // by setting it before calling this method via 'setUsageMessage()'.
103  //
104  // This allows one to set a generic message that will be used for
105  // each invocation of this method, and to additionally add one for
106  // the particular invocation:
107  //
108  // MyFlags flags;
109  // flags.setUsageMessage("Custom Usage Message");
110  // ...
111  // if (flags.foo.isNone()) {
112  // std::cerr << flags.usage("Missing required --foo flag");
113  // }
114  //
115  // The 'message' would be emitted first, followed by a blank line,
116  // then the 'usageMessage_', finally followed by the flags'
117  // description, for example:
118  //

120  //
121  // Custom Usage Message
122  //
123  // --[no-]help Prints this help message. (default: false)

126  //

411  const T2* t2,
412  F validate)
413 {
414  // Don't bother adding anything if the pointer is `nullptr`.
415  if (t1 == nullptr) {
416  return;
417  }
418 
419  Flag flag;

677 inline void FlagsBase::add(const Flag& flag)
678 {
679  // Check if the name and alias of the flag are valid.
680  std::vector<Name> names = {flag.name};
681  if (flag.alias.isSome()) {
682  if (flag.alias.get() == flag.name) {
683  EXIT(EXIT_FAILURE)
684  << "Attempted to add flag '" << flag.name.value << "' with an alias"
685  << " that is same as the flag name";
686  }
687 
688  names.push_back(flag.alias.get());
689  }
690 
691  foreach (const Name& name, names) {
692  if (flags_.count(name.value) > 0) {
693  EXIT(EXIT_FAILURE)
694  << "Attempted to add duplicate flag '" << name.value << "'";
695  } else if (name.value.find("no-") == 0) {
696  EXIT(EXIT_FAILURE)
697  << "Attempted to add flag '" << name.value
698  << "' that starts with the reserved 'no-' prefix";
699  }
700  }
701 
702  flags_[flag.name.value] = flag;
703  if (flag.alias.isSome()) {
704  aliases[flag.alias.get().value] = flag.name.value;
705  }
706 }

803  const Option<std::string>& prefix,
804  int* argc,
805  char*** argv,
806  bool unknowns,
807  bool duplicates)
808 {

810 
811  // Grab the program name from argv, without removing it.
812  programName_ = *argc > 0 ? Path(*(argv[0])).basename() : "";
813 
814  // Keep the arguments that are not being processed as flags.
815  std::vector<char*> args;
816 
817  // Read flags from the command line.
818  for (int i = 1; i < *argc; i++) {
819  const std::string arg(strings::trim((*argv)[i]));
820 
821  // Stop parsing flags after '--' is encountered.
822  if (arg == "--") {
823  // Save the rest of the arguments.
824  for (int j = i + 1; j < *argc; j++) {
825  args.push_back((*argv)[j]);
826  }
827  break;
828  }
829 
830  // Skip anything that doesn't look like a flag.
831  if (arg.find("--") != 0) {
832  args.push_back((*argv)[i]);
833  continue;
834  }
835 
836  std::string name;
837  Option<std::string> value = None();
838 
839  size_t eq = arg.find_first_of("=");
840  if (eq == std::string::npos && arg.find("--no-") == 0) { // --no-name
841  name = arg.substr(2);
842  } else if (eq == std::string::npos) { // --name
843  name = arg.substr(2);
844  } else { // --name=value
845  name = arg.substr(2, eq - 2);
846  value = arg.substr(eq + 1);
847  }
848 
849  name = strings::lower(name);
850 
851  values.put(name, value);
852  }
853 
854  if (prefix.isSome()) {
855  // Merge in flags from the environment. Command-line
856  // flags take precedence over environment flags.
857  foreachpair (const std::string& name,
858  const Option<std::string>& value,
859  extract(prefix.get())) {
860  if (!values.contains(name)) {
861  values.put(name, value);
862  }
863  }
864  }
865 
866  Try<Warnings> result = load(values, unknowns, duplicates);
867 
868  // Update 'argc' and 'argv' if we successfully loaded the flags.
869  if (!result.isError()) {
870  CHECK_LE(args.size(), (size_t) *argc);
871  size_t i = 1; // Start at '1' to skip argv[0].
872  foreach (char* arg, args) {
873  (*argv)[i++] = arg;
874  }
875 
876  *argc = i;
877 
878  // Now null terminate the array. Note that we'll "leak" the
879  // arguments that were processed here but it's not like they would
880  // have gotten deleted in normal operations anyway.
881  (*argv)[i++] = nullptr;
882  }
883 
884  return result;
885 }

914 inline Try<Warnings> FlagsBase::load(
915  const Multimap<std::string, Option<std::string>>& values,
916  bool unknowns,
917  bool duplicates)
918 {
919  Warnings warnings;
920 
921  foreachpair (const std::string& name,
922  const Option<std::string>& value,
923  values) {
924  bool is_negated = strings::startsWith(name, "no-");
925  std::string flag_name = !is_negated ? name : name.substr(3);
926 
927  auto iter = aliases.count(flag_name)
928  ? flags_.find(aliases[flag_name])
929  : flags_.find(flag_name);
930 
931  if (iter == flags_.end()) {
932  if (!unknowns) {
933  return Error("Failed to load unknown flag '" + flag_name + "'" +
934  (!is_negated ? "" : " via '" + name + "'"));
935  } else {
936  continue;
937  }
938  }
939 
940  Flag* flag = &(iter->second);
941 
942  if (!duplicates && flag->loaded_name.isSome()) {
943  return Error("Flag '" + flag_name + "' is already loaded via name '" +
944  flag->loaded_name->value + "'");
945  }
946 
947  // Validate the flag value.
948  std::string value_;
949  if (!flag->boolean) { // Non-boolean flag.
950  if (is_negated) { // Non-boolean flag cannot be loaded with "no-" prefix.
951  return Error("Failed to load non-boolean flag '" + flag_name +
952  "' via '" + name + "'");
953  }
954 
955  if (value.isNone()) {
956  return Error("Failed to load non-boolean flag '" + flag_name +
957  "': Missing value");
958  }
959 
960  value_ = value.get();
961  } else { // Boolean flag.
962  if (value.isNone() || value.get() == "") {
963  value_ = !is_negated ? "true" : "false";
964  } else if (!is_negated) {
965  value_ = value.get();
966  } else { // Boolean flag with "no-" prefix cannot have non-empty value.
967  return Error(
968  "Failed to load boolean flag '" + flag_name + "' via '" + name +
969  "' with value '" + value.get() + "'");
970  }
971  }
972 
973  Try<Nothing> load = flag->load(this, value_);
974  if (load.isError()) {
975  return Error("Failed to load flag '" + flag_name + "': " + load.error());
976  }
977 
978  // TODO(vinod): Move this logic inside `Flag::load()`.
979 
980  // Set `loaded_name` to the Name corresponding to `flag_name`.
981  if (aliases.count(flag_name)) {
982  CHECK_SOME(flag->alias);
983  flag->loaded_name = flag->alias.get();
984  } else {
985  flag->loaded_name = flag->name;
986  }
987 
988  if (flag->loaded_name->deprecated) {
989  warnings.warnings.push_back(
990  Warning("Loaded deprecated flag '" + flag_name + "'"));
991  }
992  }
993 
994  // Validate the flags value.
995  //
996  // TODO(benh): Consider validating all flags at the same time in
997  // order to provide more feedback rather than requiring a user to
998  // fix one at a time.
999  foreachvalue (const Flag& flag, flags_) {
1000  if (flag.required && flag.loaded_name.isNone()) {
1001  return Error(
1002  "Flag '" + flag.name.value +
1003  "' is required, but it was not provided");
1004  }
1005 
1006  Option<Error> error = flag.validate(*this);
1007  if (error.isSome()) {
1008  return error.get();
1009  }
1010  }
1011 
1012  return warnings;
1013 }
