79 template <>
80 inline Try<JSON::Object> parse(const std::string& value)
81 {
82  // A value that already starts with 'file://' will properly be
83  // loaded from the file and put into 'value' but if it starts with
84  // '/' we need to explicitly handle it for backwards compatibility
85  // reasons (because we used to handle it before we introduced the
86  // 'fetch' mechanism for flags that first fetches the data from URIs
87  // such as 'file://').
88  if (strings::startsWith(value, "/")) {
89  LOG(WARNING) << "Specifying an absolute filename to read a command line "
90  "option out of without using 'file:// is deprecated and "
91  "will be removed in a future release. Simply adding "
92  "'file://' to the beginning of the path should eliminate "
93  "this warning.";
94 
95  Try<std::string> read = os::read(value);
96  if (read.isError()) {
97  return Error("Error reading file '" + value + "': " + read.error());
98  }
99  return JSON::parse<JSON::Object>(read.get());
100  }
101  return JSON::parse<JSON::Object>(value);
102 }
