212 inline Try<IP> IP::create(const struct sockaddr_storage& _storage)
213 {
214  // According to POSIX: (IEEE Std 1003.1, 2004)
215  //
216  // (1) `sockaddr_storage` is "aligned at an appropriate boundary so that
217  // pointers to it can be cast as pointers to protocol-specific address
218  // structures and used to access the fields of those structures without
219  // alignment problems."
220  //
221  // (2) "When a `sockaddr_storage` structure is cast as a `sockaddr`
222  // structure, the `ss_family` field of the `sockaddr_storage` structure
223  // shall map onto the `sa_family` field of the `sockaddr` structure."
224  //
225  // Therefore, casting from `const sockaddr_storage*` to `const sockaddr*`
226  // (then subsequently dereferencing the `const sockaddr*`) should be safe.
227  // Note that casting in the reverse direction (`const sockaddr*` to
228  // `const sockaddr_storage*`) would NOT be safe, since the former might
229  // not be aligned appropriately.
230  const auto* addr = reinterpret_cast<const struct sockaddr*>(&_storage);
231  return create(*addr);
232 }

252 {

343 inline Try<IPNetwork> IPNetwork::parse(const std::string& value, int family)
344 {
345  std::vector<std::string> tokens = strings::split(value, "/");
346 
347  if (tokens.size() != 2) {
348  return Error(
349  "Unexpected number of '/' detected: " +
350  stringify(tokens.size()));
351  }
352 
353  // Parse the IP address.
354  Try<IP> address = IP::parse(tokens[0], family);
355  if (address.isError()) {
356  return Error("Failed to parse the IP address: " + address.error());
357  }
358 
359  // Parse the subnet prefix.
360  Try<int> prefix = numify<int>(tokens[1]);
361  if (prefix.isError()) {
362  return Error("Subnet prefix is not a number");
363  }
364 
365  return create(address.get(), prefix.get());
366 }

375 inline Try<IPNetwork> IPNetwork::create(const IP& address, const IP& netmask)
376 {
377  if (address.family() != netmask.family()) {
378  return Error(
379  "The network families of the IP address '" +
380  stringify(address.family()) + "' and the IP netmask '" +
381  stringify(netmask.family()) + "' do not match");
382  }
383 
384  switch (address.family()) {
385  case AF_INET: {
386  uint32_t mask = ntohl(netmask.in().get().s_addr);
387  if (((~mask + 1) & (~mask)) != 0) {
388  return Error("Netmask is not valid");
389  }
390 
391  return IPNetwork(address, netmask);
392  }
393  default: {
394  UNREACHABLE();
395  }
396  }
397 }

400 inline Try<IPNetwork> IPNetwork::create(const IP& address, int prefix)
401 {
402  if (prefix < 0) {
403  return Error("Subnet prefix is negative");
404  }
405 
406  switch (address.family()) {
407  case AF_INET: {
408  if (prefix > 32) {
409  return Error("Subnet prefix is larger than 32");
410  }
411 
412  // Avoid left-shifting by 32 bits when prefix is 0.
413  uint32_t mask = 0;
414  if (prefix > 0) {
415  mask = 0xffffffff << (32 - prefix);
416  }
417  return IPNetwork(address, IP(mask));
418  }
419  default: {
420  UNREACHABLE();
421  }
422  }

426 inline Result<IPNetwork> IPNetwork::fromLinkDevice(
427  const std::string& name,
428  int family)
429 {
430 #if !defined(__linux__) && !defined(__APPLE__) && !defined(__FreeBSD__)
431  return Error("Not implemented");
432 #else
433  if (family != AF_INET) {
434  return Error("Unsupported family type: " + stringify(family));
435  }
436 
437  struct ifaddrs* ifaddr = nullptr;
438  if (getifaddrs(&ifaddr) == -1) {
439  return ErrnoError();
440  }
441 
442  // Indicates whether the link device is found or not.
443  bool found = false;
444 
445  for (struct ifaddrs* ifa = ifaddr; ifa != nullptr; ifa = ifa->ifa_next) {
446  if (ifa->ifa_name != nullptr && !strcmp(ifa->ifa_name, name.c_str())) {
447  found = true;
448 
449  if (ifa->ifa_addr != nullptr && ifa->ifa_addr->sa_family == family) {
450  IP address = IP::create(*ifa->ifa_addr).get();
451 
452  if (ifa->ifa_netmask != nullptr &&
453  ifa->ifa_netmask->sa_family == family) {
454  IP netmask = IP::create(*ifa->ifa_netmask).get();
455 
456  freeifaddrs(ifaddr);
457 
458  Try<IPNetwork> network = IPNetwork::create(address, netmask);
459  if (network.isError()) {
460  return Error(network.error());
461  }
462 
463  return network.get();
464  }
465 
466  freeifaddrs(ifaddr);
467 
468  // Note that this is the case where netmask is not specified.
469  // We've seen such cases when VPN is used. In that case, a
470  // default /32 prefix is used.
471  Try<IPNetwork> network = IPNetwork::create(address, 32);
472  if (network.isError()) {
473  return Error(network.error());
474  }
475 
476  return network.get();
477  }
478  }
479  }
480 
481  freeifaddrs(ifaddr);
482 
483  if (!found) {
484  return Error("Cannot find the link device");
485  }
486 
487  return None();
488 #endif
489 }
