68 // Initializes libraries that net:: functions depend on, in a
69 // thread-safe way. This does not have to be called explicitly by
70 // the user of any functions in question. They will call this
71 // themselves by need.
72 inline void initialize()
73 {
74  // We use a static struct variable to initialize in a thread-safe
75  // way, at least with respect to calls within net::*, since there
76  // is no way to guarantee that another library is not concurrently
77  // initializing CURL. Thread safety is provided by the fact that
78  // the value 'curl' should get constructed (i.e., the CURL
79  // constructor invoked) in a thread safe way (as of GCC 4.3 and
80  // required for C++11).
81  struct CURL
82  {
83  CURL()
84  {
85  // This is the only one function in libcurl that is not deemed
86  // thread-safe. (And it must be called at least once before any
87  // other libcurl function is used.)
88  curl_global_init(CURL_GLOBAL_ALL);
89  }
90  };
91 
92  static CURL curl;
93 }

96 // Downloads the header of the specified HTTP URL with a HEAD request
97 // and queries its "content-length" field. (Note that according to the
98 // HTTP specification there is no guarantee that this field contains
99 // any useful value.)
100 inline Try<Bytes> contentLength(const std::string& url)
101 {
102  initialize();
103 
104  CURL* curl = curl_easy_init();
105  if (curl == nullptr) {
106  curl_easy_cleanup(curl);
107  return Error("Failed to initialize libcurl");
108  }
109 
110  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
111  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
112  curl_easy_setopt(curl, CURLOPT_HEADER, 1);
113  curl_easy_setopt(curl, CURLOPT_NOBODY, 1);
114 
115  CURLcode curlErrorCode = curl_easy_perform(curl);
116  if (curlErrorCode != 0) {
117  curl_easy_cleanup(curl);
118  return Error(curl_easy_strerror(curlErrorCode));
119  }
120 
121  double result;
122  curl_easy_getinfo(curl, CURLINFO_CONTENT_LENGTH_DOWNLOAD, &result);
123 
124  curl_easy_cleanup(curl);
125 
126  if (result < 0) {
127  return Error("No URL content-length available");
128  }
129 
130  return Bytes(uint64_t(result));
131 }

134 // Returns the HTTP response code resulting from attempting to
135 // download the specified HTTP or FTP URL into a file at the specified
136 // path.
137 inline Try<int> download(const std::string& url, const std::string& path)
138 {
139  initialize();
140 
141  Try<int> fd = os::open(
142  path,
143  O_CREAT | O_WRONLY | O_CLOEXEC,
144  S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
145 
146  if (fd.isError()) {
147  return Error(fd.error());
148  }
149 
150  CURL* curl = curl_easy_init();
151 
152  if (curl == nullptr) {
153  curl_easy_cleanup(curl);
154  os::close(fd.get());
155  return Error("Failed to initialize libcurl");
156  }
157 
158  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
159  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, nullptr);
160  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
161 
162  FILE* file = fdopen(fd.get(), "w");
163  if (file == nullptr) {
164  curl_easy_cleanup(curl);
165  os::close(fd.get());
166  return ErrnoError("Failed to open file handle of '" + path + "'");
167  }
168  curl_easy_setopt(curl, CURLOPT_WRITEDATA, file);
169 
170  CURLcode curlErrorCode = curl_easy_perform(curl);
171  if (curlErrorCode != 0) {
172  curl_easy_cleanup(curl);
173  fclose(file);
174  return Error(curl_easy_strerror(curlErrorCode));
175  }
176 
177  long code;
178  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
179  curl_easy_cleanup(curl);
180 
181  if (fclose(file) != 0) {
182  return ErrnoError("Failed to close file handle of '" + path + "'");
183  }
184 
185  return Try<int>::some(code);
186 }

203 {
204  struct sockaddr_storage storage;
205  memset(&storage, 0, sizeof(storage));

207  switch (ip.family()) {

227 inline Try<std::string> hostname()
228 {
229  char host[512];
230 
231  if (gethostname(host, sizeof(host)) < 0) {
232  return ErrnoError();
233  }
234 
235  // TODO(evelinad): Add AF_UNSPEC when we will support IPv6.
236  struct addrinfo hints = createAddrInfo(SOCK_STREAM, AF_INET, AI_CANONNAME);
237  struct addrinfo* result = nullptr;
238 
239  int error = getaddrinfo(host, nullptr, &hints, &result);
240

243  }
244 
245  std::string hostname = result->ai_canonname;
246  freeaddrinfo(result);

281 // Returns the names of all the link devices in the system.
282 inline Try<std::set<std::string>> links()
283 {

285  return Error("Not implemented");

294  if (ifa->ifa_name != nullptr) {

305 // Returns a Try of the IP for the provided hostname or an error if no IP is
306 // obtained.
307 inline Try<IP> getIP(const std::string& hostname, int family)
308 {
309  struct addrinfo hints = createAddrInfo(SOCK_STREAM, family, 0);
310  struct addrinfo* result = nullptr;
311 
312  int error = getaddrinfo(hostname.c_str(), nullptr, &hints, &result);
313 
314  if (error != 0) {
315  return Error(gai_strerror(error));
316  }
317 
318  if (result->ai_addr == nullptr) {
319  freeaddrinfo(result);
320  return Error("No addresses found");
321  }
322 
323  Try<IP> ip = IP::create(*result->ai_addr);
324 
325  if (ip.isError()) {
326  freeaddrinfo(result);
327  return Error("Unsupported family type");
328  }
329 
330  freeaddrinfo(result);
331  return ip.get();
332 }
