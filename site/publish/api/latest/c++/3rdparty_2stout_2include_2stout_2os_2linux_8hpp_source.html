52 inline pid_t clone(const lambda::function<int()>& func, int flags)
53 {
54  // Stack for the child.
55  // - unsigned long long used for best alignment.
56  // - 8 MiB appears to be the default for "ulimit -s" on OSX and Linux.
57  //
58  // NOTE: We need to allocate the stack dynamically. This is because
59  // glibc's 'clone' will modify the stack passed to it, therefore the
60  // stack must NOT be shared as multiple 'clone's can be invoked
61  // simultaneously.
62  int stackSize = 8 * 1024 * 1024;
63  unsigned long long *stack =
64  new unsigned long long[stackSize/sizeof(unsigned long long)];
65 
66  pid_t pid = ::clone(
67  childMain,
68  &stack[stackSize/sizeof(stack[0]) - 1], // stack grows down.
69  flags,
70  (void*) &func);
71 
72  // If CLONE_VM is not set, ::clone would create a process which runs in a
73  // separate copy of the memory space of the calling process. So we destroy the
74  // stack here to avoid memory leak. If CLONE_VM is set, ::clone would create a
75  // thread which runs in the same memory space with the calling process.
76  if (!(flags & CLONE_VM)) {
77  delete[] stack;
78  }
79 
80  return pid;
81 }

84 inline Result<Process> process(pid_t pid)
85 {
86  // Page size, used for memory accounting.
87  static const int pageSize = os::pagesize();
88  if (pageSize <= 0) {
89  return Error("Failed to get `os::pagesize`");
90  }
91 
92  // Number of clock ticks per second, used for cpu accounting.
93  static const long ticks = sysconf(_SC_CLK_TCK);
94  if (ticks <= 0) {
95  return Error("Failed to get sysconf(_SC_CLK_TCK)");
96  }
97 
98  const Result<proc::ProcessStatus> status = proc::status(pid);
99 
100  if (status.isError()) {
101  return Error(status.error());
102  }
103 
104  if (status.isNone()) {
105  return None();
106  }
107 
108  // There are known bugs with invalid utime / stime values coming
109  // from /proc/<pid>/stat on some Linux systems.
110  // See the following thread for details:
111  // http://mail-archives.apache.org/mod_mbox/incubator-mesos-dev/
112  // 201307.mbox/%3CCA+2n2er-Nemh0CsKLbHRkaHd=YCrNt17NLUPM2=TtEfsKOw4
113  // Rg@mail.gmail.com%3E
114  // These are similar reports:
115  // http://lkml.indiana.edu/hypermail/linux/kernel/1207.1/01388.html
116  // https://bugs.launchpad.net/ubuntu/+source/linux/+bug/1023214
117  Try<Duration> utime = Duration::create(status.get().utime / (double) ticks);
118  Try<Duration> stime = Duration::create(status.get().stime / (double) ticks);
119 
120  // The command line from 'status.get().comm' is only "arg0" from
121  // "argv" (i.e., the canonical executable name). To get the entire
122  // command line we grab '/proc/[pid]/cmdline'.
123  Result<std::string> cmdline = proc::cmdline(pid);
124 
125  return Process(status.get().pid,
126  status.get().ppid,
127  status.get().pgrp,
128  status.get().session,
129  Bytes(status.get().rss * pageSize),
130  utime.isSome() ? utime.get() : Option<Duration>::none(),
131  stime.isSome() ? stime.get() : Option<Duration>::none(),
132  cmdline.isSome() ? cmdline.get() : status.get().comm,
133  status.get().state == 'Z');
134 }

143 // Returns the total size of main and free memory.
144 inline Try<Memory> memory()
145 {
146  Memory memory;
147 
148  struct sysinfo info;
149  if (sysinfo(&info) != 0) {
150  return ErrnoError();
151  }
152 
153 # if LINUX_VERSION_CODE >= KERNEL_VERSION(2, 3, 23)
154  memory.total = Bytes(info.totalram * info.mem_unit);
155  memory.free = Bytes(info.freeram * info.mem_unit);
156  memory.totalSwap = Bytes(info.totalswap * info.mem_unit);
157  memory.freeSwap = Bytes(info.freeswap * info.mem_unit);
158 # else
159  memory.total = Bytes(info.totalram);
160  memory.free = Bytes(info.freeram);
161  memory.totalSwap = Bytes(info.totalswap);
162  memory.freeSwap = Bytes(info.freeswap);
163 # endif
164 
165  return memory;
166 }
