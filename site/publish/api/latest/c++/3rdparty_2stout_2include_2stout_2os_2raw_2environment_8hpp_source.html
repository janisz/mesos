24 // NOTE: the `os::raw` namespace contains a family of simple wrapper functions
25 // for getting environment data from either Windows or Unix machines. For
26 // example, `os::raw::environment` returns an "unstructured" `char**` that
27 // contains the raw environment variables of the executing process. Accessing
28 // "structured" version of this function, `os::environment`, returns a
29 // `map<string, string>` instead. This family of functions exists in the
30 // `os::raw` namespace because of the unstructured nature of their return
31 // values.
32 //
33 // WARNING: these functions are called `environment` and not `environ` because
34 // on Windows, `environ` is a macro, and not an `extern char**` as it is in the
35 // POSIX standard. The existance of this macro on Windows makes it impossible
36 // to use a function called `os::environ`.
37 namespace os {
38 namespace raw {

40 inline char** environment()
41 {
42  // Accessing the list of environment variables is platform-specific.
43  // On OS X, the 'environ' symbol isn't visible to shared libraries,
44  // so we must use the _NSGetEnviron() function (see 'man environ' on
45  // OS X). On other platforms, it's fine to access 'environ' from
46  // shared libraries.
47 #ifdef __APPLE__
48  return *_NSGetEnviron();
49 #else
50  // NOTE: the correct style for this expression would be `::environ`, but we
51  // leave it out because `environ` is a macro on Windows, and the `::` will
52  // break the build.
53  return environ;
54 #endif
55 }
