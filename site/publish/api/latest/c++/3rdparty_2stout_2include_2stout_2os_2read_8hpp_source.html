45 // Reads 'size' bytes from a file from its current offset.
46 // If EOF is encountered before reading 'size' bytes then the result
47 // will contain the bytes read and a subsequent read will return None.
48 inline Result<std::string> read(int fd, size_t size)
49 {
50  char* buffer = new char[size];
51  size_t offset = 0;
52 
53  while (offset < size) {
54  ssize_t length = os::read(fd, buffer + offset, size - offset);
55 
56 #ifdef __WINDOWS__
57  int error = WSAGetLastError();
58 #else
59  int error = errno;
60 #endif // __WINDOWS__
61 
62  if (length < 0) {
63  // TODO(bmahler): Handle a non-blocking fd? (EAGAIN, EWOULDBLOCK)
64  if (net::is_restartable_error(error)) {
65  continue;
66  }
67  ErrnoError error; // Constructed before 'delete' to capture errno.
68  delete[] buffer;
69  return error;
70  } else if (length == 0) {
71  // Reached EOF before expected! Only return as much data as
72  // available or None if we haven't read anything yet.
73  if (offset > 0) {
74  std::string result(buffer, offset);
75  delete[] buffer;
76  return result;
77  }
78  delete[] buffer;
79  return None();
80  }
81 
82  offset += length;
83  }
84 
85  std::string result(buffer, size);
86  delete[] buffer;
87  return result;
88 }

91 // Returns the contents of the file. NOTE: getline is not available on Solaris
92 // or Windows, so we use STL.
93 #if defined(__sun) || defined(__WINDOWS__)
94 inline Try<std::string> read(const std::string& path)
95 {
96  std::ifstream file(path.c_str());
97  if (!file.is_open()) {
98  // Does ifstream actually set errno?
99  return ErrnoError("Failed to open file");
100  }
101  return std::string((std::istreambuf_iterator<char>(file)),
102  (std::istreambuf_iterator<char>()));
103 }
104 #else
105 inline Try<std::string> read(const std::string& path)
106 {
107  FILE* file = ::fopen(path.c_str(), "r");
108  if (file == nullptr) {
109  return ErrnoError("Failed to open file");
110  }
111 
112  // Use a buffer to read the file in BUFSIZ
113  // chunks and append it to the string we return.
114  //
115  // NOTE: We aren't able to use fseek() / ftell() here
116  // to find the file size because these functions don't
117  // work properly for in-memory files like /proc/*/stat.
118  char* buffer = new char[BUFSIZ];
119  std::string result;
120 
121  while (true) {
122  size_t read = ::fread(buffer, 1, BUFSIZ, file);
123 
124  if (::ferror(file)) {
125  // NOTE: ferror() will not modify errno if the stream
126  // is valid, which is the case here since it is open.
127  ErrnoError error;
128  delete[] buffer;
129  ::fclose(file);
130  return error;
131  }
132 
133  result.append(buffer, read);
134 
135  if (read != BUFSIZ) {
136  assert(feof(file));
137  break;
138  }
139  };
140 
141  ::fclose(file);
142  delete[] buffer;
143  return result;
144 }
