31 // Returns the total disk size in bytes.
32 inline Try<Bytes> size(const std::string& path = "/")
33 {
34  struct statvfs buf;
35  if (::statvfs(path.c_str(), &buf) < 0) {
36  return ErrnoError();
37  }
38  return Bytes(buf.f_blocks * buf.f_frsize);
39 }

42 // Returns relative disk usage of the file system that the given path
43 // is mounted at.
44 inline Try<double> usage(const std::string& path = "/")
45 {
46  struct statvfs buf;
47  if (statvfs(path.c_str(), &buf) < 0) {
48  return ErrnoError("Error invoking statvfs on '" + path + "'");
49  }
50  return (double) (buf.f_blocks - buf.f_bfree) / buf.f_blocks;
51 }

65 // Returns a list of all files matching the given pattern. On POSIX builds this
66 // is just a wrapper on os::glob().
67 //
68 // This function was added is because glob() is not available on Windows, and we
69 // are not making use of any of the heavyweight features of ::glob() anyway.
70 //
71 // list() is meant to be a lightweight alternative to glob() - the only
72 // wildcards it should be used with are `?` and `*`, and only when they appear
73 // at the tail end of `pattern` (e.g. `/root/dir/subdir/*.txt` or
74 // `/root/dir/subdir/file?.txt`.
75 inline Try<std::list<std::string>> list(const std::string& pattern)
76 {
77  return os::glob(pattern);
78 }
