85 // Returns a list of all files matching the given pattern. This is meant to
86 // be a lightweight alternative to glob() - the only supported wildcards are
87 // `?` and `*`, and only when they appear at the tail end of `pattern` (e.g.
88 // `/root/dir/subdir/*.txt` or `/root/dir/subdir/file?.txt`.
89 inline Try<std::list<std::string>> list(const std::string& pattern)
90 {
91  std::list<std::string> found_files;
92  WIN32_FIND_DATA find_data;
93  const HANDLE search_handle = ::FindFirstFile(pattern.c_str(), &find_data);
94 
95  if (search_handle == INVALID_HANDLE_VALUE) {
96  // For compliance with the POSIX implementation (which uses `::glob`),
97  // return an empty list instead of an error when the path does not exist.
98  int error = ::GetLastError();
99  if (error == ERROR_FILE_NOT_FOUND || error == ERROR_PATH_NOT_FOUND) {
100  return found_files;
101  }
102 
103  return WindowsError(
104  "'fs::list' failed when searching for files with pattern '" +
105  pattern + "'");
106  }
107 
108  do {
109  const std::string current_file(find_data.cFileName);
110 
111  // Ignore `.` and `..` entries
112  if (current_file.compare(".") != 0 && current_file.compare("..") != 0) {
113  found_files.push_back(current_file);
114  }
115  } while (::FindNextFile(search_handle, &find_data));
116 
117  // Cache `FindNextFile` error, `FindClose` will overwrite it
118  const DWORD error = ::GetLastError();
119  ::FindClose(search_handle);
120 
121  if (error != ERROR_NO_MORE_FILES) {
122  ::SetLastError(error);
123  return WindowsError(
124  "'fs::list': 'FindNextFile' failed when searching for files with "
125  "'pattern '" + pattern + "'");
126  }
127 
128  return found_files;
129 }
