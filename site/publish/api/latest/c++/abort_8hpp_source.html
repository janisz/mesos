40 inline NORETURN void _Abort(const char* prefix, const char* message)
41 {
42  // Write the failure message in an async-signal safe manner,
43  // assuming strlen is async-signal safe or optimized out.
44  // In fact, it is highly unlikely that strlen would be
45  // implemented in an unsafe manner:
46  // http://austingroupbugs.net/view.php?id=692
47  while (::write(STDERR_FILENO, prefix, strlen(prefix)) == -1 &&
48  errno == EINTR);
49  while (message != nullptr &&
50  ::write(STDERR_FILENO, message, strlen(message)) == -1 &&
51  errno == EINTR);
52 
53  // NOTE: Since `1` can be interpreted as either an `unsigned int` or a
54  // `size_t`, removing the `static_cast` here makes this call ambiguous
55  // between the `write` in windows.hpp and the (deprecated) `write` in the
56  // Windows CRT headers.
57  while (::write(STDERR_FILENO, "\n", static_cast<size_t>(1)) == -1 &&
58  errno == EINTR);
59  abort();
60 }
