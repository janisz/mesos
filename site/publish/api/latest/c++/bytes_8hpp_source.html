33  static Try<Bytes> parse(const std::string& s)
34  {
35  size_t index = 0;
36 
37  while (index < s.size()) {
38  if (isdigit(s[index])) {
39  index++;
40  continue;
41  } else if (s[index] == '.') {
42  return Error("Fractional bytes '" + s + "'");
43  }
44 
45  Try<uint64_t> value = numify<uint64_t>(s.substr(0, index));
46 
47  if (value.isError()) {
48  return Error(value.error());
49  }
50 
51  const std::string unit = strings::upper(s.substr(index));
52 
53  if (unit == "B") {
54  return Bytes(value.get(), BYTES);
55  } else if (unit == "KB") {
56  return Bytes(value.get(), KILOBYTES);
57  } else if (unit == "MB") {
58  return Bytes(value.get(), MEGABYTES);
59  } else if (unit == "GB") {
60  return Bytes(value.get(), GIGABYTES);
61  } else if (unit == "TB") {
62  return Bytes(value.get(), TERABYTES);
63  } else {
64  return Error("Unknown bytes unit '" + unit + "'");
65  }
66  }
67  return Error("Invalid bytes '" + s + "'");
68  }
