87 private:
88  void discarded()

93  future.discard();
94  }
95 
96  terminate(this);
97  }
98 
99  void waited(const Future<T>& future)
100  {
101  if (future.isFailed()) {
102  promise->fail("Collect failed: " + future.failure());
103  terminate(this);
104  } else if (future.isDiscarded()) {
105  promise->fail("Collect failed: future discarded");
106  terminate(this);
107  } else {
108  CHECK_READY(future);
109  ready += 1;
110  if (ready == futures.size()) {
111  std::list<T> values;
112  foreach (const Future<T>& future, futures) {
113  values.push_back(future.get());
114  }
115  promise->set(values);
116  terminate(this);
117  }
118  }
119  }
120 
121  const std::list<Future<T>> futures;
122  Promise<std::list<T>>* promise;
123  size_t ready;
124 };

185 template <typename T>
186 inline Future<std::list<T>> collect(
187  const std::list<Future<T>>& futures)
188 {
189  if (futures.empty()) {
190  return std::list<T>();
191  }
192 
193  Promise<std::list<T>>* promise = new Promise<std::list<T>>();
194  Future<std::list<T>> future = promise->future();
195  spawn(new internal::CollectProcess<T>(futures, promise), true);
196  return future;
197 }

200 template <typename... Ts>
201 Future<std::tuple<Ts...>> collect(const Future<Ts>&... futures)
202 {
203  std::list<Future<Nothing>> wrappers = {
204  futures.then([]() { return Nothing(); })...
205  };
206 
207  // TODO(klueska): Unfortunately, we have to use a lambda followed
208  // by a bind here because of a bug in gcc 4.8 to handle variadic
209  // parameters in lambdas:
210  // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=47226
211  auto f = [](const Future<Ts>&... futures) {
212  return std::make_tuple(futures.get()...);
213  };
214 
215  return collect(wrappers)
