54 // We need an intermediate "deferred" type because when constructing a
55 // Deferred we won't always know the underlying function type (for
56 // example, if we're being passed a std::bind or a lambda). A lambda
57 // won't always implicitly convert to a std::function so instead we
58 // hold onto the functor type F and let the compiler invoke the
59 // necessary cast operator (below) when it actually has determined
60 // what type is needed. This is similar in nature to how std::bind
61 // works with it's intermediate _Bind type (which the pre-C++11
62 // implementation relied on).
63 template <typename F>
64 struct _Deferred
65 {
66  operator Deferred<void()>() const
67  {
68  // The 'pid' differentiates an already dispatched functor versus
69  // one which still needs to be dispatched (which is done
70  // below). We have to delay wrapping the dispatch (for example, in
71  // defer.hpp) as long as possible because we don't always know
72  // what type the functor is or is going to be cast to (i.e., a
73  // std::bind might can be cast to functions that do or do not take
74  // arguments which will just be dropped when invoking the
75  // underlying bound function).
76  if (pid.isNone()) {
77  return std::function<void()>(f);
78  }
79 
80  // We need to explicitly copy the members otherwise we'll
81  // implicitly copy 'this' which might not exist at invocation.
82  Option<UPID> pid_ = pid;
83  F f_ = f;
84 
85  return std::function<void()>(
86  [=]() {
87  dispatch(pid_.get(), std::function<void()>(f_));
88  });
