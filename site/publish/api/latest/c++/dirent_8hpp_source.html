57 // Windows implementation of POSIX standard `opendir`. cf. specification[1].
58 //
59 // [1] http://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
60 inline DIR* opendir(const char* path)
61 {
62  if (path == nullptr) {
63  errno = ENOTDIR;
64  return nullptr;
65  }
66 
67  const size_t path_size = strlen(path);
68 
69  if (path_size == 0 || path_size >= MAX_PATH) {
70  errno = ENOENT;
71  return nullptr;
72  }
73 
74  const char windows_folder_separator = '\\';
75  const char windows_drive_separator = ':';
76  const char wildcard[] = "*";
77  const char dir_separator_and_wildcard[] = "\\*";
78 
79  // Allocate space for directory. Be sure to leave room at the end of
80  // `directory->d_name` for a directory separator and a wildcard.
81  DIR* directory = (DIR*) malloc(sizeof(DIR));
82 
83  if (directory == nullptr) {
84  errno = ENOMEM;
85  return nullptr;
86  }
87 
88  directory->d_name =
89  (char*) malloc(path_size + strlen(dir_separator_and_wildcard) + 1);
90 
91  if (directory->d_name == nullptr) {
92  errno = ENOMEM;
93  free(directory);
94  return nullptr;
95  }
96 
97  // Copy path over and append the appropriate postfix.
98  strcpy(directory->d_name, path);
99 
100  const size_t last_char_in_name =
101  directory->d_name[strlen(directory->d_name) - 1];
102 
103  if (last_char_in_name != windows_folder_separator &&
104  last_char_in_name != windows_drive_separator) {
105  strcat(directory->d_name, dir_separator_and_wildcard);
106  } else {
107  strcat(directory->d_name, wildcard);
108  }
109 
110  if (!internal::open_dir_stream(directory)) {
111  internal::free_dir(directory);
112  return nullptr;
113  }
114 
115  return directory;
116 }

119 // Implementation of the standard POSIX function. See documentation[1].
120 //
121 // On success: returns a pointer to the next directory entry, or `nullptr` if
122 // we've reached the end of the stream.
123 //
124 // On failure: returns `nullptr` and sets `errno`.
125 //
126 // NOTE: as with most POSIX implementations of this function, you must reset
127 // `errno` before calling `readdir`.
128 //
129 // [1] http://www.gnu.org/software/libc/manual/html_node/Reading_002fClosing-Directory.html#Reading_002fClosing-Directory
130 inline struct dirent* readdir(DIR* directory)
131 {
132  if (directory == nullptr) {
133  errno = EBADF;
134  return nullptr;
135  }
136 
137  if (!internal::reentrant_advance_dir_stream(directory)) {
138  return nullptr;
139  }
140 
141  return &directory->curr;
142 }

145 // Implementation of the standard POSIX function. See documentation[1].
146 //
147 // On success: return 0 and set `*result` (note that `result` is not the same
148 // as `*result`) to point at the next directory entry, or `nullptr` if we've
149 // reached the end of the stream.
150 //
151 // On failure: return a positive error number and set `*result` (not `result`)
152 // to point at `nullptr`.
153 //
154 // [1] https://www.gnu.org/software/libc/manual/html_node/Reading_002fClosing-Directory.html
155 inline int readdir_r(
156  DIR* directory,
157  struct dirent* entry,
158  struct dirent** result)
159 {
160  if (directory == nullptr) {
161  errno = EBADF;
162  *result = nullptr;
163  return 1;
164  }
165 
166  if (!internal::reentrant_advance_dir_stream(directory)) {
167  *result = nullptr;
168  return 0;
169  }
170 
171  memcpy(entry, &directory->curr, sizeof(*entry));
172  *result = &directory->curr;
173 
174  return 0;
175 }

178 // Implementation of the standard POSIX function. See documentation[1].
179 //
180 // On success, return 0; on failure, return -1 and set `errno` appropriately.
181 //
182 // [1] http://www.gnu.org/software/libc/manual/html_node/Reading_002fClosing-Directory.html#Reading_002fClosing-Directory
183 inline int closedir(DIR* directory)
184 {
185  if (directory == nullptr) {
186  errno = EBADF;
187  return -1;
188  }
189 
190  BOOL search_closed = false;
191 
192  if (directory->handle != INVALID_HANDLE_VALUE) {
193  search_closed = FindClose(directory->handle);
194  }
195 
196  internal::free_dir(directory);
197 
198  return search_closed ? 0 : -1;
199 }

213 inline bool open_dir_stream(DIR* directory)
214 {
215  assert(directory != nullptr);
216 
217  directory->handle = FindFirstFile(directory->d_name, &directory->fd);
218 
219  if (directory->handle == INVALID_HANDLE_VALUE) {
220  errno = ENOENT;
221  return false;
222  }
223 
224  // NOTE: `d_name` can be a statically-sized array of `MAX_PATH` size because
225  // `cFileName` is. See[1]. This simplifies this copy operation because we
226  // don't have to `malloc`.
227  //
228  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/aa365740(v=vs.85).aspx
229  strcpy(directory->curr.d_name, directory->fd.cFileName);
230  directory->curr.d_namlen = strlen(directory->curr.d_name);
231 
232  return true;
233 }

244  // NOTE: `d_name` can be a statically-sized array of `MAX_PATH` size because
245  // `cFileName` is. See[1]. This simplifies this copy operation because we
246  // don't have to `malloc`.
247  //
248  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/aa365740(v=vs.85).aspx
249  strcpy(directory->curr.d_name, directory->fd.cFileName);
250  directory->curr.d_namlen = strlen(directory->curr.d_name);
251 
252  return true;
253 }
