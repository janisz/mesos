43 // This represents a docker archive. It has the same format
44 // as that tarball generated by doing a 'docker save'.
45 class DockerArchive
46 {
47 public:
48  // Create a docker test image tarball in docker registry directory.
49  // Users can define own entrypoint/cmd as JSON array of JSON string
50  // (e.g., `[\"sh\", \"-c\"]`).
51  static Future<Nothing> create(
52  const std::string& directory,
53  const std::string& name,
54  const std::string& entrypoint = "null",
55  const std::string& cmd = "null")
56  {
57  Try<Nothing> mkdir = os::mkdir(directory, true);
58  if (mkdir.isError()) {
59  return Failure("Failed to create '" + directory + "': " + mkdir.error());
60  }
61 
62  const std::string imagePath = path::join(directory, name);
63 
64  mkdir = os::mkdir(imagePath);
65  if (mkdir.isError()) {
66  return Failure("Failed to create docker test image directory '" +
67  imagePath + "': " + mkdir.error());
68  }
69 
70  const std::string layerId =
71  "815b809d588c80fd6ddf4d6ac244ad1c01ae4cbe0f91cc7480e306671ee9c346";
72 
73  const std::string layerPath = path::join(imagePath, layerId);
74 
75  // Create docker test image `repositories`.
76  JSON::Value repositories = JSON::parse(strings::format(
77  R"~( { "%s": { "latest": "%s" } })~",
78  name,
79  layerId).get()).get();
80 
81  Try<Nothing> write = os::write(
82  path::join(imagePath, "repositories"),
83  stringify(repositories));
84 
85  if (write.isError()) {
86  return Failure("Failed to save docker test image 'repositories': " +
87  write.error());
88  }
89 
90  mkdir = os::mkdir(layerPath);
91  if (mkdir.isError()) {
92  return Failure("Failed to create docker test image layer '" +
93  layerId + "': " + mkdir.error());
94  }
95 
96  JSON::Value manifest = JSON::parse(strings::format(
97  R"~( { "id": "815b809d588c80fd6ddf4d6ac244ad1c01ae4cbe0f91cc7480e306671ee9c346", "created": "2016-03-02T17:16:00.167415955Z", "container": "eb53609036555d26c39bdccfa9850426934bdfde96111d099041689b2251a377", "container_config": { "Hostname": "eb5360903655", "Domainname": "", "User": "", "AttachStdin": false, "AttachStdout": false, "AttachStderr": false, "Tty": false, "OpenStdin": false, "StdinOnce": false, "Env": null, "Cmd": [ "/bin/sh", "-c", "#(nop) ADD file:81ba6f20bdb99e6c13c434a577069860b6656908031162083b1ac9c02c71dd9f in /" ], "Image": "", "Volumes": null, "WorkingDir": "", "Entrypoint": null, "OnBuild": null, "Labels": null }, "docker_version": "1.9.1", "config": { "Hostname": "eb5360903655", "Domainname": "", "User": "", "AttachStdin": false, "AttachStdout": false, "AttachStderr": false, "Tty": false, "OpenStdin": false, "StdinOnce": false, "Env": null, "Cmd": %s, "Image": "", "Volumes": null, "WorkingDir": "", "Entrypoint": %s, "OnBuild": null, "Labels": null }, "architecture": "amd64", "os": "linux" })~",
98  entrypoint,
99  cmd).get()).get();
100 
101  write = os::write(
102  path::join(layerPath, "json"),
103  stringify(manifest));
104 
105  if (write.isError()) {
106  return Failure("Failed to save docker test image layer '" + layerId +
107  "': " + write.error());
108  }
109 
110  const std::string rootfsDir = path::join(layerPath, "layer");
111 
112  mkdir = os::mkdir(rootfsDir);
113  if (mkdir.isError()) {
114  return Failure("Failed to create layer rootfs directory '" +
115  rootfsDir + "': " + mkdir.error());
116  }
117 
118  // Create one linux rootfs for the layer.
119  Try<process::Owned<Rootfs>> rootfs = LinuxRootfs::create(rootfsDir);
120  if (rootfs.isError()) {
121  return Failure("Failed to create docker test image rootfs: " +
122  rootfs.error());
123  }
124 
125  Future<Nothing> tarRootfs = command::tar(
126  Path("."),
127  Path(path::join(layerPath, "layer.tar")),
128  rootfsDir);
129 
130  tarRootfs.await();
131 
132  if (!tarRootfs.isReady()) {
133  return Failure(
134  "Failed to tar root filesystem: " +
135  (tarRootfs.isFailed() ? tarRootfs.failure() : "discarded"));
136  }
137 
138  Try<Nothing> rmdir = os::rmdir(rootfsDir);
139  if (rmdir.isError()) {
140  return Failure("Failed to remove layer rootfs directory: " +
141  rmdir.error());
142  }
143 
144  write = os::write(
145  path::join(layerPath, "VERSION"),
146  "1.0");
147 
148  if (write.isError()) {
149  return Failure("Failed to save layer version: " + write.error());
150  }
151 
152  Future<Nothing> tarImage = command::tar(
153  Path("."),
154  Path(path::join(directory, name + ".tar")),
155  imagePath);
156 
157  tarImage.await();
158 
159  if (!tarImage.isReady()) {
160  return Failure(
161  "Failed to tar docker test image: " +
162  (tarImage.isFailed() ? tarImage.failure() : "discarded"));
163  }
164 
165  rmdir = os::rmdir(imagePath);
166  if (rmdir.isError()) {
167  return Failure("Failed to remove image directory: " +
168  rmdir.error());
169  }
170 
171  return Nothing();
172  }
173 };
