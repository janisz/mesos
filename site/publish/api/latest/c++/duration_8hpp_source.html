35  static Try<Duration> parse(const std::string& s)
36  {
37  // TODO(benh): Support negative durations (i.e., starts with '-').
38  size_t index = 0;
39  while (index < s.size()) {
40  if (isdigit(s[index]) || s[index] == '.') {
41  index++;
42  continue;
43  }
44 
45  Try<double> value = numify<double>(s.substr(0, index));
46 
47  if (value.isError()) {
48  return Error(value.error());
49  }
50 
51  const std::string unit = s.substr(index);
52 
53  if (unit == "ns") {
54  return Duration(value.get(), NANOSECONDS);
55  } else if (unit == "us") {
56  return Duration(value.get(), MICROSECONDS);
57  } else if (unit == "ms") {
58  return Duration(value.get(), MILLISECONDS);
59  } else if (unit == "secs") {
60  return Duration(value.get(), SECONDS);
61  } else if (unit == "mins") {
62  return Duration(value.get(), MINUTES);
63  } else if (unit == "hrs") {
64  return Duration(value.get(), HOURS);
65  } else if (unit == "days") {
66  return Duration(value.get(), DAYS);
67  } else if (unit == "weeks") {
68  return Duration(value.get(), WEEKS);
69  } else {
70  return Error(
71  "Unknown duration unit '" + unit + "'; supported units are"
72  " 'ns', 'us', 'ms', 'secs', 'mins', 'hrs', 'days', and 'weeks'");
73  }
74  }

76  }

309 inline std::ostream& operator<<(std::ostream& stream, const Duration& duration_)
310 {
311  long precision = stream.precision();
312 
313  // Output the duration in full double precision.
314  stream.precision(std::numeric_limits<double>::digits10);
315 
316  // Parse the duration as the sign and the absolute value.
317  Duration duration = duration_;
318  if (duration_ < Duration::zero()) {
319  stream << "-";
320 
321  // Duration::min() may not be representable as a positive Duration.
322  if (duration_ == Duration::min()) {
323  duration = Duration::max();
324  } else {
325  duration = duration_ * -1;
326  }
327  }
328 
329  // First determine which bucket of time unit the duration falls into
330  // then check whether the duration can be represented as a whole
331  // number with this time unit or a smaller one.
332  // e.g. 1.42857142857143weeks falls into the 'Weeks' bucket but
333  // reads better with a smaller unit: '10days'. So we use 'days'
334  // instead of 'weeks' to output the duration.
335  int64_t nanoseconds = duration.ns();
336  if (duration < Microseconds(1)) {
337  stream << duration.ns() << Nanoseconds::units();
338  } else if (duration < Milliseconds(1)) {
339  if (nanoseconds % Duration::MICROSECONDS != 0) {
340  // We can't get a whole number using this unit but we can at
341  // one level down.
342  stream << duration.ns() << Nanoseconds::units();
343  } else {
344  stream << duration.us() << Microseconds::units();
345  }
346  } else if (duration < Seconds(1)) {
347  if (nanoseconds % Duration::MILLISECONDS != 0 &&
348  nanoseconds % Duration::MICROSECONDS == 0) {
349  stream << duration.us() << Microseconds::units();
350  } else {
351  stream << duration.ms() << Milliseconds::units();
352  }
353  } else if (duration < Minutes(1)) {
354  if (nanoseconds % Duration::SECONDS != 0 &&
355  nanoseconds % Duration::MILLISECONDS == 0) {
356  stream << duration.ms() << Milliseconds::units();
357  } else {
358  stream << duration.secs() << Seconds::units();
359  }
360  } else if (duration < Hours(1)) {
361  if (nanoseconds % Duration::MINUTES != 0 &&
362  nanoseconds % Duration::SECONDS == 0) {
363  stream << duration.secs() << Seconds::units();
364  } else {
365  stream << duration.mins() << Minutes::units();
366  }
367  } else if (duration < Days(1)) {
368  if (nanoseconds % Duration::HOURS != 0 &&
369  nanoseconds % Duration::MINUTES == 0) {
370  stream << duration.mins() << Minutes::units();
371  } else {
372  stream << duration.hrs() << Hours::units();
373  }
374  } else if (duration < Weeks(1)) {
375  if (nanoseconds % Duration::DAYS != 0 &&
376  nanoseconds % Duration::HOURS == 0) {
377  stream << duration.hrs() << Hours::units();
378  } else {
379  stream << duration.days() << Days::units();
380  }
381  } else {
382  if (nanoseconds % Duration::WEEKS != 0 &&
383  nanoseconds % Duration::DAYS == 0) {
384  stream << duration.days() << Days::units();
385  } else {
386  stream << duration.weeks() << Weeks::units();
387  }
388  }
389 
390  // Return the stream to original formatting state.
391  stream.precision(precision);
392 
393  return stream;
394 }
