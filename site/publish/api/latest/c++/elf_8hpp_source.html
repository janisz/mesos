56 // This class is used to represent the contents of a standard ELF
57 // binary. The current implementation is incomplete.
58 //
59 // TODO(klueska): For now, we only include functionality to extract
60 // the SONAME from shared libraries and read their external library
61 // dependencies. In the future, we will expand this class to include
62 // helpers for full access to an ELF file's contents.
63 class File {
64 public:
65  // Open an ELF binary. We do some preliminary parsing as part of
66  // opening the ELF to get quick references to all of its internal
67  // sections for all future calls.
68  //
69  // TODO(klueska): Consider adding a 'stout::Owned' type to avoid
70  // returning a raw pointer here.
71  static Try<File*> open(const std::string& path)
72  {
73  // Set the elf library operating version.
74  if (elf_version(EV_CURRENT) == EV_NONE) {
75  return Error("Failed to set ELF library version: " +
76  stringify(elf_errmsg(-1)));
77  }
78 
79  File* file = new File();
80 
81  file->fd = ::open(path.c_str(), O_RDONLY);
82  if (file->fd < 0) {
83  delete file;
84  return ErrnoError("Failed to open file");
85  }
86 
87  file->elf = elf_begin(file->fd, ELF_C_READ, nullptr);
88  if (file->elf == nullptr) {
89  delete file;
90  return Error("elf_begin() failed: " + stringify(elf_errmsg(-1)));
91  }
92 
93  if (elf_kind(file->elf) != ELF_K_ELF) {
94  delete file;
95  return Error("File is not an ELF binary");
96  }
97 
98  // Create the mapping from section type to section locations.
99  Elf_Scn* section = nullptr;
100  while ((section = elf_nextscn(file->elf, section)) != nullptr) {
101  GElf_Shdr section_header;
102  if (gelf_getshdr(section, &section_header) == nullptr) {
103  delete file;
104  return Error("gelf_getshdr() failed: " + stringify(elf_errmsg(-1)));
105  }
106 
107  SectionType section_type = (SectionType) section_header.sh_type;
108  switch (section_type) {
109  case SectionType::DYNAMIC:
110  file->sections[section_type].push_back(section);
111  }
112  }
113 
114  return file;
115  }

118  {
119  if (elf != nullptr) {
120  elf_end(elf);
121  elf = nullptr;
122  }
123 
124  if (fd >= 0) {
125  ::close(fd);
126  fd = -1;
127  }
128  }

148  // Extract the strings associated with the provided `DynamicTag`
149  // from the DYNAMIC section of the ELF binary.
150  Try<std::vector<std::string>> GetDynamicStrings(DynamicTag tag) const
151  {
152  if (sections.count(SectionType::DYNAMIC) == 0) {
153  return Error("No DYNAMIC sections found in ELF");
154  }
155 
156  if (sections.count(SectionType::DYNAMIC) != 1) {
157  return Error("Multiple DYNAMIC sections found in ELF");
158  }
159 
160  Elf_Scn* dynamic_section = sections.at(SectionType::DYNAMIC)[0];
161 
162  // Walk through the entries in the dynamic section and look for
163  // entries with the provided tag. These entries contain offsets to
164  // strings in the dynamic section's string table. Consequently, we
165  // also have to look for an entry containing a pointer to the
166  // dynamic section's string table so we can resolve the strings
167  // associated with the provided tag later on.
168  Elf_Data* dynamic_data = elf_getdata(dynamic_section, nullptr);
169  if (dynamic_data == nullptr) {
170  return Error("elf_getdata() failed: " + stringify(elf_errmsg(-1)));
171  }
172 
173  Option<uintptr_t> strtab_pointer = None();
174  std::vector<uintptr_t> strtab_offsets;
175 
176  for (size_t i = 0; i < dynamic_data->d_size / sizeof(GElf_Dyn); i++) {
177  GElf_Dyn entry;
178  if (gelf_getdyn(dynamic_data, i, &entry) == nullptr) {
179  return Error("gelf_getdyn() failed: " + stringify(elf_errmsg(-1)));
180  }
181 
182  if ((DynamicTag)entry.d_tag == DynamicTag::STRTAB) {
183  strtab_pointer = entry.d_un.d_ptr;
184  }
185 
186  if ((DynamicTag)entry.d_tag == tag) {
187  strtab_offsets.push_back(entry.d_un.d_ptr);
188  }
189  }
190 
191  if (strtab_offsets.empty()) {
192  return std::vector<std::string>();
193  }
194 
195  if (strtab_pointer.isNone()) {
196  return Error("Failed to find string table");
197  }
198 
199  // Get a reference to the actual string table so we can index into it.
200  Elf_Scn* string_table_section = gelf_offscn(elf, strtab_pointer.get());
201  if (string_table_section == nullptr) {
202  return Error("gelf_offscn() failed: " + stringify(elf_errmsg(-1)));
203  }
204 
205  size_t strtab_index = elf_ndxscn(string_table_section);
206  if (strtab_index == SHN_UNDEF) {
207  return Error("elf_ndxscn() failed: " + stringify(elf_errmsg(-1)));
208  }
209 
210  // Find the strings in the string table from their offsets and return them.
211  std::vector<std::string> strings;
212  foreach (uintptr_t offset, strtab_offsets) {
213  char* string = elf_strptr(elf, strtab_index, offset);
214  if (string == nullptr) {
215  return Error("elf_strptr() failed: " + stringify(elf_errmsg(-1)));
216  }
217 
218  strings.push_back(string);
219  }
220 
221  return strings;
222  }
