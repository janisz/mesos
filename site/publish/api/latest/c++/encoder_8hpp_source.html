120  static std::string encode(Message* message)
121  {
122  std::ostringstream out;
123 
124  if (message != nullptr) {
125  out << "POST ";
126  // Nothing keeps the 'id' component of a PID from being an empty
127  // string which would create a malformed path that has two
128  // '//' unless we check for it explicitly.
129  // TODO(benh): Make the 'id' part of a PID optional so when it's
130  // missing it's clear that we're simply addressing an ip:port.
131  if (message->to.id != "") {
132  out << "/" << message->to.id;
133  }
134 
135  out << "/" << message->name << " HTTP/1.1\r\n"
136  << "User-Agent: libprocess/" << message->from << "\r\n"
137  << "Libprocess-From: " << message->from << "\r\n"
138  << "Connection: Keep-Alive\r\n"
139  << "Host: \r\n";
140 
141  if (message->body.size() > 0) {
142  out << "Transfer-Encoding: chunked\r\n\r\n"
143  << std::hex << message->body.size() << "\r\n";
144  out.write(message->body.data(), message->body.size());
145  out << "\r\n"
146  << "0\r\n"
147  << "\r\n";
148  } else {
149  out << "\r\n";
150  }
151  }
152 
153  return out.str();
154  }
155 
156 private:
157  Message* message;
158 };

170  static std::string encode(
171  const http::Response& response,
172  const http::Request& request)
173  {
174  std::ostringstream out;
175 
176  // TODO(benh): Check version?
177 
178  out << "HTTP/1.1 " << response.status << "\r\n";
179 
180  auto headers = response.headers;
181 
182  // HTTP 1.1 requires the "Date" header. In the future once we
183  // start checking the version (above) then we can conditionally
184  // add this header, but for now, we always do.
185  time_t rawtime;
186  time(&rawtime);
187 
188  char date[256];
189 
190  tm tm_;
191  PCHECK(os::gmtime_r(&rawtime, &tm_) != nullptr)
192  << "Failed to convert the current time to a tm struct "
193  << "using os::gmtime_r()";
194 
195  // TODO(benh): Check return code of strftime!
196  strftime(date, 256, "%a, %d %b %Y %H:%M:%S GMT", &tm_);
197 
198  headers["Date"] = date;
199 
200  // Should we compress this response?
201  std::string body = response.body;
202 
203  if (response.type == http::Response::BODY &&
204  response.body.length() >= GZIP_MINIMUM_BODY_LENGTH &&
205  !headers.contains("Content-Encoding") &&
206  request.acceptsEncoding("gzip")) {
207  Try<std::string> compressed = gzip::compress(body);
208  if (compressed.isError()) {
209  LOG(WARNING) << "Failed to gzip response body: " << compressed.error();
210  } else {
211  body = compressed.get();
212  headers["Content-Length"] = stringify(body.length());
213  headers["Content-Encoding"] = "gzip";
214  }
215  }
216 
217  foreachpair (const std::string& key, const std::string& value, headers) {
218  out << key << ": " << value << "\r\n";
219  }
220 
221  // Add a Content-Length header if the response is of type "none"
222  // or "body" and no Content-Length header has been supplied.
223  if (response.type == http::Response::NONE &&
224  !headers.contains("Content-Length")) {
225  out << "Content-Length: 0\r\n";
226  } else if (response.type == http::Response::BODY &&
227  !headers.contains("Content-Length")) {
228  out << "Content-Length: " << body.size() << "\r\n";
229  }
230 
231  // Use a CRLF to mark end of headers.
232  out << "\r\n";
233 
234  // Add the body if necessary.
235  if (response.type == http::Response::BODY) {
236  // If the Content-Length header was supplied, only write as much data
237  // as the length specifies.
238  Result<uint32_t> length = numify<uint32_t>(headers.get("Content-Length"));
239  if (length.isSome() && length.get() <= body.length()) {
240  out.write(body.data(), length.get());
241  } else {
242  out.write(body.data(), body.size());
243  }
244  }
245 
246  return out.str();
247  }
