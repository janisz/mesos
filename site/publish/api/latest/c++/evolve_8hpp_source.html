63 // Helper for repeated field evolving to 'T1' from 'T2'.
64 template <typename T1, typename T2>
65 google::protobuf::RepeatedPtrField<T1> evolve(
66  google::protobuf::RepeatedPtrField<T2> t2s)
67 {
68  google::protobuf::RepeatedPtrField<T1> t1s;
69 
70  foreach (const T2& t2, t2s) {
71  t1s.Add()->CopyFrom(evolve(t2));
72  }
73 
74  return t1s;
75 }

109 // Before the v1 API we had REST endpoints that returned JSON. The JSON was not
110 // specified in any formal way, i.e., there were no protobufs which captured the
111 // structure. As part of the v1 API we introduced the Call/Response protobufs
112 // (see v1/master.proto and v1/agent.proto). This evolve variant transforms a
113 // JSON object that would have been returned from a particular REST endpoint to
114 // a `Response` protobuf suitable for returning from the new v1 API endpoints.
