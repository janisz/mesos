82 // Returns our JSON representation of a file or directory.
83 // The JSON contains all of the information one would find in ls -l.
84 // Example JSON:
85 // {
86 // 'path': '\/some\/file',
87 // 'mode': '-rwxrwxrwx',
88 // 'nlink': 5,
89 // 'uid': 'bmahler',
90 // 'gid': 'employee',
91 // 'size': 4096, // Bytes.
92 // 'mtime': 1348258116, // Unix timestamp.
93 // }
94 inline JSON::Object jsonFileInfo(const std::string& path,
95  const struct stat& s)
96 {
97  JSON::Object file;
98  file.values["path"] = path;
99  file.values["nlink"] = s.st_nlink;
100  file.values["size"] = s.st_size;
101  file.values["mtime"] = s.st_mtime;
102 
103  char filetype;
104  if (S_ISREG(s.st_mode)) {
105  filetype = '-';
106  } else if (S_ISDIR(s.st_mode)) {
107  filetype = 'd';
108  } else if (S_ISCHR(s.st_mode)) {
109  filetype = 'c';
110  } else if (S_ISBLK(s.st_mode)) {
111  filetype = 'b';
112  } else if (S_ISFIFO(s.st_mode)) {
113  filetype = 'p';
114  } else if (S_ISLNK(s.st_mode)) {
115  filetype = 'l';
116  } else if (S_ISSOCK(s.st_mode)) {
117  filetype = 's';
118  } else {
119  filetype = '-';
120  }
121 
122  struct os::Permissions permissions(s.st_mode);
123 
124  file.values["mode"] = strings::format(
125  "%c%c%c%c%c%c%c%c%c%c",
126  filetype,
127  permissions.owner.r ? 'r' : '-',
128  permissions.owner.w ? 'w' : '-',
129  permissions.owner.x ? 'x' : '-',
130  permissions.group.r ? 'r' : '-',
131  permissions.group.w ? 'w' : '-',
132  permissions.group.x ? 'x' : '-',
133  permissions.others.r ? 'r' : '-',
134  permissions.others.w ? 'w' : '-',
135  permissions.others.x ? 'x' : '-').get();
136 
137  // NOTE: `getpwuid` and `getgrgid` return `nullptr` on Windows.
138  passwd* p = getpwuid(s.st_uid);
139  if (p != nullptr) {
140  file.values["uid"] = p->pw_name;
141  } else {
142  file.values["uid"] = stringify(s.st_uid);
143  }
144 
145  struct group* g = getgrgid(s.st_gid);
146  if (g != nullptr) {
147  file.values["gid"] = g->gr_name;
148  } else {
149  file.values["gid"] = stringify(s.st_gid);
150  }
151 
152  return file;
153 }
