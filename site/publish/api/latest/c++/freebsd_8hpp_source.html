30 inline Result<Process> process(pid_t pid)
31 {
32  // KERN_PROC_PID fails for zombies, so we fetch the whole process table and
33  // find our process manually.
34 
35  const Try<std::vector<kinfo_proc>> kinfos =
36  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_ALL).table();
37 
38  if (kinfos.isError()) {
39  return Error("Failed to retrieve process table via sysctl: " +
40  kinfos.error());
41  }
42 
43  foreach (const kinfo_proc& kinfo, kinfos.get()) {
44  if (kinfo.ki_pid == pid) {
45  int pagesize = os::pagesize();
46  return Process(kinfo.ki_pid,
47  kinfo.ki_ppid,
48  kinfo.ki_pgid,
49  kinfo.ki_sid,
50  kinfo.ki_rssize * pagesize,
51  kinfo.ki_rusage.ru_utime,
52  kinfo.ki_rusage.ru_stime,
53  kinfo.ki_comm,
54  kinfo.ki_stat == SZOMB);
55  }
56  }
57 
58  return None();
59 }

76 // Returns the total size of main and free memory.
77 inline Try<Memory> memory()
78 {
79  Memory memory;
80 
81  const Try<int64_t> physicalMemory = os::sysctl(CTL_HW, HW_PHYSMEM).integer();
82  if (physicalMemory.isError()) {
83  return Error(physicalMemory.error());
84  }
85  memory.total = Bytes(physicalMemory.get());
86 
87  const int pageSize = os::pagesize();
88 
89  unsigned int freeCount;
90  size_t length = sizeof(freeCount);
91 
92  if (sysctlbyname(
93  "vm.stats.v_free_count",
94  &freeCount,
95  &length,
96  nullptr,
97  0) != 0) {
98  return ErrnoError();
99  }
100  memory.free = Bytes(freeCount * pageSize);
101 
102  int totalBlocks = 0;
103  int usedBlocks = 0;
104 
105  int mib[3];
106  size_t mibSize = 2;
107  if (::sysctlnametomib("vm.swap_info", mib, &mibSize) != 0) {
108  return ErrnoError();
109  }
110 
111  // FreeBSD supports multiple swap devices. Here we sum across all of them.
112  struct xswdev xswd;
113  size_t xswdSize = sizeof(xswd);
114  int* mibDevice = &(mib[mibSize + 1]);
115  for (*mibDevice = 0; ; (*mibDevice)++) {
116  if (::sysctl(mib, 3, &xswd, &xswdSize, nullptr, 0) != 0) {
117  if (errno == ENOENT) {
118  break;
119  }
120  return ErrnoError();
121  }
122 
123  totalBlocks += xswd.xsw_nblks;
124  usedBlocks += xswd.xsw_used;
125  }
126 
127  memory.totalSwap = Bytes(totalBlocks * pageSize);
128  memory.freeSwap = Bytes((totalBlocks - usedBlocks) * pageSize);
129 
130  return memory;
131 }
