128  // computation but rather acts as a suggestion or indication that

688 template <typename T>
689 bool Promise<T>::associate(const Future<T>& future)
690 {
691  bool associated = false;
692 
693  synchronized (f.data->lock) {
694  // Don't associate if this promise has completed. Note that this
695  // does not include if Future::discard was called on this future
696  // since in that case that would still leave the future PENDING
697  // (note that we cover that case below).
698  if (f.data->state == Future<T>::PENDING && !f.data->associated) {
699  associated = f.data->associated = true;
700 
701  // After this point we don't allow 'f' to be completed via the
702  // promise since we've set 'associated' but Future::discard on
703  // 'f' might get called which will get propagated via the
704  // 'f.onDiscard' below. Note that we currently don't propagate a
705  // discard from 'future.onDiscard' but these semantics might
706  // change if/when we make 'f' and 'future' true aliases of one
707  // another.
708  }
709  }
710 
711  // Note that we do the actual associating after releasing the lock
712  // above to avoid deadlocking by attempting to require the lock
713  // within from invoking 'f.onDiscard' and/or 'f.set/fail' via the
714  // bind statements from doing 'future.onReady/onFailed'.
715  if (associated) {
716  // TODO(jieyu): Make 'f' a true alias of 'future'. Currently, only
717  // 'discard' is associated in both directions. In other words, if
718  // a future gets discarded, the other future will also get
719  // discarded. For 'set' and 'fail', they are associated only in
720  // one direction. In other words, calling 'set' or 'fail' on this
721  // promise will not affect the result of the future that we
722  // associated.
723  f.onDiscard(lambda::bind(&internal::discard<T>, WeakFuture<T>(future)));
724 
725  // Need to disambiguate for the compiler.
726  bool (Future<T>::*set)(const T&) = &Future<T>::set;
727

844 template <typename T>
845 bool Promise<T>::discard(Future<T> future)
846 {
847  std::shared_ptr<typename Future<T>::Data> data = future.data;
848 
849  bool result = false;
850 
851  synchronized (data->lock) {
852  if (data->state == Future<T>::PENDING) {
853  data->state = Future<T>::DISCARDED;
854  result = true;
855  }

863  internal::run(future.data->onAnyCallbacks, future);

983 template <typename T>
984 bool Future<T>::discard()
985 {
986  bool result = false;
987 
988  std::vector<DiscardCallback> callbacks;
989  synchronized (data->lock) {
990  if (!data->discard && data->state == PENDING) {
991  result = data->discard = true;
992 
993  // NOTE: We make a copy of the onDiscard callbacks here
994  // because it is possible that another thread completes this
995  // future (ready, failed or discarded) when the current thread
996  // is out of this critical section but *before* it executed the
997  // onDiscard callbacks. If that happens, the other thread might
998  // be clearing the onDiscard callbacks (via clearAllCallbacks())
999  // while the current thread is executing or clearing the
1000  // onDiscard callbacks, causing thread safety issue.
1001  callbacks = data->onDiscardCallbacks;
1002  data->onDiscardCallbacks.clear();
1003  }
1004  }
1005 
1006  // Invoke all callbacks associated with doing a discard on this
1007  // future. We don't need a lock because 'Data::discard' should now
1008  // be set so we won't be adding anything else to
1009  // 'Data::onDiscardCallbacks'.
1010  if (result) {
1011  internal::run(callbacks);
1012  }
1013 
1014  return result;
1015 }

1063 template <typename T>
1064 bool Future<T>::await(const Duration& duration) const
1065 {
1066  // NOTE: We need to preemptively allocate the Latch on the stack
1067  // instead of lazily create it in the critical section below because
1068  // instantiating a Latch requires creating a new process (at the
1069  // time of writing this comment) which might need to do some
1070  // synchronization in libprocess which might deadlock if some other
1071  // code in libprocess is already holding a lock and then attempts to
1072  // do Promise::set (or something similar) that attempts to acquire
1073  // the lock that we acquire here. This is an artifact of using
1074  // Future/Promise within the implementation of libprocess.
1075  //
1076  // We mostly only call 'await' in tests so this should not be a
1077  // performance concern.
1078  Owned<Latch> latch(new Latch());
1079 
1080  bool pending = false;
1081 
1082  synchronized (data->lock) {
1083  if (data->state == PENDING) {
1084  pending = true;
1085  data->onAnyCallbacks.push_back(lambda::bind(&internal::awaited, latch));
1086  }
1087  }
1088 
1089  if (pending) {
1090  return latch->await(duration);
1091  }
1092 
1093  return true;
1094 }
