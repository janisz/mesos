311  struct Slave
312  {
313  // Total amount of regular *and* oversubscribed resources.
314  Resources total;
315 
316  // Regular *and* oversubscribed resources that are allocated.
317  //
318  // NOTE: We keep track of slave's allocated resources despite
319  // having that information in sorters. This is because the
320  // information in sorters is not accurate if some framework
321  // hasn't reregistered. See MESOS-2919 for details.
322  Resources allocated;
323 
324  // We track the total and allocated resources on the slave, the
325  // available resources are computed as follows:
326  //
327  // available = total - allocated
328  //
329  // Note that it's possible for the slave to be over-allocated!
330  // In this case, allocated > total.
331 
332  bool activated; // Whether to offer resources.
333 
334  std::string hostname;
335 
336  // Represents a scheduled unavailability due to maintenance for a specific
337  // slave, and the responses from frameworks as to whether they will be able
338  // to gracefully handle this unavailability.
339  //
340  // NOTE: We currently implement maintenance in the allocator to be able to
341  // leverage state and features such as the FrameworkSorter and OfferFilter.
342  struct Maintenance
343  {
344  Maintenance(const Unavailability& _unavailability)
345  : unavailability(_unavailability) {}
346 
347  // The start time and optional duration of the event.
348  Unavailability unavailability;
349 
350  // A mapping of frameworks to the inverse offer status associated with
351  // this unavailability.
352  //
353  // NOTE: We currently lose this information during a master fail over
354  // since it is not persisted or replicated. This is ok as the new master's
355  // allocator will send out new inverse offers and re-collect the
356  // information. This is similar to all the outstanding offers from an old
357  // master being invalidated, and new offers being sent out.
358  hashmap<FrameworkID, mesos::master::InverseOfferStatus> statuses;
359 
360  // Represents the "unit of accounting" for maintenance. When a
361  // `FrameworkID` is present in the hashset it means an inverse offer has
362  // been sent out. When it is not present it means no offer is currently
363  // outstanding.
364  hashset<FrameworkID> offersOutstanding;
365  };
366 
367  // When the `maintenance` is set the slave is scheduled to be unavailable at
368  // a given point in time, for an optional duration. This information is used
369  // to send out `InverseOffers`.
370  Option<Maintenance> maintenance;
371  };
372 
373  hashmap<SlaveID, Slave> slaves;
374 
375  // Number of registered frameworks for each role. When a role's active
376  // count drops to zero, it is removed from this map; the role is also
377  // removed from `roleSorter` and its `frameworkSorter` is deleted.
378  hashmap<std::string, size_t> activeRoles;
379 
380  // Configured weight for each role, if any; if a role does not
381  // appear here, it has the default weight of 1.
382  hashmap<std::string, double> weights;
383 
384  // Configured quota for each role, if any. Setting quota for a role
385  // changes the order that the role's frameworks are offered
386  // resources. Quota comes before fair share, hence setting quota moves
387  // the role's frameworks towards the front of the allocation queue.
388  //
389  // NOTE: We currently associate quota with roles, but this may
390  // change in the future.
391  hashmap<std::string, Quota> quotas;
392 
393  // Slaves to send offers for.
394  Option<hashset<std::string>> whitelist;
395 
396  // There are two stages of allocation. During the first stage resources
397  // are allocated only to frameworks in roles with quota set. During the
398  // second stage remaining resources that would not be required to satisfy
399  // un-allocated quota are then allocated to all frameworks.
400  //
401  // Each stage comprises two levels of sorting, hence "hierarchical".
402  // Level 1 sorts across roles:
403  // Currently, only the allocated portion of the reserved resources are
404  // accounted for fairness calculation.
405  //
406  // TODO(mpark): Reserved resources should be accounted for fairness
407  // calculation whether they are allocated or not, since they model a long or
408  // forever running task. That is, the effect of reserving resources is
409  // equivalent to launching a task in that the resources that make up the
410  // reservation are not available to other roles as non-revocable.
411  //
412  // Level 2 sorts across frameworks within a particular role:
413  // Reserved resources at this level are, and should be accounted for
414  // fairness calculation only if they are allocated. This is because
415  // reserved resources are fairly shared across the frameworks in the role.
416  //
417  // The allocator relies on `Sorter`s to employ a particular sorting
418  // algorithm. Each level has its own sorter and hence may have different
419  // fairness calculations.
420  //
421  // NOTE: The hierarchical allocator considers revocable resources as
422  // regular resources when doing fairness calculations.
423  //
424  // TODO(vinod): Consider using a different fairness algorithm for
425  // revocable resources.
426 
427  // A sorter for active roles. This sorter determines the order in which
428  // roles are allocated resources during Level 1 of the second stage.
429  process::Owned<Sorter> roleSorter;
430 
431  // A dedicated sorter for roles for which quota is set. This sorter
432  // determines the order in which quota'ed roles are allocated resources
433  // during Level 1 of the first stage. Quota'ed roles have resources
434  // allocated up to their alloted quota (the first stage) prior to
435  // non-quota'ed roles (the second stage).
436  //
437  // NOTE: A role appears in `quotaRoleSorter` if it has a quota (even if
438  // no frameworks are currently registered in that role). In contrast,
439  // `roleSorter` only contains entries for roles with one or more
440  // registered frameworks.
441  //
442  // NOTE: We do not include revocable resources in the quota role sorter,
443  // because the quota role sorter's job is to perform fair sharing between
444  // the quota roles as it pertains to their level of quota satisfaction.
445  // Since revocable resources do not increase a role's level of satisfaction
446  // toward its quota, we choose to exclude them from the quota role sorter.
447  process::Owned<Sorter> quotaRoleSorter;
448 
449  // A collection of sorters, one per active role. Each sorter determines
450  // the order in which frameworks that belong to the same role are allocated
451  // resources inside the role's share. These sorters are used during Level 2
452  // for both the first and the second stages.
453  hashmap<std::string, process::Owned<Sorter>> frameworkSorters;
454 
455  // Factory functions for sorters.
456  const std::function<Sorter*()> roleSorterFactory;
457  const std::function<Sorter*()> frameworkSorterFactory;
458  const std::function<Sorter*()> quotaRoleSorterFactory;
459 };
