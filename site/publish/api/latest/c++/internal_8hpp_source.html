50 // A helper class for managing netlink objects (e.g., rtnl_link,
51 // nl_sock, etc.). It manages the life cycle of a netlink object. It
52 // is copyable and assignable, and multiple copies share the same
53 // underlying netlink object. A netlink object specific cleanup
54 // function will be invoked when the last copy of this wrapper is
55 // being deleted (similar to Future<T>). We use this class to simplify
56 // our code, especially for error handling.
57 template <typename T>
58 class Netlink
59 {
60 public:
61  explicit Netlink(T* object) : data(new Data(object)) {}
62 
63  T* get() const { return data->object; }
64 
65 private:
66  struct Data
67  {
68  explicit Data(T* _object) : object(_object) {}
69 
70  ~Data()
71  {
72  if (object != nullptr) {
73  cleanup(object);
74  }
75  }
76 
77  T* object;
78  };
79 
80  std::shared_ptr<Data> data;
81 };

84 // Returns a netlink socket for communicating with the kernel. This
85 // socket is needed for most of the operations. The default protocol
86 // of the netlink socket is NETLINK_ROUTE, but you can optionally
87 // provide a different one.
88 // TODO(chzhcn): Consider renaming 'routing' to 'netlink'.
89 inline Try<Netlink<struct nl_sock>> socket(int protocol = NETLINK_ROUTE)
90 {
91  Try<Nothing> checking = check();
92  if (checking.isError()) {
93  return Error(checking.error());
94  }
95 
96  struct nl_sock* s = nl_socket_alloc();
97  if (s == nullptr) {
98  return Error("Failed to allocate netlink socket");
99  }
100 
101  Netlink<struct nl_sock> sock(s);
102 
103  int error = nl_connect(sock.get(), protocol);
104  if (error != 0) {
105  return Error(
106  "Failed to connect to netlink protocol: " +
107  std::string(nl_geterror(error)));
108  }
109 
110  return sock;
111 }
