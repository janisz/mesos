47 // Implementation of the JavaScript Object Notation (JSON) grammar
48 // using boost::variant. We explicitly define each "type" of the
49 // grammar, including 'true' (json::True), 'false' (json::False), and
50 // 'null' (json::Null), for clarity and also because boost::variant
51 // "picks" the wrong type when we try and use std::string, long (or
52 // int), double (or float), and bool all in the same variant (while it
53 // does work with explicit casts, it seemed bad style to force people
54 // to put those casts in place). We could have avoided using
55 // json::String or json::Number and just used std::string and double
56 // respectively, but we choose to include them for completeness
57 // (although, this does pay a 2x cost when compiling thanks to all the
58 // extra template instantiations).

263  // Returns true if and only if 'other' is contained by 'this'.
264  // 'Other' is contained by 'this' if the following conditions are
265  // fulfilled:
266  // 1. If 'other' is a JSON object, then 'this' is also a JSON
267  // object, all keys of 'other' are also present in 'this' and
268  // the value for each key in 'this' also contain the value for
269  // the same key in 'other', i.e. for all keys 'k' in 'other',
270  // 'this[k].contains(other[k])' is true.
271  // 2. If 'other' is a JSON array, 'this' is also a JSON array, the
272  // length of both arrays is the same and each element in 'this'
273  // also contains the element in 'other' at the same position,
274  // i.e. it holds that this.length() == other.length() and
275  // for each i, 0 <= i < this.length,
276  // 'this[i].contains(other[i])'.
277  // 3. For all other types, 'this' is of the same type as 'other' and
278  // 'this == other'.
279  // NOTE: For a given key 'k', if 'this[k] == null' then
280  // 'this.contains(other)' holds if either 'k' is not present in
281  // 'other.keys()' or 'other[k] == null'.
282  // Similarly, if 'other[k] == null', 'this.contains(other)' only if
283  // 'this[k] == null'. This is a consequence of the containment
284  // definition.
285  bool contains(const Value& other) const;

340 template <typename T>
341 Result<T> Object::find(const std::string& path) const
342 {
343  const std::vector<std::string> names = strings::split(path, ".", 2);
344 
345  if (names.empty()) {
346  return None();
347  }
348 
349  std::string name = names[0];
350 
351  // Determine if we have an array subscript. If so, save it but
352  // remove it from the name for doing the lookup.
353  Option<size_t> subscript = None();
354  size_t index = name.find('[');
355  if (index != std::string::npos) {
356  // Check for the closing bracket.
357  if (name.at(name.length() - 1) != ']') {
358  return Error("Malformed array subscript, expecting ']'");
359  }
360 
361  // Now remove the closing bracket (last character) and everything
362  // before and including the opening bracket.
363  std::string s = name.substr(index + 1, name.length() - index - 2);
364 
365  // Now numify the subscript.
366  Try<int> i = numify<int>(s);
367 
368  if (i.isError()) {
369  return Error("Failed to numify array subscript '" + s + "'");
370  } else if (i.get() < 0) {
371  return Error("Array subscript '" + s + "' must be >= 0");
372  }
373 
374  subscript = i.get();
375 
376  // And finally remove the array subscript from the name.
377  name = name.substr(0, index);
378  }
379 
380  std::map<std::string, Value>::const_iterator entry = values.find(name);
381 
382  if (entry == values.end()) {
383  return None();
384  }
385 
386  Value value = entry->second;
387 
388  if (value.is<Array>() && subscript.isSome()) {
389  Array array = value.as<Array>();
390  if (subscript.get() >= array.values.size()) {
391  return None();
392  }
393  value = array.values[subscript.get()];
394  }
395 
396  if (names.size() == 1) {
397  if (!value.is<T>()) {
398  // TODO(benh): Use a visitor to print out the type found.
399  return Error("Found JSON value of wrong type");
400  }
401  return value.as<T>();
402  } else if (!value.is<Object>()) {
403  // TODO(benh): Use a visitor to print out the intermediate type.
404  return Error("Intermediate JSON value not an object");
405  }
406 
407  return value.as<Object>().find<T>(names[1]);
408 }

441 inline bool Value::ContainmentComparator::operator()(const Object& other) const
442 {
443  if (!self.is<Object>()) {
444  return false;
445  }
446 
447  // The empty set is contained in every set.
448  if (other.values.empty()) {
449  return true;
450  }
451 
452  const Object& _self = self.as<Object>();
453 
454  // All entries in 'other' should exists in 'self', which implies
455  // there should be at most as many entries in other as in self.
456  if (other.values.size() > _self.values.size()) {
457  return false;
458  }
459 
460  foreachpair (const std::string& key, const Value& value, other.values) {
461  auto _selfIterator = _self.values.find(key);
462 
463  if (_selfIterator == _self.values.end()) {
464  return false;
465  }
466 
467  if (!_selfIterator->second.contains(value)) {
468  return false;
469  }
470  }
471 
472  return true;
473 }

485 inline bool Value::ContainmentComparator::operator()(const Number& other) const
486 {
487  if (!self.is<Number>()) {
488  return false;
489  }
490 
491  // NOTE: For the following switch statements, we do not set a default
492  // case in order to leverage the compiler errors when the enumeration
493  // is augmented to find all the cases we need to provide.
494 
495  // NOTE: Using '==' is unsafe for unsigned-signed integer comparisons.
496  // The compiler will automatically cast the signed integer to an
497  // unsigned integer. i.e. If the signed integer was negative, it
498  // might be converted to a large positive number.
499  // Using the '==' operator *is* safe for double-integer comparisons.
500 
501  const Number& number = self.as<Number>();
502  switch (number.type) {
503  case Number::FLOATING:
504  switch (other.type) {
505  case Number::FLOATING:
506  return number.value == other.value;
507  case Number::SIGNED_INTEGER:
508  return number.value == other.signed_integer;
509  case Number::UNSIGNED_INTEGER:
510  return number.value == other.unsigned_integer;
511  }
512 
513  case Number::SIGNED_INTEGER:
514  switch (other.type) {
515  case Number::FLOATING:
516  return number.signed_integer == other.value;
517  case Number::SIGNED_INTEGER:
518  return number.signed_integer == other.signed_integer;
519  case Number::UNSIGNED_INTEGER:
520  // See note above for why this is not a simple '==' expression.
521  return number.signed_integer >= 0 &&
522  number.as<uint64_t>() == other.unsigned_integer;
523  }
524 
525  case Number::UNSIGNED_INTEGER:
526  switch (other.type) {
527  case Number::FLOATING:
528  return number.unsigned_integer == other.value;
529  case Number::SIGNED_INTEGER:
530  // See note above for why this is not a simple '==' expression.
531  return other.signed_integer >= 0 &&
532  number.unsigned_integer == other.as<uint64_t>();
533  case Number::UNSIGNED_INTEGER:
534  return number.unsigned_integer == other.unsigned_integer;
535  }
536  }
537 
538  UNREACHABLE();
539 }

653 inline std::ostream& operator<<(std::ostream& out, const Number& number)
654 {
655  switch (number.type) {
656  case Number::FLOATING: {
657  // Prints a floating point value, with the specified precision, see:
658  // http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2006/n2005.pdf
659  // Additionally ensures that a decimal point is in the output.
660  char buffer[50] {}; // More than long enough for the specified precision.
661  snprintf(
662  buffer,
663  sizeof(buffer),
664  "%#.*g",
665  std::numeric_limits<double>::digits10,
666  number.value);
667 
668  // Get rid of excess trailing zeroes before outputting.
669  // Otherwise, printing 1.0 would result in "1.00000000000000".
670  // NOTE: valid JSON numbers cannot end with a '.'.
671  std::string trimmed = strings::trim(buffer, strings::SUFFIX, "0");
672  return out << trimmed << (trimmed.back() == '.' ? "0" : "");
673  }
674  case Number::SIGNED_INTEGER:
675  return out << number.signed_integer;
676  case Number::UNSIGNED_INTEGER:
677  return out << number.unsigned_integer;
678

680  // errors when the enumeration is augmented to find all
681  // the cases we need to provide.

733 inline Value convert(const picojson::value& value)
734 {
735  if (value.is<picojson::null>()) {
736  return Null();
737  } else if (value.is<bool>()) {
738  return Boolean(value.get<bool>());
739  } else if (value.is<picojson::value::object>()) {
740  Object object;
741  foreachpair (const std::string& name,
742  const picojson::value& v,
743  value.get<picojson::value::object>()) {
744  object.values[name] = convert(v);
745  }
746  return object;
747  } else if (value.is<picojson::value::array>()) {
748  Array array;
749  foreach (const picojson::value& v, value.get<picojson::value::array>()) {
750  array.values.push_back(convert(v));
751  }
752  return array;
753  } else if (value.is<int64_t>()) {
754  return Number(value.get<int64_t>());
755  } else if (value.is<double>()) {

766 inline Try<Value> parse(const std::string& s)
767 {
768  const char* parseBegin = s.c_str();
769  picojson::value value;
770  std::string error;
771 
772  // Because PicoJson supports repeated parsing of multiple objects/arrays in a
773  // stream, it will quietly ignore trailing non-whitespace characters. We would
774  // rather throw an error, however, so use `last_char` to check for this.
775  const char* lastVisibleChar =
776  parseBegin + s.find_last_not_of(strings::WHITESPACE);
777 
778  // Parse the string, returning a pointer to the character
779  // immediately following the last one parsed.
780  const char* parseEnd =
781  picojson::parse(value, parseBegin, parseBegin + s.size(), &error);
782 
783  if (!error.empty()) {
784  return Error(error);
785  } else if (parseEnd != lastVisibleChar + 1) {
786  return Error(
787  "Parsed JSON included non-whitespace trailing characters: "
788  + s.substr(parseEnd - parseBegin, lastVisibleChar + 1 - parseEnd));
789  }
790 
791  return internal::convert(value);
792 }
