28 // object that can either be implicitly converted to a `std::string`, or

31 // `jsonify(const T&)` is implemented by calling the function `json`.
32 // We perform unqualified function call so that it can detect overloads via

52 // The result of `jsonify`. This is a light-weight proxy object that can either
53 // be implicitly converted to a `std::string`, or directly inserted into an
54 // output stream.
55 //
56 // In order to make this object light-weight, variables are captured by
57 // reference. This gives rise to similar semantics as `std::forward_as_tuple`.
58 // If the arguments are temporaries, `JSON::Proxy` does not extend their
59 // lifetime; they have to be used before the end of the full expression.
60 class Proxy
61 {
62 public:
63  operator std::string() &&
64  {
65  std::ostringstream stream;
66  stream << std::move(*this);
67  return stream.str();
68  }
69 
70 private:
71  Proxy(std::function<void(std::ostream*)> write) : write_(std::move(write)) {}
72 
73  // We declare copy/move constructors `private` to prevent statements that try
74  // to "save" an instance of `Proxy` such as:
75  //
76  // ```
77  // std::string F();
78  // Proxy proxy = jsonify(F());
79  // ```
80  //
81  // Since `proxy` in the above example would be holding onto a reference to a
82  // temporary string returned by `F()`.
83  Proxy(const Proxy&) = default;
84  Proxy(Proxy&&) = default;
85 
86  std::function<void(std::ostream*)> write_;
87 
88  template <typename T>
89  friend Proxy (::jsonify)(const T&);
90 
91  friend std::ostream& operator<<(std::ostream& stream, Proxy&& that);
92 };

125 // The number writer. If `set` is not called at all, `0` is printed.
126 // If `set` is called more than once, only the last value is printed.
127 class NumberWriter
128 {
129 public:
130  NumberWriter(std::ostream* stream)
131  : stream_(stream), type_(INT), int_(0) {}
132 
133  NumberWriter(const NumberWriter&) = delete;
134  NumberWriter(NumberWriter&&) = delete;
135 
136  ~NumberWriter()
137  {
138  switch (type_) {
139  case INT: {
140  *stream_ << int_;
141  break;
142  }
143  case UINT: {
144  *stream_ << uint_;
145  break;
146  }
147  case DOUBLE: {
148  // Prints a floating point value, with the specified precision, see:
149  // http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2006/n2005.pdf
150  // Additionally ensures that a decimal point is in the output.
151  char buffer[50]; // More than enough for the specified precision.
152  const int size = snprintf(
153  buffer,
154  sizeof(buffer),
155  "%#.*g",
156  std::numeric_limits<double>::digits10,
157  double_);
158 
159  // Get rid of excess trailing zeroes before outputting.
160  // Otherwise, printing 1.0 would result in "1.00000000000000".
161  //
162  // NOTE: We intentionally do not use `strings::trim` here in order to
163  // avoid construction of temporary strings.
164  int back = size - 1;
165  for (; back > 0; --back) {
166  if (buffer[back] != '0') {
167  break;
168  }
169  buffer[back] = '\0';
170  }
171 
172  // NOTE: valid JSON numbers cannot end with a '.'.
173  *stream_ << buffer << (buffer[back] == '.' ? "0" : "");
174  break;
175  }
176  }
177  }

267  void append(char c)
268  {
269  switch (c) {
270  case '"' : *stream_ << "\\\""; break;
271  case '\\': *stream_ << "\\\\"; break;
272  case '/' : *stream_ << "\\/"; break;
273  case '\b': *stream_ << "\\b"; break;
274  case '\f': *stream_ << "\\f"; break;
275  case '\n': *stream_ << "\\n"; break;
276  case '\r': *stream_ << "\\r"; break;
277  case '\t': *stream_ << "\\t"; break;
278  default: {
279  if (static_cast<unsigned char>(c) < 0x20 || c == 0x7f) {
280  char buffer[7];
281  snprintf(buffer, sizeof(buffer), "\\u%04x", c & 0xff);
282  stream_->write(buffer, sizeof(buffer) - 1);
283  } else {
284  *stream_ << c;
285  }
286  break;
287  }
288  }
289  }

440 // The member `value` is `true` if `T` is a sequence, and `false` otherwise.
441 template <typename T>
442 struct IsSequence
443 {
444 private:
445  // This overload only participates in overload resolution if the following
446  // expressions are valid.
447  // (1) begin(t) != end(t)
448  // (2) auto iter = begin(t); ++iter
449  // (3) *begin(t)
450  //
451  // The expressions are only used for SFINAE purposes, and comma operators are
452  // used to ignore the results of the expressions. That is, the return type of
453  // this function is `decltype(expr0, expr1, expr2, std::true_type{})` which is
454  // `std::true_type`.
455  template <typename U>
456  static auto test(Prefer) -> decltype(
457  // Cast to `void` to suppress `-Wunused-comparison` warnings.
458  void(std::begin(std::declval<U&>()) != std::end(std::declval<U&>())),
459  ++std::declval<decltype(std::begin(std::declval<U&>()))&>(),
460  *std::begin(std::declval<U&>()),
461  std::true_type{});
462 
463  // This overload is chosen if the preferred version is SFINAE'd out.
464  template <typename U>
465  static std::false_type test(LessPrefer);
466 
467 public:
468  static constexpr bool value = decltype(test<T>(Prefer()))::value;
469 };

492 // This function is only enabled if `Iterable` is iterable, is not a
493 // `const char (&)[N]` (in order to avoid ambiguity with the string literal
494 // overload), and does not have a member typedef `mapped_type` (we take the
495 // existence of `mapped_type` as the indication of an associative container).
496 template <
497  typename Iterable,
498  typename std::enable_if<
499  internal::IsSequence<Iterable>::value &&
500  !(std::is_array<Iterable>::value &&
501  std::rank<Iterable>::value == 1 &&

506 {

531 // An object that can be converted to a pointer to any of the JSON writers.
532 // This is used to resolve the following scenario:
533 //
534 // ```
535 // void json(JSON::ObjectWriter*, const Resources&);
536 //
537 // void json(
538 // JSON::ArrayWriter*,
539 // const google::protobuf::RepeatedPtrField<Resource>&);
540 //
541 // Resources resources;
542 // std::cout << jsonify(resources); // We want to use the first overload!
543 // ```
544 //
545 // The goal is to perform overload resolution based on the second parameter.
546 // Since `WriterProxy` is convertible to any of the writers equivalently, we
547 // force overload resolution of `json(WriterProxy(stream), value)` to depend
548 // only on the second parameter.
549 class WriterProxy
550 {
551 public:
552  WriterProxy(std::ostream* stream) : stream_(stream) {}
553 
554  ~WriterProxy()
555  {
556  switch (type_) {
557  case BOOLEAN_WRITER: {
558  writer_.boolean_writer.~BooleanWriter();
559  break;
560  }
561  case NUMBER_WRITER: {
562  writer_.number_writer.~NumberWriter();
563  break;
564  }
565  case STRING_WRITER: {
566  writer_.string_writer.~StringWriter();
567  break;
568  }
569  case ARRAY_WRITER: {
570  writer_.array_writer.~ArrayWriter();
571  break;
572  }
573  case OBJECT_WRITER: {
574  writer_.object_writer.~ObjectWriter();
575  break;
576  }
577  }
578  }
