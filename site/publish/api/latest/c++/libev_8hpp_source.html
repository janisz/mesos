56 // Wrapper around function we want to run in the event loop.
57 template <typename T>
58 void _run_in_event_loop(
59  const lambda::function<Future<T>()>& f,
60  const Owned<Promise<T>>& promise)
61 {
62  // Don't bother running the function if the future has been discarded.
63  if (promise->future().hasDiscard()) {
64  promise->discard();
65  } else {
66  promise->set(f());
67  }
68 }

71 // Helper for running a function in the event loop.
72 template <typename T>
73 Future<T> run_in_event_loop(const lambda::function<Future<T>()>& f)
74 {
75  // If this is already the event loop then just run the function.
76  if (__in_event_loop__) {
77  return f();
78  }
79 
80  Owned<Promise<T>> promise(new Promise<T>());
81 
82  Future<T> future = promise->future();
83 
84  // Enqueue the function.
85  synchronized (watchers_mutex) {
86  functions->push(lambda::bind(&_run_in_event_loop<T>, f, promise));
87  }
88 
89  // Interrupt the loop.
90  ev_async_send(loop, &async_watcher);
91 
92  return future;
93 }
