59 // Returns true if the future becomes ready, discarded, or failed
60 // after the wait.
61 template <typename T>
62 bool await(const process::Future<T>& future, const Duration& duration)
63 {
64  if (!process::Clock::paused()) {
65  return future.await(duration);
66  }
67 
68  // If the clock is paused, no new timers will expire.
69  // Future::await(duration) may hang forever because it depends on
70  // a timer to expire after 'duration'. We instead ensure all
71  // expired timers are flushed and check if the future is satisfied.
72  Stopwatch stopwatch;
73  stopwatch.start();
74 
75  // Settle to make sure all expired timers are executed (not
76  // necessarily finished, see below).
77  process::Clock::settle();
78 
79  while (future.isPending() && stopwatch.elapsed() < duration) {
80  // Synchronous operations and asynchronous process::Process
81  // operations should finish when the above 'settle()' returns.
82  // Other types of async operations such as io::write() may not.
83  // Therefore we wait the specified duration for it to complete.
84  // Note that nothing prevents the operations to schedule more
85  // timeouts for some time in the future. These timeouts will
86  // never be executed due to the paused process::Clock. In this
87  // case we return after the stopwatch (system clock) runs out.
88  os::sleep(Milliseconds(10));
89  }
90 
91  return !future.isPending();
92 }

115 ::testing::AssertionResult AwaitAssertReady(
116  const char* expr,
117  const char*, // Unused string representation of 'duration'.

120 {

132  return ::testing::AssertionSuccess();
