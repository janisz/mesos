150  void generate_keys_and_certs() {
151  // We store the allocated objects in these results so that we can
152  // have a consolidated 'cleanup()' function. This makes all the
153  // 'EXIT()' calls more readable and less error prone.
154  Result<EVP_PKEY*> private_key = None();
155  Result<X509*> certificate = None();
156  Result<EVP_PKEY*> scrap_key = None();
157  Result<X509*> scrap_certificate = None();
158 
159  auto cleanup = [&private_key, &certificate, &scrap_key, &scrap_certificate](
160  const Option<std::string> abort_message = None()) {
161  if (private_key.isSome()) { EVP_PKEY_free(private_key.get()); }
162  if (certificate.isSome()) { X509_free(certificate.get()); }
163  if (scrap_key.isSome()) { EVP_PKEY_free(scrap_key.get()); }
164  if (scrap_certificate.isSome()) { X509_free(scrap_certificate.get()); }
165 
166  // We abort here because failure during setup indicates that something
167  // is horribly and irrecoverably wrong.
168  if (abort_message.isSome()) {
169  ABORT(abort_message.get());
170  }
171  };
172 
173  // Generate the authority key.
174  private_key = process::network::openssl::generate_private_rsa_key();
175  if (private_key.isError()) {
176  cleanup("Could not generate private key: " + private_key.error());
177  }
178 
179  // Figure out the hostname that 'INADDR_LOOPBACK' will bind to.
180  // Set the hostname of the certificate to this hostname so that
181  // hostname verification of the certificate will pass.
182  Try<std::string> hostname = net::getHostname(net::IP(INADDR_LOOPBACK));
183  if (hostname.isError()) {
184  cleanup("Could not determine hostname of 'INADDR_LOOPBACK': " +
185  hostname.error());
186  }
187 
188  // Generate an authorized certificate.
189  certificate = process::network::openssl::generate_x509(
190  private_key.get(),
191  private_key.get(),
192  None(),
193  1,
194  365,
195  hostname.get());
196 
197  if (certificate.isError()) {
198  cleanup("Could not generate certificate: " + certificate.error());
199  }
200 
201  // Write the authority key to disk.
202  Try<Nothing> key_write =
203  process::network::openssl::write_key_file(private_key.get(), key_path());
204 
205  if (key_write.isError()) {
206  cleanup("Could not write private key to disk: " + key_write.error());
207  }
208 
209  // Write the authorized certificate to disk.
210  Try<Nothing> certificate_write =
211  process::network::openssl::write_certificate_file(
212  certificate.get(),
213  certificate_path());
214 
215  if (certificate_write.isError()) {
216  cleanup("Could not write certificate to disk: " +
217  certificate_write.error());
218  }
219 
220  // Generate a scrap key.
221  scrap_key = process::network::openssl::generate_private_rsa_key();
222  if (scrap_key.isError()) {
223  cleanup("Could not generate a scrap private key: " + scrap_key.error());
224  }
225 
226  // Write the scrap key to disk.
227  key_write = process::network::openssl::write_key_file(
228  scrap_key.get(),
229  scrap_key_path());
230 
231  if (key_write.isError()) {
232  cleanup("Could not write scrap key to disk: " + key_write.error());
233  }
234 
235  // Generate a scrap certificate.
236  scrap_certificate = process::network::openssl::generate_x509(
237  scrap_key.get(),
238  scrap_key.get());
239 
240  if (scrap_certificate.isError()) {
241  cleanup("Could not generate a scrap certificate: " +
242  scrap_certificate.error());
243  }
244 
245  // Write the scrap certificate to disk.
246  certificate_write = process::network::openssl::write_certificate_file(
247  scrap_certificate.get(),
248  scrap_certificate_path());
249 
250  if (certificate_write.isError()) {
251  cleanup("Could not write scrap certificate to disk: " +
252  certificate_write.error());
253  }
254 
255  // Since we successfully set up all our state, we call cleanup
256  // without an abort message (so as not to abort).
257  cleanup();
258  }
