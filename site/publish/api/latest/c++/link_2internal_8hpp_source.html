48 // Returns the netlink link object associated with a given link by its
49 // name. Returns None if the link is not found.
50 inline Result<Netlink<struct rtnl_link>> get(const std::string& link)
51 {
52  Try<Netlink<struct nl_sock>> socket = routing::socket();
53  if (socket.isError()) {
54  return Error(socket.error());
55  }
56 
57  // Dump all the netlink link objects from kernel. Note that the flag
58  // AF_UNSPEC means all available families.
59  struct nl_cache* c = nullptr;
60  int error = rtnl_link_alloc_cache(socket.get().get(), AF_UNSPEC, &c);
61  if (error != 0) {
62  return Error(nl_geterror(error));
63  }
64 
65  Netlink<struct nl_cache> cache(c);
66  struct rtnl_link* l = rtnl_link_get_by_name(cache.get(), link.c_str());
67  if (l == nullptr) {
68  return None();
69  }
70 
71  return Netlink<struct rtnl_link>(l);
72 }

102 // Tests if the flags are set on the link. Returns None if the link is
103 // not found.
104 inline Result<bool> test(const std::string& _link, unsigned int flags)
105 {
106  Result<Netlink<struct rtnl_link>> link = get(_link);
107  if (link.isError()) {
108  return Error(link.error());
109  } else if (link.isNone()) {
110  return None();
111  }
112 
113  return flags == (rtnl_link_get_flags(link.get().get()) & flags);
114 }

117 // Sets the flags on the link. Returns false if the link is not found.
118 inline Try<bool> set(const std::string& _link, unsigned int flags)
119 {
120  Result<Netlink<struct rtnl_link>> link = get(_link);
121  if (link.isError()) {
122  return Error(link.error());
123  } else if (link.isNone()) {
124  return false;
125  }
126 
127  // TODO(jieyu): We use ioctl to set the flags because the interfaces
128  // in libnl have some issues with virtual devices.
129  struct ifreq ifr;
130  memset(&ifr, 0, sizeof(ifr));
131 
132  // Get the existing flags and take a bit-wise OR.
133  ifr.ifr_flags = (rtnl_link_get_flags(link.get().get()) | flags);
134 
135  strncpy(ifr.ifr_name, _link.c_str(), IFNAMSIZ);
136 
137  int fd = ::socket(AF_INET, SOCK_STREAM, 0);
138  if (fd == -1) {
139  return ErrnoError();
140  }
141 
142  if (ioctl(fd, SIOCSIFFLAGS, &ifr) == -1) {
143  if (errno == ENODEV) {
144  os::close(fd);
145  return false;
146  } else {
147  // Save the error string as os::close may overwrite errno.
148  const std::string message = os::strerror(errno);
149  os::close(fd);
150  return Error(message);
151  }
152  }
153 
154  return true;
155 }
