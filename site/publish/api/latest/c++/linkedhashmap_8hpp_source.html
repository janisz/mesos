22 // Implementation of a hashmap that maintains the insertion order
23 // of the keys. Note that re-insertion of a key (i.e., update)
24 // doesn't update its insertion order.
25 // TODO(vinod/bmahler): Consider extending from stout::hashmap and/or
26 // having a compatible API with stout::hashmap.
27 template <typename Key, typename Value>
28 class LinkedHashMap
29 {
30 public:
31  typedef std::list<Key> list;
32  typedef hashmap<Key, std::pair<Value, typename list::iterator> > map;
33 
34  Value& operator[] (const Key& key)
35  {
36  if (!values_.contains(key)) {
37  // Insert into the list and get the "pointer" into the list.
38  typename list::iterator i = keys_.insert(keys_.end(), key);
39  values_[key] = std::make_pair(Value(), i); // Store default value.
40  }
41  return values_[key].first;
42  }
43 
44  Option<Value> get(const Key& key) const
45  {
46  if (values_.contains(key)) {
47  return values_.at(key).first;
48  }
49  return None();
50  }
51 
52  bool contains(const Key& key) const
53  {
54  return values_.contains(key);
55  }

57  size_t erase(const Key& key)
58  {
59  if (values_.contains(key)) {
60  // Get the "pointer" into the list.
61  typename list::iterator i = values_[key].second;
62  keys_.erase(i);
63  return values_.erase(key);
64  }
65  return 0;
66  }

73  std::list<Value> values() const
74  {
75  std::list<Value> result;
76  foreach (const Key& key, keys_) {
77  result.push_back(values_.at(key).first);
78  }
79  return result;
80  }
