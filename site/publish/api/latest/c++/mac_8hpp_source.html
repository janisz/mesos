55 // Represents a MAC address. A MAC address is a 48-bit unique
56 // identifier assigned to a network interface for communications on
57 // the physical network segment. We use a byte array (in transmission
58 // order) to represent a MAC address. For example, for a MAC address
59 // 01:23:34:67:89:ab, the format is shown as follows:
60 //
61 // MSB LSB
62 // | |
63 // v v
64 // +--------+--------+--------+--------+--------+--------+
65 // |bytes[0]|bytes[1]|bytes[2]|bytes[3]|bytes[4]|bytes[5]|
66 // +--------+--------+--------+--------+--------+--------+
67 //
68 // 01 : 23 : 45 : 67 : 89 : ab
69 // NOLINT(readability/ending_punctuation)
70 class MAC
71 {
72 public:
73  // Constructs a MAC address from a byte array.
74  explicit MAC(const uint8_t (&_bytes)[6])
75  {
76  for (size_t i = 0; i < 6; i++) {
77  bytes[i] = _bytes[i];
78  }
79  }
80 
81  // Returns the byte at the given index. For example, for a MAC
82  // address 01:23:45:67:89:ab, mac[0] = 01, mac[1] = 23 and etc.
83  uint8_t operator[](size_t index) const
84  {
85  if (index >= 6) {
86  ABORT("Invalid index specified in MAC::operator[]\n");
87  }
88 
89  return bytes[index];
90  }
91 
92  bool operator==(const MAC& that) const
93  {
94  for (size_t i = 0; i < 6; i++) {
95  if (bytes[i] != that.bytes[i]) {
96  return false;
97  }
98  }
99  return true;
100  }
101 
102  bool operator!=(const MAC& that) const
103  {
104  return !(*this == that);
105  }
106 
107 private:
108  // Byte array of this MAC address (in transmission order).
109  uint8_t bytes[6];
110 };

135 // Returns the MAC address of a given link device. The link device is
136 // specified using its name (e.g., eth0). Returns error if the link
137 // device is not found. Returns none if the link device is found, but
138 // does not have a MAC address (e.g., loopback).
139 inline Result<MAC> mac(const std::string& name)
140 {
141 #if !defined(__linux__) && !defined(__APPLE__) && !defined(__FreeBSD__)
142  return Error("Not implemented");
143 #else
144  struct ifaddrs* ifaddr = nullptr;
145  if (getifaddrs(&ifaddr) == -1) {
146  return ErrnoError();
147  }
148 
149  // Indicates whether the link device is found or not.
150  bool found = false;
151 
152  for (struct ifaddrs* ifa = ifaddr; ifa != nullptr; ifa = ifa->ifa_next) {
153  if (ifa->ifa_name != nullptr && !strcmp(ifa->ifa_name, name.c_str())) {
154  found = true;
155 
156 # if defined(__linux__)
157  if (ifa->ifa_addr != nullptr && ifa->ifa_addr->sa_family == AF_PACKET) {
158  struct sockaddr_ll* link = (struct sockaddr_ll*) ifa->ifa_addr;
159 
160  if (link->sll_halen == 6) {
161  struct ether_addr* addr = (struct ether_addr*) link->sll_addr;
162  MAC mac(addr->ether_addr_octet);
163 
164  // Ignore if the address is 0 so that the results are
165  // consistent on both OSX and Linux.
166  if (stringify(mac) == "00:00:00:00:00:00") {
167  continue;
168  }
169 
170  freeifaddrs(ifaddr);
171  return mac;
172  }
173  }
174 # elif defined(__APPLE__)
175  if (ifa->ifa_addr != nullptr && ifa->ifa_addr->sa_family == AF_LINK) {
176  struct sockaddr_dl* link = (struct sockaddr_dl*) ifa->ifa_addr;
177 
178  if (link->sdl_type == IFT_ETHER && link->sdl_alen == 6) {
179  struct ether_addr* addr = (struct ether_addr*) LLADDR(link);
180  MAC mac(addr->octet);
181 
182  freeifaddrs(ifaddr);
183  return mac;
184  }
185  }
186 # endif
187  }
188  }
189 
190  freeifaddrs(ifaddr);
191 
192  if (!found) {
193  return Error("Cannot find the link device");
194  }
195 
196  return None();
197 #endif
198 }
