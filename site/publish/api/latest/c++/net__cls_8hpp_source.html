39 // This defines the net_cls handle. The handle is composed of two
40 // parts, a 16-bit primary handle and a 16-bit secondary handle.
41 //
42 // TODO(asridharan): Currently we need to define the net_cls handle
43 // here, since we cannot use the definitions in
44 // `src/linux/routing/handle.hpp` due to its dependency on `libnl`,
45 // which is under GPL. Once we have been able to resolve these issues
46 // we should remove this definition and use the definition presented
47 // in `src/linux/routing/handle.hpp`.
48 struct NetClsHandle
49 {
50  NetClsHandle(uint16_t _primary, uint16_t _secondary)
51  : primary(_primary), secondary(_secondary) {};
52 
53  explicit NetClsHandle(uint32_t handle)
54  {
55  primary = handle >> 16;
56  secondary = handle & 0xffff;
57  };
58 
59  // Get the 32-bit representation of the handle in the form of
60  // 0xAAAABBBB. Where 0xAAAA is the primary handle and 0xBBBB is the
61  // secondary handle.
62  uint32_t get() const
63  {
64  uint32_t handle = primary;
65 
66  handle <<= 16;
67  handle |= secondary;
68 
69  return handle;
70  };
71 
72  uint16_t primary;
73  uint16_t secondary;
74 };

80 // This manages the net_cls handles for the `cgroup/net_cls` isolator.
81 // The isolator can use this with a range of primary handles, which
82 // will be managed by this class. For each primary handle there are
83 // 64K possible secondary handles. For a given primary handle the
84 // isolator can get a secondary handle by calling `alloc` and release
85 // an allocated handle by calling `free` on the secondary handle. For
86 // a given primary handle, the isolator can also explicitly reserve a
87 // secondary handle by calling `reserve`.
88 class NetClsHandleManager
89 {
90 public:
91  NetClsHandleManager(
92  const IntervalSet<uint32_t>& _primaries,
93  const IntervalSet<uint32_t>& _secondaries = IntervalSet<uint32_t>());
94 
95  ~NetClsHandleManager() {};
96

98  Try<uint16_t> allocPrimary() { return Error("Not Implemented"); }
99  Try<NetClsHandle> alloc(const Option<uint16_t>& primary = None());
100 
101  Try<Nothing> reserve(const NetClsHandle& handle);
102  Try<Nothing> free(const NetClsHandle& handle);
103 
104  // Check if a handle is used.
105  Try<bool> isUsed(const NetClsHandle& handle);
106 
107 private:
108  // The key to this hashmap is the 16-bit primary handle.
109  hashmap<uint16_t, std::bitset<0x10000>> used;
110 
111  // NOTE: Though the primary and secondary handles are 16 bit, we
112  // cannot use an `IntervalSet` specialization of type `uint16_t`
113  // since the intervals are stored in right openf format -- [x,y) --
114  // and setting the type to `uint16_t` would lead to overflow errors.
115  // For e.g., we would not be able to store the range [0xffff,0xffff]
116  // in `IntervalSet<uint16_t>` due to overflow error.
117  IntervalSet<uint32_t> primaries;
118  IntervalSet<uint32_t> secondaries;
119 };
