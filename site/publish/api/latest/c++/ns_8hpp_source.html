77 // Returns all the supported namespaces by the kernel.
78 inline std::set<std::string> namespaces()
79 {
80  std::set<std::string> result;
81  Try<std::list<std::string> > entries = os::ls("/proc/self/ns");
82  if (entries.isSome()) {
83  foreach (const std::string& entry, entries.get()) {
84  result.insert(entry);
85  }
86  }
87  return result;

91 // Returns the nstype (e.g., CLONE_NEWNET, CLONE_NEWNS, etc.) for the
92 // given namespace which will be used when calling ::setns.
93 inline Try<int> nstype(const std::string& ns)
94 {
95  hashmap<std::string, int> nstypes;
96 
97  nstypes["mnt"] = CLONE_NEWNS;
98  nstypes["uts"] = CLONE_NEWUTS;
99  nstypes["ipc"] = CLONE_NEWIPC;
100  nstypes["net"] = CLONE_NEWNET;
101  nstypes["user"] = CLONE_NEWUSER;
102  nstypes["pid"] = CLONE_NEWPID;
103 
104  if (!nstypes.contains(ns)) {
105  return Error("Unknown namespace '" + ns + "'");
106  }

112 // Re-associate the calling process with the specified namespace. The
113 // path refers to one of the corresponding namespace entries in the
114 // /proc/[pid]/ns/ directory (or bind mounted elsewhere). We do not
115 // allow a process with multiple threads to call this function because
116 // it will lead to some weird situations where different threads of a
117 // process are in different namespaces.
118 inline Try<Nothing> setns(const std::string& path, const std::string& ns)
119 {
120  // Return error if there're multiple threads in the calling process.
121  Try<std::set<pid_t> > threads = proc::threads(::getpid());
122  if (threads.isError()) {
123  return Error(
124  "Failed to get the threads of the current process: " +
125  threads.error());
126  } else if (threads.get().size() > 1) {
127  return Error("Multiple threads exist in the current process");
128  }
129 
130  if (ns::namespaces().count(ns) == 0) {
131  return Error("Namespace '" + ns + "' is not supported");
132  }
133 
134  // Currently, we don't support pid namespace as its semantics is
135  // different from other namespaces (instead of re-associating the
136  // calling thread, it re-associates the *children* of the calling
137  // thread with the specified namespace).
138  if (ns == "pid") {
139  return Error("Pid namespace is not supported");
140  }
141 
142  Try<int> fd = os::open(path, O_RDONLY | O_CLOEXEC);
143 
144  if (fd.isError()) {
145  return Error("Failed to open '" + path + "': " + fd.error());
146  }
147 
148 #ifndef O_CLOEXEC
149  Try<Nothing> cloexec = os::cloexec(fd.get());
150  if (cloexec.isError()) {
151  os::close(fd.get());
152  return Error("Failed to cloexec: " + cloexec.error());
153  }
154 #endif
155 
156  Try<int> nstype = ns::nstype(ns);
157  if (nstype.isError()) {
158  return Error(nstype.error());
159  }
160 
161 #ifdef SYS_setns
162  int ret = ::syscall(SYS_setns, fd.get(), nstype.get());
163 #elif __x86_64__
164  // A workaround for those hosts that have an old glibc (older than
165  // 2.14) but have a new kernel. The magic number '308' here is the
166  // syscall number for 'setns' on x86_64 architecture.
167  int ret = ::syscall(308, fd.get(), nstype.get());
168 #else
169 #error "setns is not available"
170 #endif
171 
172  if (ret == -1) {
173  // Save the errno as it might be overwritten by 'os::close' below.
174  ErrnoError error;
175  os::close(fd.get());
176  return error;
177  }
178 
179  os::close(fd.get());
180  return Nothing();
181 }

184 // Re-associate the calling process with the specified namespace. The
185 // pid specifies the process whose namespace we will associate.
186 inline Try<Nothing> setns(pid_t pid, const std::string& ns)
187 {
188  if (!os::exists(pid)) {
189  return Error("Pid " + stringify(pid) + " does not exist");
190  }
191 
192  std::string path = path::join("/proc", stringify(pid), "ns", ns);
193  if (!os::exists(path)) {
194  return Error("Namespace '" + ns + "' is not supported");
195  }
196 
197  return ns::setns(path, ns);
198 }

201 // Get the inode number of the specified namespace for the specified
202 // pid. The inode number identifies the namespace and can be used for
203 // comparisons, i.e., two processes with the same inode for a given
204 // namespace type are in the same namespace.
205 inline Try<ino_t> getns(pid_t pid, const std::string& ns)
206 {
207  if (!os::exists(pid)) {
208  return Error("Pid " + stringify(pid) + " does not exist");
209  }
210 
211  if (ns::namespaces().count(ns) < 1) {
212  return Error("Namespace '" + ns + "' is not supported");
213  }
214 
215  std::string path = path::join("/proc", stringify(pid), "ns", ns);
216  struct stat s;
217  if (::stat(path.c_str(), &s) < 0) {
218  return ErrnoError("Failed to stat " + ns + " namespace handle"
219  " for pid " + stringify(pid));
220  }
221 
222  return s.st_ino;
223 }

234 inline process::Future<Nothing> destroy(ino_t inode)
235 {
236  // Check we're not trying to kill the root namespace.
237  Try<ino_t> ns = ns::getns(1, "pid");
238  if (ns.isError()) {
239  return process::Failure(ns.error());
240  }
241 
242  if (ns.get() == inode) {
243  return process::Failure("Cannot destroy root pid namespace");
244  }
245 
246  // Or ourselves.
247  ns = ns::getns(::getpid(), "pid");
248  if (ns.isError()) {
249  return process::Failure(ns.error());
250  }
251 
252  if (ns.get() == inode) {
253  return process::Failure("Cannot destroy own pid namespace");
254  }
255 
256  // Signal all pids in the namespace, including the init pid if it's
257  // still running. Once the init pid has been signalled the kernel
258  // will prevent any new children forking in the namespace and will
259  // also signal all other pids in the namespace.
260  Try<std::set<pid_t>> pids = os::pids();
261  if (pids.isError()) {
262  return process::Failure("Failed to list of processes");
263  }
264 
265  foreach (pid_t pid, pids.get()) {
266  // Ignore any errors, probably because the process no longer
267  // exists, and ignorable otherwise.
268  Try<ino_t> ns = ns::getns(pid, "pid");
269  if (ns.isSome() && ns.get() == inode) {
270  kill(pid, SIGKILL);
271  }
272  }
273 
274  // Get a new snapshot and do a second pass of the pids to capture
275  // any pids that are dying so we can reap them.
276  pids = os::pids();
277  if (pids.isError()) {
278  return process::Failure("Failed to list of processes");
279  }
280 
281  std::list<process::Future<Option<int>>> futures;
282 
283  foreach (pid_t pid, pids.get()) {
284  Try<ino_t> ns = ns::getns(pid, "pid");
285  if (ns.isSome() && ns.get() == inode) {
286  futures.push_back(process::reap(pid));
287  }
288 
289  // Ignore any errors, probably because the process no longer
290  // exists, and ignorable otherwise.
291  }
292 
293  // Wait for all the signalled processes to terminate. The pid
294  // namespace will then be empty and will be released by the kernel
295  // (unless there are additional references).
296  return process::collect(futures)
297  .then(lambda::bind(&internal::_nothing));
298 }

303 // Returns the namespace flags in the string form of bitwise-ORing the
304 // flags, e.g., CLONE_NEWNS | CLONE_NEWNET.
305 inline std::string stringify(int flags)
306 {
307  hashmap<unsigned int, std::string> names = {
308  {CLONE_NEWNS, "CLONE_NEWNS"},
309  {CLONE_NEWUTS, "CLONE_NEWUTS"},
310  {CLONE_NEWIPC, "CLONE_NEWIPC"},
311  {CLONE_NEWPID, "CLONE_NEWPID"},
312  {CLONE_NEWNET, "CLONE_NEWNET"},
313  {CLONE_NEWUSER, "CLONE_NEWUSER"}
314  };
315 
316  std::vector<std::string> namespaces;
317  foreachpair (unsigned int flag, const std::string& name, names) {
318  if (flags & flag) {
319  namespaces.push_back(name);
320  }
321  }
322 
323  return strings::join(" | ", namespaces);
324 }
