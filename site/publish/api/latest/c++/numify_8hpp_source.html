28 template <typename T>
29 Try<T> numify(const std::string& s)
30 {
31  try {
32  return boost::lexical_cast<T>(s);
33  } catch (const boost::bad_lexical_cast&) {
34  // Unfortunately boost::lexical_cast cannot cast a hexadecimal
35  // number even with a "0x" prefix, we have to workaround this
36  // issue here.
37  if (strings::startsWith(s, "0x") || strings::startsWith(s, "0X")) {
38  // NOTE: Hexadecimal floating-point constants (e.g., 0x1p-5,
39  // 0x10.0), are allowed in C99, but cannot be used as floating
40  // point literals in standard C++. Some C++ compilers might
41  // accept them as an extension; for consistency, we always
42  // disallow them. See:
43  // https://gcc.gnu.org/onlinedocs/gcc/Hex-Floats.html
44  if (!strings::contains(s, ".") && !strings::contains(s, "p")) {
45  T result;
46  std::stringstream ss;
47  ss << std::hex << s;
48  ss >> result;
49  // Make sure we really hit the end of the string.
50  if (!ss.fail() && ss.eof()) {
51  return result;
52  }
53  }
54  }
55 
56  return Error("Failed to convert '" + s + "' to number");
57  }
58 }

70 {
