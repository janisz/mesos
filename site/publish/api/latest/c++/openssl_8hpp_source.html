54 // Initializes the _global_ OpenSSL context (SSL_CTX) as well as the
55 // crypto library in order to support multi-threading. The global
56 // context gets initialized using the environment variables:
57 //
58 // SSL_ENABLED=(false|0,true|1)
59 // SSL_SUPPORT_DOWNGRADE=(false|0,true|1)
60 // SSL_CERT_FILE=(path to certificate)
61 // SSL_KEY_FILE=(path to key)
62 // SSL_VERIFY_CERT=(false|0,true|1)
63 // SSL_REQUIRE_CERT=(false|0,true|1)
64 // SSL_VERIFY_DEPTH=(4)
65 // SSL_CA_DIR=(path to CA directory)
66 // SSL_CA_FILE=(path to CA file)
67 // SSL_CIPHERS=(accepted ciphers separated by ':')
68 // SSL_ENABLE_SSL_V3=(false|0,true|1)
69 // SSL_ENABLE_TLS_V1_0=(false|0,true|1)
70 // SSL_ENABLE_TLS_V1_1=(false|0,true|1)
71 // SSL_ENABLE_TLS_V1_2=(false|0,true|1)
72 //
73 // TODO(benh): When/If we need to support multiple contexts in the
74 // same process, for example for Server Name Indication (SNI), then
75 // we'll add other functions for initializing an SSL_CTX based on
76 // these environment variables.
77 // TODO(nneilsen): Support certification revocation.
