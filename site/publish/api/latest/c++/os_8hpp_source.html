166 inline Try<std::list<Process>> processes()
167 {
168  const Try<std::set<pid_t>> pids = os::pids();
169  if (pids.isError()) {
170  return Error(pids.error());
171  }
172 
173  std::list<Process> result;
174  foreach (pid_t pid, pids.get()) {
175  const Result<Process> process = os::process(pid);
176 
177  // Ignore any processes that disappear between enumeration and now.
178  if (process.isSome()) {
179  result.push_back(process.get());
180  }
181  }
182  return result;
183 }

199 inline std::set<pid_t> children(
200  pid_t pid,
201  const std::list<Process>& processes,
202  bool recursive = true)
203 {
204  // Perform a breadth first search for descendants.
205  std::set<pid_t> descendants;
206  std::queue<pid_t> parents;
207  parents.push(pid);
208 
209  do {
210  pid_t parent = parents.front();
211  parents.pop();
212 
213  // Search for children of parent.
214  foreach (const Process& process, processes) {
215  if (process.parent == parent) {
216  // Have we seen this child yet?
217  if (descendants.insert(process.pid).second) {
218  parents.push(process.pid);
219  }
220  }
221  }
222  } while (recursive && !parents.empty());
223 
224  return descendants;
225 }
