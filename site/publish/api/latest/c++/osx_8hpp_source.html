41 inline Result<Process> process(pid_t pid)
42 {
43  const Try<std::vector<kinfo_proc>> processes =
44  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_PID, pid).table(1);
45 
46  if (processes.isError()) {
47  return Error("Failed to get process via sysctl: " + processes.error());
48  } else if (processes.get().size() != 1) {
49  return None();
50  }
51 
52  const kinfo_proc process = processes.get()[0];
53 
54  // The command line from 'process.kp_proc.p_comm' only includes the
55  // first 16 characters from "arg0" (i.e., the canonical executable
56  // name). We can try to get "argv" via some sysctl magic. This first
57  // requires determining "argc" via KERN_PROCARGS2 followed by the
58  // actual arguments via KERN_PROCARGS. This is still insufficient
59  // with insufficient privilege (e.g., not being root). If we were
60  // only interested in the "executable path" (i.e., the first
61  // argument to 'exec' but none of the arguments) we could use
62  // proc_pidpath() instead.
63  Option<std::string> command = None();
64 
65  // Alternative implemenations using KERN_PROCARGS2:
66  // https://bitbucket.org/flub/psi-win32/src/d38f288de3b8/src/arch/macosx/macosx_process.c#cl-552
67  // https://gist.github.com/nonowarn/770696
68 
69 #ifdef KERN_PROCARGS2
70  // Looking at the source code of XNU (the Darwin kernel for OS X:
71  // www.opensource.apple.com/source/xnu/xnu-1699.24.23/bsd/kern/kern_sysctl.c),
72  // it appears as though KERN_PROCARGS2 writes 'argc' as the first
73  // word of the returned bytes.
74  Try<std::string> args = os::sysctl(CTL_KERN, KERN_PROCARGS2, pid).string();
75 
76  if (args.isSome()) {
77  int argc = *((int*) args.get().data());
78 
79  if (argc > 0) {
80  // Now grab the arguments.
81  args = os::sysctl(CTL_KERN, KERN_PROCARGS, pid).string();
82 
83  if (args.isSome()) {
84  // At this point 'args' contains the parameters to 'exec'
85  // delimited by null bytes, i.e., "executable path", then
86  // "arg0" (the canonical executable name), then "arg1", then
87  // "arg2", etc. Sometimes there are no arguments (argc = 1) so
88  // all we care about is the "executable path", but when there
89  // are arguments we grab "arg0" and on assuming that "arg0"
90  // really is the canonical executable name.
91 
92  // Tokenize the args by the null byte ('\0').
93  std::vector<std::string> tokens =
94  strings::tokenize(args.get(), std::string(1, '\0'));
95 
96  if (!tokens.empty()) {
97  if (argc == 1) {
98  // When there are no arguments, all we care about is the
99  // "executable path".
100  command = tokens[0];
101  } else if (argc > 1) {
102  // When there are arguments, we skip the "executable path"
103  // and just grab "arg0" -> "argN", assuming "arg0" is the
104  // canonical executable name. In the case that we didn't
105  // get enough tokens back from KERN_PROCARGS the following
106  // code will end up just keeping 'command' None (i.e.,
107  // tokens.size() will be <= 0).
108  tokens.erase(tokens.begin()); // Remove path.
109 
110  // In practice it appeared that we might get an 'argc'
111  // which is greater than 'tokens.size()' which caused us
112  // to get an invalid iterator from 'tokens.begin() +
113  // argc', thus we now check that condition here.
114  if (static_cast<size_t>(argc) <= tokens.size()) {
115  tokens.erase(tokens.begin() + argc, tokens.end());
116  }
117 
118  // If any tokens are left, consider them the command!
119  if (tokens.size() > 0) {
120  command = strings::join(" ", tokens);
121  }
122  }
123  }
124  }
125  }
126  }
127 #endif

129  // We also use proc_pidinfo() to get memory and CPU usage.
130  // NOTE: There are several pitfalls to using proc_pidinfo().
131  // In particular:
132  // -This will not work for many root processes.
133  // -This may not work for processes owned by other users.
134  // -However, this always works for processes owned by the same user.
135  // This beats using task_for_pid(), which only works for the same pid.
136  // For further discussion around these issues,
137  // see: http://code.google.com/p/psutil/issues/detail?id=297
138  proc_taskinfo task;
139  int size = proc_pidinfo(pid, PROC_PIDTASKINFO, 0, &task, sizeof(task));
140 
141  // It appears that zombie processes on OS X do not have sessions and
142  // result in ESRCH.
143  int session = getsid(pid);
144 
145  if (size != sizeof(task)) {
146  return Process(process.kp_proc.p_pid,
147  process.kp_eproc.e_ppid,
148  process.kp_eproc.e_pgid,
149  session > 0 ? session : Option<pid_t>::none(),
150  None(),
151  None(),
152  None(),
153  command.getOrElse(std::string(process.kp_proc.p_comm)),
154  process.kp_proc.p_stat & SZOMB);
155  } else {
156  return Process(process.kp_proc.p_pid,
157  process.kp_eproc.e_ppid,
158  process.kp_eproc.e_pgid,
159  session > 0 ? session : Option<pid_t>::none(),
160  Bytes(task.pti_resident_size),
161  Nanoseconds(task.pti_total_user),
162  Nanoseconds(task.pti_total_system),
163  command.getOrElse(std::string(process.kp_proc.p_comm)),
164  process.kp_proc.p_stat & SZOMB);
165  }
166 }

169 inline Try<std::set<pid_t> > pids()
170 {
171  const Try<int> maxproc = os::sysctl(CTL_KERN, KERN_MAXPROC).integer();
172 
173  if (maxproc.isError()) {
174  return Error(maxproc.error());
175  }
176 
177  const Try<std::vector<kinfo_proc>> processes =
178  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_ALL).table(maxproc.get());
179 
180  if (processes.isError()) {
181  return Error(processes.error());
182  }
183 
184  std::set<pid_t> result;
185  foreach (const kinfo_proc& process, processes.get()) {
186  result.insert(process.kp_proc.p_pid);
187  }
188  return result;
189 }

192 // Returns the total size of main and free memory.
193 inline Try<Memory> memory()
194 {
195  Memory memory;
196 
197  const Try<int64_t> totalMemory = os::sysctl(CTL_HW, HW_MEMSIZE).integer();
198 
199  if (totalMemory.isError()) {
200  return Error(totalMemory.error());
201  }
202  memory.total = Bytes(totalMemory.get());
203 
204  // Size of free memory is available in terms of number of
205  // free pages on Mac OS X.
206  const int pageSize = os::pagesize();
207  if (pageSize < 0) {
208  return ErrnoError();
209  }
210 
211  unsigned int freeCount;
212  size_t length = sizeof(freeCount);
213 
214  if (sysctlbyname(
215  "vm.page_free_count",
216  &freeCount,
217  &length,
218  nullptr,
219  0) != 0) {
220  return ErrnoError();
221  }
222  memory.free = Bytes(freeCount * pageSize);
223 
224  struct xsw_usage usage;
225  length = sizeof(struct xsw_usage);
226  if (sysctlbyname(
227  "vm.swapusage",
228  &usage,
229  &length,
230  nullptr,
231  0) != 0) {
232  return ErrnoError();
233  }
234  memory.totalSwap = Bytes(usage.xsu_total * pageSize);
235  memory.freeSwap = Bytes(usage.xsu_avail * pageSize);
236 
237  return memory;
238 }
