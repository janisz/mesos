115  inline std::string basename() const
116  {
117  if (value.empty()) {
118  return std::string(".");
119  }
120 
121  size_t end = value.size() - 1;
122 
123  // Remove trailing slashes.
124  if (value[end] == os::PATH_SEPARATOR) {
125  end = value.find_last_not_of(os::PATH_SEPARATOR, end);
126 
127  // Paths containing only slashes result into "/".
128  if (end == std::string::npos) {
129  return stringify(os::PATH_SEPARATOR);
130  }
131  }
132 
133  // 'start' should point towards the character after the last slash
134  // that is non trailing.
135  size_t start = value.find_last_of(os::PATH_SEPARATOR, end);
136 
137  if (start == std::string::npos) {
138  start = 0;
139  } else {
140  start++;
141  }
142 
143  return value.substr(start, end + 1 - start);
144  }
