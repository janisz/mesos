22 inline size_t dirent_size(DIR* dir)
23 {
24  // Calculate the size for a "directory entry".
25  long name_max = fpathconf(dirfd(dir), _PC_NAME_MAX);
26 
27  // If we don't get a valid size, check NAME_MAX, but fall back on
28  // 255 in the worst case ... Danger, Will Robinson!
29  if (name_max == -1) {
30  name_max = (NAME_MAX > 255) ? NAME_MAX : 255;
31  }
32 
33  size_t name_end = (size_t) offsetof(dirent, d_name) + name_max + 1;
34 
35  size_t size = (name_end > sizeof(dirent) ? name_end : sizeof(dirent));
36 
37  return size;
38 }
