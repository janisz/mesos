104 // Suspends execution of the calling process until a child specified by `pid`
105 // has changed state. Unlike the POSIX standard function `::waitpid`, this
106 // function does not use -1 and 0 to signify errors and nonblocking return.
107 // Instead, we return `Result<pid_t>`:
108 // * In case of error, we return `Error` rather than -1. For example, we
109 // would return an `Error` in case of `EINVAL`.
110 // * In case of nonblocking return, we return `None` rather than 0. For
111 // example, if we pass `WNOHANG` in the `options`, we would expect 0 to be
112 // returned in the case that children specified by `pid` exist, but have
113 // not changed state yet. In this case we return `None` instead.
114 //
115 // NOTE: There are important differences between the POSIX and Windows
116 // implementations of this function:
117 // * On POSIX, `pid_t` is a signed number, but on Windows, PIDs are `DWORD`,
118 // which is `unsigned long`. Thus, if we use `DWORD` to represent the `pid`
119 // argument, we would not be able to pass -1 as the `pid`.
120 // * Since it is important to be able to detect -1 has been passed to
121 // `os::waitpid`, as a matter of practicality, we choose to:
122 // (1) Use `long` to represent the `pid` argument.
123 // (2) Disable using any value <= 0 for `pid` on Windows.
124 // * This decision is pragmatic. The reasoning is:
125 // (1) The Windows code paths call `os::waitpid` in only a handful of
126 // places, and in none of these conditions do we need `-1` as a value.
127 // (2) Since PIDs virtually never take on values outside the range of
128 // vanilla signed `long` it is likely that an accidental conversion
129 // will never happen.
130 // (3) Even though it is not formalized in the C specification, the
131 // implementation of `long` on the vast majority of production servers
132 // is 2's complement, so we expect that when we accidentally do
133 // implicitly convert from `unsigned long` to `long`, we will "wrap
134 // around" to negative values. And since we've disabled the negative
135 // `pid` in the Windows implementation, we should error out.
136 inline Result<pid_t> waitpid(pid_t pid, int* status, int options)
137 {
138  const pid_t child_pid = ::waitpid(pid, status, options);
139 
140  if (child_pid == 0) {
141  return None();
142  } else if (child_pid < 0) {
143  return ErrnoError("os::waitpid: Call to `waitpid` failed");
144  } else {
145  return child_pid;
146  }
147 }

214 // Suspends execution for the given duration.
215 inline Try<Nothing> sleep(const Duration& duration)
216 {
217  timespec remaining;
218  remaining.tv_sec = static_cast<long>(duration.secs());
219  remaining.tv_nsec =
220  static_cast<long>((duration - Seconds(remaining.tv_sec)).ns());
221 
222  while (nanosleep(&remaining, &remaining) == -1) {
223  if (errno == EINTR) {
224  continue;
225  } else {
226  return ErrnoError();
227  }
228  }
229 
230  return Nothing();
231 }

311 // Overload of os::pids for filtering by groups and sessions.
312 // A group / session id of 0 will fitler on the group / session ID
313 // of the calling process.
314 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
315 {
316  if (group.isNone() && session.isNone()) {
317  return os::pids();
318  } else if (group.isSome() && group.get() < 0) {
319  return Error("Invalid group");
320  } else if (session.isSome() && session.get() < 0) {
321  return Error("Invalid session");
322  }
323 
324  const Try<std::list<Process>> processes = os::processes();
325 
326  if (processes.isError()) {
327  return Error(processes.error());
328  }
329 
330  // Obtain the calling process group / session ID when 0 is provided.
331  if (group.isSome() && group.get() == 0) {
332  group = getpgid(0);
333  }
334  if (session.isSome() && session.get() == 0) {
335  session = getsid(0);
336  }
337 
338  std::set<pid_t> result;
339  foreach (const Process& process, processes.get()) {
340  // Group AND Session (intersection).
341  if (group.isSome() && session.isSome()) {
342  if (group.get() == process.group &&
343  process.session.isSome() &&
344  session.get() == process.session.get()) {
345  result.insert(process.pid);
346  }
347  } else if (group.isSome() && group.get() == process.group) {
348  result.insert(process.pid);
349  } else if (session.isSome() && process.session.isSome() &&
350  session.get() == process.session.get()) {
351  result.insert(process.pid);
352  }
353  }
354 
355  return result;
356 }

388 // Return the OS release numbers.
389 inline Try<Version> release()
390 {
391  Try<UTSInfo> info = uname();
392  if (info.isError()) {
393  return Error(info.error());
394  }

397 #ifndef __FreeBSD__

402  "%d.%d.%d",

406  return Error("Failed to parse: " + info.get().release);

408 #else

411  return Error("Failed to parse: " + info.get().release);
412  }
413 #endif
414  return Version(major, minor, patch);

418 inline Option<std::string> which(const std::string& command)
419 {
420  Option<std::string> path = getenv("PATH");
421  if (path.isNone()) {
422  return None();
423  }
424 
425  std::vector<std::string> tokens = strings::tokenize(path.get(), ":");
426  foreach (const std::string& token, tokens) {
427  const std::string commandPath = path::join(token, command);
428  if (!os::exists(commandPath)) {
429  continue;
430  }
431 
432  Try<os::Permissions> permissions = os::permissions(commandPath);
433  if (permissions.isError()) {
434  continue;
435  }
436 
437  if (!permissions.get().owner.x &&
438  !permissions.get().group.x &&
439  !permissions.get().others.x) {
440  continue;
441  }
442 
443  return commandPath;
444  }
445 
446  return None();
447 }
