29 // By default, recursively deletes a directory akin to: 'rm -r'. If
30 // `recursive` is false, it deletes a directory akin to: 'rmdir'. In
31 // recursive mode, `removeRoot` can be set to false to enable removing
32 // all the files and directories beneath the given root directory, but
33 // not the root directory itself.
34 // Note that this function expects an absolute path.
35 #ifndef __sun // FTS is not available on Solaris.
36 inline Try<Nothing> rmdir(
37  const std::string& directory,
38  bool recursive = true,
39  bool removeRoot = true)
40 {
41  if (!recursive) {
42  if (::rmdir(directory.c_str()) < 0) {
43  return ErrnoError();
44  }
45  } else {
46  // NOTE: `fts_open` will not always return `nullptr` if the path does not
47  // exist. We manually induce an error here to indicate that we can't remove
48  // a directory that does not exist.
49  if (!os::exists(directory)) {
50  errno = ENOENT;
51  return ErrnoError();
52  }
53 
54  char* paths[] = {const_cast<char*>(directory.c_str()), nullptr};
55 
56  // Using `FTS_PHYSICAL` here because we need `FTSENT` for the
57  // symbolic link in the directory and not the target it links to.
58  FTS* tree = fts_open(paths, (FTS_NOCHDIR | FTS_PHYSICAL), nullptr);
59  if (tree == nullptr) {
60  return ErrnoError();
61  }
62 
63  FTSENT* node;
64  while ((node = fts_read(tree)) != nullptr) {
65  switch (node->fts_info) {
66  case FTS_DP:
67  // Don't remove the root of the traversal of `removeRoot`
68  // is false.
69  if (!removeRoot && node->fts_level == FTS_ROOTLEVEL) {
70  continue;
71  }
72 
73  if (::rmdir(node->fts_path) < 0 && errno != ENOENT) {
74  Error error = ErrnoError();
75  fts_close(tree);
76  return error;
77  }
78  break;
79  // `FTS_DEFAULT` would include any file type which is not
80  // explicitly described by any of the other `fts_info` values.
81  case FTS_DEFAULT:
82  case FTS_F:
83  case FTS_SL:
84  // `FTS_SLNONE` should never be the case as we don't set
85  // `FTS_COMFOLLOW` or `FTS_LOGICAL`. Adding here for completion.
86  case FTS_SLNONE:
87  if (::unlink(node->fts_path) < 0 && errno != ENOENT) {
88  Error error = ErrnoError();
89  fts_close(tree);
90  return error;
91  }
92  break;
93  default:
94  break;
95  }
96  }
97 
98  if (errno != 0) {
99  Error error = ErrnoError();
100  fts_close(tree);
101  return error;
102  }
103 
104  if (fts_close(tree) < 0) {
105  return ErrnoError();
106  }
107  }
108 
109  return Nothing();
110 }
111 #endif // __sun
