70 Try<std::string> shell(const std::string& fmt, const T&... t)
71 {
72  const Try<std::string> command = strings::internal::format(fmt, t...);
73  if (command.isError()) {
74  return Error(command.error());
75  }
76 
77  FILE* file;
78  std::ostringstream stdout;
79 
80  if ((file = popen(command.get().c_str(), "r")) == nullptr) {
81  return Error("Failed to run '" + command.get() + "'");
82  }
83 
84  char line[1024];
85  // NOTE(vinod): Ideally the if and while loops should be interchanged. But
86  // we get a broken pipe error if we don't read the output and simply close.
87  while (fgets(line, sizeof(line), file) != nullptr) {
88  stdout << line;
89  }
90 
91  if (ferror(file) != 0) {
92  pclose(file); // Ignoring result since we already have an error.
93  return Error("Error reading output of '" + command.get() + "'");
94  }
95 
96  int status;
97  if ((status = pclose(file)) == -1) {
98  return Error("Failed to get status of '" + command.get() + "'");
99  }
100 
101  if (WIFSIGNALED(status)) {
102  return Error(
103  "Running '" + command.get() + "' was interrupted by signal '" +
104  strsignal(WTERMSIG(status)) + "'");
105  } else if ((WEXITSTATUS(status) != EXIT_SUCCESS)) {
106  LOG(ERROR) << "Command '" << command.get()
107  << "' failed; this is the output:\n" << stdout.str();
108  return Error(
109  "Failed to execute '" + command.get() + "'; the command was either "
110  "not found or exited with a non-zero exit status: " +
111  stringify(WEXITSTATUS(status)));
112  }
113 
114  return stdout.str();
115 }

118 // Executes a command by calling "/bin/sh -c <command>", and returns
119 // after the command has been completed. Returns 0 if succeeds, and
120 // return -1 on error (e.g., fork/exec/waitpid failed). This function
121 // is async signal safe. We return int instead of returning a Try
122 // because Try involves 'new', which is not async signal safe.
123 inline int system(const std::string& command)
124 {
125  pid_t pid = ::fork();
126 
127  if (pid == -1) {
128  return -1;
129  } else if (pid == 0) {
130  // In child process.
131  ::execlp(
132  Shell::name, Shell::arg0, Shell::arg1, command.c_str(), (char*)nullptr);
133  ::exit(127);
134  } else {
135  // In parent process.
136  int status;
137  while (::waitpid(pid, &status, 0) == -1) {
138  if (errno != EINTR) {
139  return -1;
140  }
141  }
142 
143  return status;
144  }
145 }
