32 inline Result<uid_t> getuid(const Option<std::string>& user = None())
33 {
34  if (user.isNone()) {
35  return ::getuid();
36  }
37 
38  struct passwd passwd;
39  struct passwd* result = nullptr;
40 
41  int size = sysconf(_SC_GETPW_R_SIZE_MAX);
42  if (size == -1) {
43  // Initial value for buffer size.
44  size = 1024;
45  }
46 
47  while (true) {
48  char* buffer = new char[size];
49 
50  if (getpwnam_r(user.get().c_str(), &passwd, buffer, size, &result) == 0) {
51  // The usual interpretation of POSIX is that getpwnam_r will
52  // return 0 but set result == nullptr if the user is not found.
53  if (result == nullptr) {
54  delete[] buffer;
55  return None();
56  }
57 
58  uid_t uid = passwd.pw_uid;
59  delete[] buffer;
60  return uid;
61  } else {
62  // RHEL7 (and possibly other systems) will return non-zero and
63  // set one of the following errors for "The given name or uid
64  // was not found." See 'man getpwnam_r'. We only check for the
65  // errors explicitly listed, and do not consider the ellipsis.
66  if (errno == ENOENT ||
67  errno == ESRCH ||
68  errno == EBADF ||
69  errno == EPERM) {
70  delete[] buffer;
71  return None();
72  }
73 
74  if (errno != ERANGE) {
75  delete[] buffer;
76  return ErrnoError("Failed to get username information");
77  }
78  // getpwnam_r set ERANGE so try again with a larger buffer.
79  size *= 2;
80  delete[] buffer;
81  }
82  }
83 
84  UNREACHABLE();
85 }

186 inline Try<Nothing> su(const std::string& user)
187 {
188  Result<gid_t> gid = os::getgid(user);
189  if (gid.isError() || gid.isNone()) {
190  return Error("Failed to getgid: " +
191  (gid.isError() ? gid.error() : "unknown user"));
192  } else if (::setgid(gid.get())) {
193  return ErrnoError("Failed to set gid");
194  }
195 
196  // Set the supplementary group list. We ignore EPERM because
197  // performing a no-op call (switching to same group) still
198  // requires being privileged, unlike 'setgid' and 'setuid'.
199  if (::initgroups(user.c_str(), gid.get()) == -1 && errno != EPERM) {
200  return ErrnoError("Failed to set supplementary groups");
201  }
202 
203  Result<uid_t> uid = os::getuid(user);
204  if (uid.isError() || uid.isNone()) {
205  return Error("Failed to getuid: " +
206  (uid.isError() ? uid.error() : "unknown user"));
207  } else if (::setuid(uid.get())) {
208  return ErrnoError("Failed to setuid");
209  }
210 
211  return Nothing();
212 }
