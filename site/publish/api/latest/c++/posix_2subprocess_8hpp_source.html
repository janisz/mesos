127 // Creates a seperate watchdog process to monitor the child process and
128 // kill it in case the parent process dies.
129 //
130 // NOTE: This function needs to be async signal safe. In fact,
131 // all the library functions we used in this function are async
132 // signal safe.
133 inline int watchdogProcess()
134 {
135 #ifdef __linux__
136  // Send SIGTERM to the current process if the parent (i.e., the
137  // slave) exits.
138  // NOTE:: This function should always succeed because we are passing
139  // in a valid signal.
140  prctl(PR_SET_PDEATHSIG, SIGTERM);
141 
142  // Put the current process into a separate process group so that
143  // we can kill it and all its children easily.
144  if (setpgid(0, 0) != 0) {
145  abort();
146  }
147 
148  // Install a SIGTERM handler which will kill the current process
149  // group. Since we already setup the death signal above, the
150  // signal handler will be triggered when the parent (e.g., the
151  // slave) exits.
152  if (os::signals::install(SIGTERM, &signalHandler) != 0) {
153  abort();
154  }
155 
156  pid_t pid = fork();
157  if (pid == -1) {
158  abort();
159  } else if (pid == 0) {
160  // Child. This is the process that is going to exec the
161  // process if zero is returned.
162 
163  // We setup death signal for the process as well in case
164  // someone, though unlikely, accidentally kill the parent of
165  // this process (the bookkeeping process).
166  prctl(PR_SET_PDEATHSIG, SIGKILL);
167 
168  // NOTE: We don't need to clear the signal handler explicitly
169  // because the subsequent 'exec' will clear them.
170  return 0;
171  } else {
172  // Parent. This is the bookkeeping process which will wait for
173  // the child process to finish.
174 
175  // Close the files to prevent interference on the communication
176  // between the slave and the child process.
177  ::close(STDIN_FILENO);
178  ::close(STDOUT_FILENO);
179  ::close(STDERR_FILENO);
180 
181  // Block until the child process finishes.
182  int status = 0;
183  if (waitpid(pid, &status, 0) == -1) {
184  abort();
185  }
186 
187  // Forward the exit status if the child process exits normally.
188  if (WIFEXITED(status)) {
189  _exit(WEXITSTATUS(status));
190  }
191 
192  abort();
193  UNREACHABLE();
194  }
195 #endif
196  return 0;
197 }

200 // The main entry of the child process.
201 //
202 // NOTE: This function has to be async signal safe.
203 inline int childMain(
204  const string& path,
205  char** argv,
206  char** envp,
207  const Setsid set_sid,
208  const InputFileDescriptors& stdinfds,
209  const OutputFileDescriptors& stdoutfds,
210  const OutputFileDescriptors& stderrfds,
211  bool blocking,
212  int pipes[2],
213  const Option<string>& working_directory,
214  const Watchdog watchdog)
215 {
216  // Close parent's end of the pipes.
217  if (stdinfds.write.isSome()) {
218  ::close(stdinfds.write.get());
219  }
220  if (stdoutfds.read.isSome()) {
221  ::close(stdoutfds.read.get());
222  }
223  if (stderrfds.read.isSome()) {
224  ::close(stderrfds.read.get());
225  }
226 
227  // Currently we will block the child's execution of the new process
228  // until all the parent hooks (if any) have executed.
229  if (blocking) {
230  ::close(pipes[1]);
231  }
232 
233  // Redirect I/O for stdin/stdout/stderr.
234  while (::dup2(stdinfds.read, STDIN_FILENO) == -1 && errno == EINTR);
235  while (::dup2(stdoutfds.write, STDOUT_FILENO) == -1 && errno == EINTR);
236  while (::dup2(stderrfds.write, STDERR_FILENO) == -1 && errno == EINTR);
237 
238  // Close the copies. We need to make sure that we do not close the
239  // file descriptor assigned to stdin/stdout/stderr in case the
240  // parent has closed stdin/stdout/stderr when calling this
241  // function (in that case, a dup'ed file descriptor may have the
242  // same file descriptor number as stdin/stdout/stderr).
243  if (stdinfds.read != STDIN_FILENO &&
244  stdinfds.read != STDOUT_FILENO &&
245  stdinfds.read != STDERR_FILENO) {
246  ::close(stdinfds.read);
247  }
248  if (stdoutfds.write != STDIN_FILENO &&
249  stdoutfds.write != STDOUT_FILENO &&
250  stdoutfds.write != STDERR_FILENO) {
251  ::close(stdoutfds.write);
252  }
253  if (stderrfds.write != STDIN_FILENO &&
254  stderrfds.write != STDOUT_FILENO &&
255  stderrfds.write != STDERR_FILENO) {
256  ::close(stderrfds.write);
257  }
258 
259  if (blocking) {
260  // Do a blocking read on the pipe until the parent signals us to
261  // continue.
262  char dummy;
263  ssize_t length;
264  while ((length = ::read(pipes[0], &dummy, sizeof(dummy))) == -1 &&
265  errno == EINTR);
266 
267  if (length != sizeof(dummy)) {
268  ABORT("Failed to synchronize with parent");
269  }
270 
271  // Now close the pipe as we don't need it anymore.
272  ::close(pipes[0]);
273  }
274 
275  // Move to a different session (and new process group) so we're
276  // independent from the caller's session (otherwise children will
277  // receive SIGHUP if the slave exits).
278  if (set_sid == SETSID) {
279  // POSIX guarantees a forked child's pid does not match any existing
280  // process group id so only a single `setsid()` is required and the
281  // session id will be the pid.
282  if (::setsid() == -1) {
283  ABORT("Failed to put child in a new session");
284  }
285  }
286 
287  if (working_directory.isSome()) {
288  if (::chdir(working_directory->c_str()) == -1) {
289  ABORT("Failed to change directory");
290  }
291  }
292 
293  // If the child process should die together with its parent we spawn a
294  // separate watchdog process which kills the child when the parent dies.
295  //
296  // NOTE: The watchdog process sets the process group id in order for it and
297  // its child processes to be killed together. We should not (re)set the sid
298  // after this.
299  if (watchdog == MONITOR) {
300  watchdogProcess();
301  }
302 
303  os::execvpe(path.c_str(), argv, envp);
304 
305  ABORT("Failed to os::execvpe on path '" + path + "': " + os::strerror(errno));
306 }

309 inline Try<pid_t> cloneChild(
310  const string& path,
311  vector<string> argv,
312  const Setsid set_sid,
313  const Option<map<string, string>>& environment,
314  const Option<lambda::function<
315  pid_t(const lambda::function<int()>&)>>& _clone,
316  const vector<Subprocess::Hook>& parent_hooks,
317  const Option<string>& working_directory,
318  const Watchdog watchdog,
319  const InputFileDescriptors stdinfds,
320  const OutputFileDescriptors stdoutfds,
321  const OutputFileDescriptors stderrfds)
322 {
323  // The real arguments that will be passed to 'os::execvpe'. We need
324  // to construct them here before doing the clone as it might not be
325  // async signal safe to perform the memory allocation.
326  char** _argv = new char*[argv.size() + 1];
327  for (size_t i = 0; i < argv.size(); i++) {
328  _argv[i] = (char*) argv[i].c_str();
329  }
330  _argv[argv.size()] = nullptr;
331 
332  // Like above, we need to construct the environment that we'll pass
333  // to 'os::execvpe' as it might not be async-safe to perform the
334  // memory allocations.
335  char** envp = os::raw::environment();
336 
337  if (environment.isSome()) {
338  // NOTE: We add 1 to the size for a `nullptr` terminator.
339  envp = new char*[environment.get().size() + 1];
340 
341  size_t index = 0;
342  foreachpair (const string& key, const string& value, environment.get()) {
343  string entry = key + "=" + value;
344  envp[index] = new char[entry.size() + 1];
345  strncpy(envp[index], entry.c_str(), entry.size() + 1);
346  ++index;
347  }
348 
349  envp[index] = nullptr;
350  }
351 
352  // Determine the function to clone the child process. If the user
353  // does not specify the clone function, we will use the default.
354  lambda::function<pid_t(const lambda::function<int()>&)> clone =
355  (_clone.isSome() ? _clone.get() : defaultClone);
356 
357  // Currently we will block the child's execution of the new process
358  // until all the `parent_hooks` (if any) have executed.
359  int pipes[2];
360  const bool blocking = !parent_hooks.empty();
361 
362  if (blocking) {
363  // We assume this should not fail under reasonable conditions so we
364  // use CHECK.
365  CHECK_EQ(0, ::pipe(pipes));
366  }
367 
368  // Now, clone the child process.
369  pid_t pid = clone(lambda::bind(
370  &childMain,
371  path,
372  _argv,
373  envp,
374  set_sid,
375  stdinfds,
376  stdoutfds,
377  stderrfds,
378  blocking,
379  pipes,
380  working_directory,
381  watchdog));
382 
383  delete[] _argv;
384 
385  // Need to delete 'envp' if we had environment variables passed to
386  // us and we needed to allocate the space.
387  if (environment.isSome()) {
388  CHECK_NE(os::raw::environment(), envp);
389 
390  // We ignore the last 'envp' entry since it is nullptr.
391  for (size_t index = 0; index < environment->size(); index++) {
392  delete[] envp[index];
393  }
394 
395  delete[] envp;
396  }
397 
398  if (pid == -1) {
399  // Save the errno as 'close' below might overwrite it.
400  ErrnoError error("Failed to clone");
401  internal::close(stdinfds, stdoutfds, stderrfds);
402 
403  if (blocking) {
404  os::close(pipes[0]);
405  os::close(pipes[1]);
406  }
407 
408  return error;
409  }
410 
411  if (blocking) {
412  os::close(pipes[0]);
413 
414  // Run the parent hooks.
415  foreach (const Subprocess::Hook& hook, parent_hooks) {
416  Try<Nothing> callback = hook.parent_callback(pid);
417 
418  // If the hook callback fails, we shouldn't proceed with the
419  // execution and hence the child process should be killed.
420  if (callback.isError()) {
421  LOG(WARNING)
422  << "Failed to execute Subprocess::Hook in parent for child '"
423  << pid << "': " << callback.error();
424 
425  os::close(pipes[1]);
426 
427  // Close the child-ends of the file descriptors that are created
428  // by this function.
429  os::close(stdinfds.read);
430  os::close(stdoutfds.write);
431  os::close(stderrfds.write);
432 
433  // Ensure the child is killed.
434  ::kill(pid, SIGKILL);
435 
436  return Error(
437  "Failed to execute Subprocess::Hook in parent for child '" +
438  stringify(pid) + "': " + callback.error());
439  }
440  }
441 
442  // Now that we've executed the parent hooks, we can signal the child to
443  // continue by writing to the pipe.
444  char dummy;
445  ssize_t length;
446  while ((length = ::write(pipes[1], &dummy, sizeof(dummy))) == -1 &&
447  errno == EINTR);
448 
449  os::close(pipes[1]);
450 
451  if (length != sizeof(dummy)) {
452  // Ensure the child is killed.
453  ::kill(pid, SIGKILL);
454 
455  // Close the child-ends of the file descriptors that are created
456  // by this function.
457  os::close(stdinfds.read);
458  os::close(stdoutfds.write);
459  os::close(stderrfds.write);
460  return Error("Failed to synchronize child process");
461  }
462  }
463 
464  return pid;
465 }
