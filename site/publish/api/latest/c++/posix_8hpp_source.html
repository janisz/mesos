37 // A basic MesosIsolatorProcess that keeps track of the pid but
38 // doesn't do any resource isolation. Subclasses must implement
39 // usage() for their appropriate resource(s).
40 class PosixIsolatorProcess : public MesosIsolatorProcess
41 {
42 public:
43  virtual process::Future<Nothing> recover(
44  const std::list<mesos::slave::ContainerState>& state,
45  const hashset<ContainerID>& orphans)
46  {
47  foreach (const mesos::slave::ContainerState& run, state) {
48  // This should (almost) never occur: see comment in
49  // PosixLauncher::recover().
50  if (pids.contains(run.container_id())) {
51  return process::Failure("Container already recovered");
52  }
53 
54  pids.put(run.container_id(), run.pid());
55 
56  process::Owned<process::Promise<mesos::slave::ContainerLimitation>>
57  promise(new process::Promise<mesos::slave::ContainerLimitation>());
58  promises.put(run.container_id(), promise);
59  }
60 
61  return Nothing();
62  }
63 
64  virtual process::Future<Option<mesos::slave::ContainerLaunchInfo>> prepare(
65  const ContainerID& containerId,
66  const mesos::slave::ContainerConfig& containerConfig)
67  {
68  if (promises.contains(containerId)) {
69  return process::Failure("Container " + stringify(containerId) +
70  " has already been prepared");
71  }
72 
73  process::Owned<process::Promise<mesos::slave::ContainerLimitation>> promise(
74  new process::Promise<mesos::slave::ContainerLimitation>());
75  promises.put(containerId, promise);
76 
77  return None();
78  }
79 
80  virtual process::Future<Nothing> isolate(
81  const ContainerID& containerId,
82  pid_t pid)
83  {
84  if (!promises.contains(containerId)) {
85  return process::Failure("Unknown container: " + stringify(containerId));
86  }
87 
88  pids.put(containerId, pid);
89 
90  return Nothing();
91  }
92 
93  virtual process::Future<mesos::slave::ContainerLimitation> watch(
94  const ContainerID& containerId)
95  {
96  if (!promises.contains(containerId)) {
97  return process::Failure("Unknown container: " + stringify(containerId));
98  }
99 
100  return promises[containerId]->future();
101  }
102 
103  virtual process::Future<Nothing> update(
104  const ContainerID& containerId,
105  const Resources& resources)
106  {
107  if (!promises.contains(containerId)) {
108  return process::Failure("Unknown container: " + stringify(containerId));
109  }
110 
111  // No resources are actually isolated so nothing to do.
112  return Nothing();
113  }
114 
115  virtual process::Future<Nothing> cleanup(const ContainerID& containerId)
116  {
117  if (!promises.contains(containerId)) {
118  return process::Failure("Unknown container: " + stringify(containerId));
119  }
120 
121  // TODO(idownes): We should discard the container's promise here to signal
122  // to anyone that holds the future from watch().
123  promises.erase(containerId);
124 
125  pids.erase(containerId);
126 
127  return Nothing();
128  }
129 
130 protected:
131  hashmap<ContainerID, pid_t> pids;
132  hashmap<ContainerID,
133  process::Owned<process::Promise<mesos::slave::ContainerLimitation>>>
134  promises;
135 };

149  virtual process::Future<ResourceStatistics> usage(
150  const ContainerID& containerId)
151  {
152  if (!pids.contains(containerId)) {
153  LOG(WARNING) << "No resource usage for unknown container '"
154  << containerId << "'";
155  return ResourceStatistics();
156  }
157 
158  // Use 'mesos-usage' but only request 'cpus_' values.
159  Try<ResourceStatistics> usage =
160  mesos::internal::usage(pids.get(containerId).get(), false, true);
161  if (usage.isError()) {
162  return process::Failure(usage.error());
163  }
164  return usage.get();
165  }
166 
167 protected:
168  PosixCpuIsolatorProcess() {}
169 };
