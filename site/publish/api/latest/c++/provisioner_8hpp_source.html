73  // NOTE: Made 'virtual' for mocking and testing.
74  virtual ~Provisioner();
75 
76  // Recover root filesystems for containers from the run states and
77  // the orphan containers (known to the launcher but not known to the
78  // slave) detected by the launcher. This function is also
79  // responsible for cleaning up any intermediate artifacts (e.g.
80  // directories) to not leak anything.
81  virtual process::Future<Nothing> recover(
82  const std::list<mesos::slave::ContainerState>& states,
83  const hashset<ContainerID>& orphans);
84 
85  // Provision a root filesystem for the container using the specified
86  // image and return the absolute path to the root filesystem.
87  virtual process::Future<ProvisionInfo> provision(
88  const ContainerID& containerId,
89  const Image& image);
90 
91  // Destroy a previously provisioned root filesystem. Assumes that
92  // all references (e.g., mounts, open files) to the provisioned
93  // filesystem have been removed. Return false if there is no
94  // provisioned root filesystem for the given container.
95  virtual process::Future<bool> destroy(const ContainerID& containerId);
96 
97 protected:
98  Provisioner() {} // For creating mock object.
99 
100 private:
101  Provisioner(const Provisioner&) = delete; // Not copyable.
102  Provisioner& operator=(const Provisioner&) = delete; // Not assignable.
103 
104  process::Owned<ProvisionerProcess> process;
105 };
