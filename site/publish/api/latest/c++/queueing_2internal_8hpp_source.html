73 // Encodes a queueing discipline (in our representation) to a libnl
74 // queueing discipline (rtnl_qdisc). We use template here so that it
75 // works for any type of queueing discipline.
76 template <typename Config>
77 Try<Netlink<struct rtnl_qdisc>> encodeDiscipline(
78  const Netlink<struct rtnl_link>& link,
79  const Discipline<Config>& discipline)
80 {
81  struct rtnl_qdisc* q = rtnl_qdisc_alloc();
82  if (q == nullptr) {
83  return Error("Failed to allocate a libnl qdisc");
84  }
85 
86  Netlink<struct rtnl_qdisc> qdisc(q);
87 
88  rtnl_tc_set_link(TC_CAST(qdisc.get()), link.get());
89  rtnl_tc_set_parent(TC_CAST(qdisc.get()), discipline.parent.get());
90 
91  if (discipline.handle.isSome()) {
92  rtnl_tc_set_handle(TC_CAST(qdisc.get()), discipline.handle.get().get());
93  }
94 
95  int error = rtnl_tc_set_kind(TC_CAST(qdisc.get()), discipline.kind.c_str());
96  if (error != 0) {
97  return Error(
98  "Failed to set the kind of the queueing discipline: " +
99  std::string(nl_geterror(error)));
100  }
101 
102  // Perform queue discipline specific encoding.
103  Try<Nothing> encoding = encode(qdisc, discipline.config);
104  if (encoding.isError()) {
105  return Error(
106  "Failed to encode the queueing discipline: " +
107  encoding.error());
108  }
109 
110  return qdisc;
111 }

117 // Returns all the libnl queue discipline (rtnl_qdisc) on the link.
118 inline Try<std::vector<Netlink<struct rtnl_qdisc>>> getQdiscs(
119  const Netlink<struct rtnl_link>& link)
120 {
121  Try<Netlink<struct nl_sock>> socket = routing::socket();
122  if (socket.isError()) {
123  return Error(socket.error());
124  }
125 
126  // Dump all the queueing discipline from kernel.
127  struct nl_cache* c = nullptr;
128  int error = rtnl_qdisc_alloc_cache(socket.get().get(), &c);
129  if (error != 0) {
130  return Error(
131  "Failed to get queueing discipline info from kernel: " +
132  std::string(nl_geterror(error)));
133  }
134 
135  Netlink<struct nl_cache> cache(c);
136 
137  std::vector<Netlink<struct rtnl_qdisc>> results;
138 
139  for (struct nl_object* o = nl_cache_get_first(cache.get());
140  o != nullptr; o = nl_cache_get_next(o)) {
141  if (rtnl_tc_get_ifindex(TC_CAST(o)) == rtnl_link_get_ifindex(link.get())) {
142  // NOTE: We increment the reference counter here because 'cache'
143  // will be freed when this function finishes and we want this
144  // object's life to be longer than this function.
145  nl_object_get(o);
146 
147  results.push_back(Netlink<struct rtnl_qdisc>((struct rtnl_qdisc*) o));
148  }
149  }
150 
151  return results;
152 }

155 // Returns the libnl queueing discipline (rtnl_qdisc) attached to the
156 // given parent that matches the specified queueing discipline kind on
157 // the link. Return None if no match has been found.
158 inline Result<Netlink<struct rtnl_qdisc>> getQdisc(
159  const Netlink<struct rtnl_link>& link,
160  const Handle& parent,
161  const std::string& kind)
162 {
163  Try<std::vector<Netlink<struct rtnl_qdisc>>> qdiscs = getQdiscs(link);
164  if (qdiscs.isError()) {
165  return Error(qdiscs.error());
166  }
167 
168  foreach (const Netlink<struct rtnl_qdisc>& qdisc, qdiscs.get()) {
169  if (rtnl_tc_get_parent(TC_CAST(qdisc.get())) == parent.get() &&
170  rtnl_tc_get_kind(TC_CAST(qdisc.get())) == kind) {
171  return qdisc;
172  }
173  }
174 
175  return None();
176 }

206 // Creates a new queueing discipline on the link. Returns false if a
207 // queueing discipline attached to the same parent with the same
208 // configuration already exists on the link. We use template here so
209 // that it works for any type of queueing discipline.
210 template <typename Config>
211 Try<bool> create(
212  const std::string& _link,
213  const Discipline<Config>& discipline)
214 {
215  Result<Netlink<struct rtnl_link>> link = link::internal::get(_link);
216  if (link.isError()) {
217  return Error(link.error());
218  } else if (link.isNone()) {
219  return Error("Link '" + _link + "' is not found");
220  }
221 
222  Try<Netlink<struct rtnl_qdisc>> qdisc =
223  encodeDiscipline(link.get(), discipline);
224 
225  if (qdisc.isError()) {
226  return Error("Failed to encode the queueing discipline: " + qdisc.error());
227  }
228 
229  Try<Netlink<struct nl_sock>> socket = routing::socket();
230  if (socket.isError()) {
231  return Error(socket.error());
232  }
233 
234  // The flag NLM_F_EXCL tells libnl that if the qdisc already exists,
235  // this function should return error.
236  int error = rtnl_qdisc_add(
237  socket.get().get(),
238  qdisc.get().get(),
239  NLM_F_CREATE | NLM_F_EXCL);
240 
241  if (error != 0) {
242  if (error == -NLE_EXIST) {
243  return false;
244  }
245  return Error(
246  "Failed to add a queueing discipline to the link: " +
247  std::string(nl_geterror(error)));
248  }
249 
250  return true;
251 }

292 // Returns the set of common Traffic Control statistics for the
293 // queueing discipline on the link, None() if the link or qdisc does
294 // not exist or an error if we cannot cannot determine the result.
295 inline Result<hashmap<std::string, uint64_t>> statistics(
296  const std::string& _link,
297  const Handle& parent,
298  const std::string& kind)
299 {
300  Result<Netlink<struct rtnl_link>> link = link::internal::get(_link);
301  if (link.isError()) {
302  return Error(link.error());
303  } else if (link.isNone()) {
304  return None();
305  }
306 
307  Result<Netlink<struct rtnl_qdisc>> qdisc = getQdisc(link.get(), parent, kind);
308  if (qdisc.isError()) {
309  return Error(qdisc.error());
310  } else if (qdisc.isNone()) {
311  return None();
312  }
313 
314  hashmap<std::string, uint64_t> results;
315  char name[32];
316 
317  // NOTE: We use '<=' here because RTNL_TC_STATS_MAX is set to be the
318  // value of the last enum entry.
319  for (size_t i = 0; i <= static_cast<size_t>(RTNL_TC_STATS_MAX); i++) {
320  if (rtnl_tc_stat2str(static_cast<rtnl_tc_stat>(i), name, sizeof(name))) {
321  results[name] = rtnl_tc_get_stat(
322  TC_CAST(qdisc.get().get()),
323  static_cast<rtnl_tc_stat>(i));
324  }
325  }
326  return results;
327 }
