40 // Defines an abstraction for operations that can be applied on the
41 // Registry.
42 // TODO(xujyan): Make Operation generic so that we can apply them
43 // against a generic "batch operation applier" abstraction, see TODO
44 // below for more details.
45 class Operation : public process::Promise<bool>
46 {
47 public:
48  Operation() : success(false) {}
49  virtual ~Operation() {}
50 
51  // Attempts to invoke the operation on the registry object.
52  // Aided by accumulator(s):
53  // slaveIDs - is the set of registered slaves.
54  //
55  // NOTE: the "strict" parameter only applies to operations that
56  // affect slaves (i.e. registration). See Flags::registry_strict
57  // in master/flags.cpp for more information.
58  //
59  // Returns whether the operation mutates 'registry', or an error if
60  // the operation cannot be applied successfully.
61  Try<bool> operator()(
62  Registry* registry,
63  hashset<SlaveID>* slaveIDs,
64  bool strict)
65  {
66  const Try<bool> result = perform(registry, slaveIDs, strict);
67 
68  success = !result.isError();
69 
70  return result;
71  }
72 
73  // Sets the promise based on whether the operation was successful.
74  bool set() { return process::Promise<bool>::set(success); }
75 
76 protected:
77  virtual Try<bool> perform(
78  Registry* registry,
79  hashset<SlaveID>* slaveIDs,
80  bool strict) = 0;
81 
82 private:
83  bool success;
84 };

87 // TODO(xujyan): Add a generic abstraction for applying batches of
88 // operations against State Variables. The Registrar and other
89 // components could leverage this. This abstraction would be
90 // templatized to take the type, along with any accumulators:
91 // template <typename T,
92 // typename X = None,
93 // typename Y = None,
94 // typename Z = None>
95 // T: the data type that the batch operations can be applied on.
96 // X, Y, Z: zero to 3 generic accumulators that facilitate the batch
97 // of operations.
98 // This allows us to reuse the idea of "doing batches of operations"
99 // on other potential new state variables (i.e. repair state, offer
100 // reservations, etc).
101 class Registrar
102 {
103 public:
104  // If flags.registry_strict is true, all operations will be
105  // permitted.
106  Registrar(const Flags& flags,
107  mesos::state::protobuf::State* state,
108  const Option<std::string>& authenticationRealm = None());
109  ~Registrar();
110 
111  // Recovers the Registry, persisting the new Master information.
112  // The Registrar must be recovered to allow other operations to
113  // proceed.
114  // TODO(bmahler): Consider a "factory" for constructing the
115  // Registrar, to eliminate the need for passing 'MasterInfo'.
116  // This is required as the Registrar is injected into the Master,
117  // and therefore MasterInfo is unknown during construction.
118  process::Future<Registry> recover(const MasterInfo& info);
119 
120  // Applies an operation on the Registry.
121  // Returns:
122  // true if the operation is permitted.
123  // false if the operation is not permitted.

125  // or recovery failed.
126  process::Future<bool> apply(process::Owned<Operation> operation);
127 
128  // Gets the pid of the underlying process.

132 private:
