27 // We pass this struct to `DeviceIoControl` to get information about a reparse
28 // point (including things like whether it's a symlink). It is normally part of
29 // the Device Driver Kit (DDK), specifically `nitfs.h`, but rather than taking
30 // a dependency on the DDK, we choose to simply copy the struct here. This is a
31 // well-worn path used by Boost FS[1], among others. See documentation
32 // here[2][3].
33 //
34 // [1] http://www.boost.org/doc/libs/1_46_1/libs/filesystem/v3/src/operations.cpp
35 // [2] https://msdn.microsoft.com/en-us/library/cc232007.aspx
36 // [3] https://msdn.microsoft.com/en-us/library/cc232005.aspx
37 //
38 // We are declaring this structure (and the REPARSE_DATA_BUFFER_HEADER_SIZE
39 // macro right below it in the global namespace, to be consistent with the
40 // original Windows DDK declarations.
41 typedef struct _REPARSE_DATA_BUFFER
42 {
43  // Describes, among other things, which type of reparse point this is (e.g.,
44  // a symlink).
45  ULONG ReparseTag;
46  // Size in bytes of common portion of the `REPARSE_DATA_BUFFER`.
47  USHORT ReparseDataLength;
48  // Unused. Ignore.
49  USHORT Reserved;
50  union
51  {
52  // Holds symlink data.
53  struct
54  {
55  // Byte offset in `PathBuffer` where the substitute name begins.
56  // Calculated as an offset from 0.
57  USHORT SubstituteNameOffset;
58  // Length in bytes of the substitute name.
59  USHORT SubstituteNameLength;
60  // Byte offset in `PathBuffer` where the print name begins. Calculated as
61  // an offset from 0.
62  USHORT PrintNameOffset;
63  // Length in bytes of the print name.
64  USHORT PrintNameLength;
65  // Indicates whether symlink is absolute or relative. If flags containing
66  // `SYMLINK_FLAG_RELATIVE`, then the substitute name is a relative
67  // symlink.
68  ULONG Flags;

70  // this is followed in memory by the rest of the path string. The "path
71  // string" itself is a unicode char array containing both substitute name
72  // and print name. They can occur in any order. Use the offset and length
73  // of each in this struct to calculate where each starts and ends.
74  //
75  // [1] https://msdn.microsoft.com/en-us/library/windows/hardware/ff552012(v=vs.85).aspx

112 // Checks file/folder attributes for a path to see if the reparse point
113 // attribute is set; this indicates whether the path points at a reparse point,
114 // rather than a "normal" file or folder.
115 inline Try<bool> reparse_point_attribute_set(const std::string& absolute_path)
116 {
117  const DWORD attributes = ::GetFileAttributes(absolute_path.c_str());
118  if (attributes == INVALID_FILE_ATTRIBUTES) {
119  return WindowsError(
120  "Failed to get attributes for file '" + absolute_path + "'");
121  }
122 
123  return (attributes & FILE_ATTRIBUTE_REPARSE_POINT) != 0;
124 }

127 // Attempts to extract symlink data out of a `REPARSE_DATA_BUFFER` (which could
128 // hold other things, e.g., mount point data).
129 inline Try<SymbolicLink> build_symbolic_link(const REPARSE_DATA_BUFFER& data)
130 {
131  const bool is_symLink = (data.ReparseTag & IO_REPARSE_TAG_SYMLINK) != 0;
132  if (!is_symLink) {
133  return Error("Data buffer is not a symlink");
134  }
135 
136  // NOTE: This buffer is not null terminated.
137  const WCHAR* substitute_name =
138  data.SymbolicLinkReparseBuffer.PathBuffer +
139  data.SymbolicLinkReparseBuffer.SubstituteNameOffset / sizeof(WCHAR);
140  const size_t substitute_name_length =
141  data.SymbolicLinkReparseBuffer.SubstituteNameLength / sizeof(WCHAR);
142 
143  // NOTE: This buffer is not null terminated.
144  const WCHAR* print_name =
145  data.SymbolicLinkReparseBuffer.PathBuffer +
146  data.SymbolicLinkReparseBuffer.PrintNameOffset / sizeof(WCHAR);
147  const size_t print_name_length =
148  data.SymbolicLinkReparseBuffer.PrintNameLength / sizeof(WCHAR);
149 
150  return SymbolicLink{
151  std::wstring(substitute_name, substitute_name_length),
152  std::wstring(print_name, print_name_length),
153  data.SymbolicLinkReparseBuffer.Flags};
154 }

157 // Attempts to get a file or folder handle for an absolute path, and does not
158 // follow symlinks. That is, if the path points at a symlink, the handle will
159 // refer to the symlink rather than the file or folder the symlink points at.
160 inline Try<SharedHandle> get_handle_no_follow(const std::string& absolute_path)
161 {
162  struct _stat absolute_path_stat;
163  bool resolved_path_is_directory = false;
164  if (::_stat(absolute_path.c_str(), &absolute_path_stat) == 0) {
165  resolved_path_is_directory = S_ISDIR(absolute_path_stat.st_mode);
166  } else {
167  return ErrnoError("'_stat' failed on path '" + absolute_path + "'");
168  }
169 
170  // NOTE: According to the `CreateFile` documentation[1], the `OPEN_EXISTING`
171  // and `FILE_FLAG_OPEN_REPARSE_POINT` flags need to be used when getting a
172  // handle for the symlink.
173  //
174  // Note also that `CreateFile` will appropriately generate a handle for
175  // either a folder or a file, as long as the appropriate flag is being set:
176  // `FILE_FLAG_BACKUP_SEMANTICS` or `FILE_FLAG_OPEN_REPARSE_POINT`.
177  //
178  // The `FILE_FLAG_BACKUP_SEMANTICS` flag is being set whenever the target is
179  // a directory. According to MSDN[1]: "You must set this flag to obtain a
180  // handle to a directory. A directory handle can be passed to some functions
181  // instead of a file handle". More `FILE_FLAG_BACKUP_SEMANTICS` documentation
182  // can be found in MSDN[2].
183  //
184  // The `GENERIC_READ` flag is being used because it's the most common way of
185  // opening a file for reading only. The `FILE_SHARE_READ` allows other
186  // processes to read the file at the same time. MSDN[1] provides a more
187  // detailed explanation of these flags.
188  //
189  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/aa363858(v=vs.85).aspx
190  // [2] https://msdn.microsoft.com/en-us/library/windows/desktop/aa364399(v=vs.85).aspx
191  const DWORD access_flags = resolved_path_is_directory
192  ? (FILE_FLAG_OPEN_REPARSE_POINT | FILE_FLAG_BACKUP_SEMANTICS)
193  : FILE_FLAG_OPEN_REPARSE_POINT;
194 
195  const HANDLE handle = ::CreateFile(
196  absolute_path.c_str(),
197  GENERIC_READ, // Open the file for reading only.
198  FILE_SHARE_READ, // Just reading this file, allow others to do the same.
199  nullptr, // Ignored.
200  OPEN_EXISTING, // Open existing symlink.
201  access_flags, // Open symlink, not the file it points to.
202  nullptr); // Ignored.
203 
204  if (handle == INVALID_HANDLE_VALUE) {
205  return WindowsError(
206  "'internal::windows::get_handle_no_follow': 'CreateFile' call failed "
207  "at path '" + absolute_path + "'");
208  }
209 
210  return SharedHandle(handle, CloseHandle);
211 }

214 // Attempts to get the symlink data for a file or folder handle.
215 inline Try<SymbolicLink> get_symbolic_link_data(const HANDLE handle)
216 {
217  // To get the symlink data, we call `DeviceIoControl`. This function is part
218  // of the Device Driver Kit (DDK)[1] and, along with `FSCTL_GET_REPARSE_POINT`
219  // is used to emit information about reparse points (and, thus, symlinks,
220  // since symlinks are implemented with reparse points). This technique is
221  // being used in Boost FS code as well[2].
222  //
223  // Summarized, the documentation tells us that we need to pass in
224  // `FSCTL_GET_REPARSE_POINT` to get the function to populate a
225  // `REPARSE_DATA_BUFFER` struct with data about a reparse point.
226  // The `REPARSE_DATA_BUFFER` struct is defined in a DDK header file,
227  // so to avoid bringing in a multitude of DDK headers we take a cue from
228  // Boost FS, and copy the struct into this header (see above).
229  //
230  // Finally, for context, it may be worth looking at the MSDN
231  // documentation[3] for `DeviceIoControl` itself.
232  //
233  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/aa364571(v=vs.85).aspx
234  // [2] https://svn.boost.org/trac/boost/ticket/4663
235  // [3] https://msdn.microsoft.com/en-us/library/windows/desktop/aa363216(v=vs.85).aspx
236  const size_t reparse_point_data_size = MAXIMUM_REPARSE_DATA_BUFFER_SIZE;
237  BYTE buffer[reparse_point_data_size];
238  REPARSE_DATA_BUFFER* reparse_point_data =
239  reinterpret_cast<REPARSE_DATA_BUFFER*>(buffer);
240 
241  DWORD ignored = 0;
242 
243  // The semantics of this function are: get the reparse data associated with
244  // the `handle` of some open directory or file, and that data in
245  // `reparse_point_data`.
246  const BOOL reparse_data_obtained = ::DeviceIoControl(
247  handle, // Handle to file or directory.
248  FSCTL_GET_REPARSE_POINT, // Gets reparse point data for file/folder.
249  nullptr, // Ignored.
250  0, // Ignored.
251  reparse_point_data,
252  reparse_point_data_size,
253  &ignored, // Ignored.
254  nullptr); // Ignored.
255 
256  if (!reparse_data_obtained) {
257  return WindowsError(
258  "'internal::windows::get_symbolic_link_data': 'DeviceIoControl' call "
259  "failed");
260  }
261 
262  return build_symbolic_link(*reparse_point_data);
263 }

266 // Creates a reparse point with the specified target. The target can be either
267 // a file (in which case a junction is created), or a folder (in which case a
268 // mount point is created).
269 //
270 // Calling this function results in a temporary elevation of the process token's
271 // privileges (if those privileges are not already held), to allow for junction
272 // or mount point creation. This operation is gated by a static mutex, which
273 // makes it thread-safe.
274 inline Try<Nothing> create_symbolic_link(
275  const std::string& target,
276  const std::string& reparse_point)
277 {
278  // Normalize input paths.
279  const Result<std::string> real_reparse_point_path =
280  os::realpath(reparse_point);
281 
282  const Result<std::string> real_target_path = os::realpath(target);
283 
284  if (!real_reparse_point_path.isSome()) {
285  return Error(
286  "Failed to get realpath for '" +
287  reparse_point + "': " + (real_reparse_point_path.isError() ?
288  real_reparse_point_path.error() : "No such directory"));
289  }
290 
291  if (!real_target_path.isSome()) {
292  return Error(
293  "Failed to get realpath for '" +
294  target + "': " + (real_target_path.isError() ?
295  real_target_path.error() : "No such directory"));
296  }
297 
298  const std::string& absolute_target_path = real_target_path.get();
299 
300  // Determine if target is a folder or a file. This makes a difference
301  // in the way we call `create_symbolic_link`.
302  struct _stat absolute_target_path_stat;
303  if (::_stat(absolute_target_path.c_str(), &absolute_target_path_stat) != 0) {
304  return ErrnoError("'_stat' failed on path '" + absolute_target_path + "'");
305  }
306 
307  const bool target_is_folder = S_ISDIR(absolute_target_path_stat.st_mode);
308 
309  // Bail out if target is already a reparse point.
310  Try<bool> attribute_set = reparse_point_attribute_set(absolute_target_path);
311  if (!attribute_set.isSome()) {
312  return Error(
313  "Could not get reparse point attribute for '" + absolute_target_path +
314  "'" + (attribute_set.isError() ? ": " + attribute_set.error() : ""));
315  }
316 
317  if (attribute_set.get()) {
318  return Error(
319  "Path '" + absolute_target_path + "' is already a reparse point");
320  }
321 
322  // `CreateSymbolicLink` adjusts the process token's privileges to allow for
323  // symlink creation. MSDN[1] makes no guarantee when it comes to the thread
324  // safety of this operation, so we are making use of a mutex to prevent
325  // multiple concurrent calls.
326  //
327  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/aa363866(v=vs.85).aspx
328  static std::mutex adjust_privileges_mutex;
329  synchronized(adjust_privileges_mutex) {
330  if (!::CreateSymbolicLink(
331  reparse_point.c_str(), // path to the symbolic link
332  target.c_str(), // symlink target
333  target_is_folder ? SYMBOLIC_LINK_FLAG_DIRECTORY : 0)) {
334  return WindowsError(
335  "'internal::windows::create_symbolic_link': 'CreateSymbolicLink' "
336  "call failed");
337  }
338  }
339 
340  return Nothing();
341 }
