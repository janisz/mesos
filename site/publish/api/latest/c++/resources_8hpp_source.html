42 // Resources come in three types: scalar, ranges, and sets. These are
43 // represented using protocol buffers. To make manipulation of
44 // resources easier within the Mesos core and for scheduler writers,
45 // we provide generic overloaded operators (see below) as well as a
46 // general Resources class that encapsulates a collection of protocol
47 // buffer Resource objects. The Resources class also provides a few
48 // static routines to allow parsing resources (e.g., from the command
49 // line), as well as determining whether or not a Resource object is
50 // valid. Note that many of these operations have not been optimized
51 // but instead just written for correct semantics.

166  // Returns the summed up Resources given a hashmap<Key, Resources>.
167  //
168  // NOTE: While scalar resources such as "cpus" sum correctly,
169  // non-scalar resources such as "ports" do not.
170  // e.g. "cpus:2" + "cpus:1" = "cpus:3"
171  // "ports:[0-100]" + "ports:[0-100]" = "ports:[0-100]"
172  //
173  // TODO(mpark): Deprecate this function once we introduce the
174  // concept of "cluster-wide" resources which provides correct
175  // semantics for summation over all types of resources. (e.g.
176  // non-scalar)
177  template <typename Key>
178  static Resources sum(const hashmap<Key, Resources>& _resources)
179  {
180  Resources result;
181 
182  foreachvalue (const Resources& resources, _resources) {
183  result += resources;
184  }
185 
186  return result;
187  }

255  // Returns a Resources object that contains all the scalar resources

281  // Certain offer operations (e.g., RESERVE, UNRESERVE, CREATE or

288  template <typename Iterable>
289  Try<Resources> apply(const Iterable& operations) const
290  {
291  Resources result = *this;
292 
293  foreach (const Offer::Operation& operation, operations) {
294  Try<Resources> transformed = result.apply(operation);
295  if (transformed.isError()) {
296  return Error(transformed.error());
297  }
298 
299  result = transformed.get();
300  }
301 
302  return result;
303  }
