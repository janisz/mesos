45  // Add a host directory or file to the root filesystem. Note that
46  // the host directory or file needs to be an absolute path.
47  Try<Nothing> add(const std::string& path)
48  {
49  if (!os::exists(path)) {
50  return Error("File or directory not found on the host");
51  }
52 
53  if (!strings::startsWith(path, "/")) {
54  return Error("Not an absolute path");
55  }
56 
57  std::string dirname = Path(path).dirname();
58  std::string target = path::join(root, dirname);
59 
60  if (!os::exists(target)) {
61  Try<Nothing> mkdir = os::mkdir(target);
62  if (mkdir.isError()) {
63  return Error("Failed to create directory in rootfs: " +
64  mkdir.error());
65  }
66  }
67 
68  // TODO(jieyu): Make sure 'path' is not under 'root'.
69 
70  if (os::stat::isdir(path)) {
71  if (os::system("cp -r '" + path + "' '" + target + "'") != 0) {
72  return ErrnoError("Failed to copy '" + path + "' to rootfs");
73  }
74  } else {
75  if (os::system("cp '" + path + "' '" + target + "'") != 0) {
76  return ErrnoError("Failed to copy '" + path + "' to rootfs");
77  }
78  }
79 
80  return Nothing();
81  }
82 
83  const std::string root;
84 
85 protected:
86  Rootfs(const std::string& _root) : root(_root) {}

93  static Try<process::Owned<Rootfs>> create(const std::string& root)
94  {
95  process::Owned<Rootfs> rootfs(new LinuxRootfs(root));
96 
97  if (!os::exists(root)) {
98  Try<Nothing> mkdir = os::mkdir(root);
99  if (mkdir.isError()) {
100  return Error("Failed to create root directory: " + mkdir.error());
101  }
102  }
103 
104  std::vector<std::string> directories = {
105  "/bin",
106  "/lib",
107  "/lib64",
108  "/etc"
109  };
110 
111  foreach (const std::string& directory, directories) {
112  // Some linux distros are moving all binaries and libraries to
113  // /usr, in which case /bin, /lib, and /lib64 will be symlinks
114  // to their equivalent directories in /usr.
115  Result<std::string> realpath = os::realpath(directory);
116  if (!realpath.isSome()) {
117  return Error("Failed to get realpath for '" +
118  directory + "': " + (realpath.isError() ?
119  realpath.error() : "No such directory"));
120  }
121 
122  Try<Nothing> result = rootfs->add(realpath.get());
123  if (result.isError()) {
124  return Error("Failed to add '" + realpath.get() +
125  "' to rootfs: " + result.error());
126  }
127 
128  if (os::stat::islink(directory)) {
129  result = rootfs->add(directory);
130  if (result.isError()) {
131  return Error("Failed to add '" + directory + "' to rootfs: " +
132  result.error());
133  }
134  }
135  }
136 
137  directories = {
138  "/proc",
139  "/sys",
140  "/dev",
141  "/tmp"
142  };
143 
144  foreach (const std::string& directory, directories) {
145  Try<Nothing> mkdir = os::mkdir(path::join(root, directory));
146  if (mkdir.isError()) {
147  return Error("Failed to create '" + directory +
148  "' in rootfs: " + mkdir.error());
