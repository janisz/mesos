30 // Provides an abstraction that serializes the execution of a sequence
31 // of callbacks.
32 class Sequence
33 {
34 public:
35  Sequence();
36  ~Sequence();
37 
38  // Registers a callback that will be invoked when all the futures
39  // returned by the previously registered callbacks are in
40  // non-pending status (i.e., ready, discarded or failed). Due to
41  // these semantics, we should avoid registering a callback which
42  // returns a future that could be in non-pending status for a long
43  // time because it will prevent all the subsequent callbacks from
44  // being invoked. This is analogous to the requirement that we want
45  // to avoid invoking blocking function in a libprocess handler. A
46  // user is allowed to cancel a registered callback by discarding the
47  // returned future. Other callbacks in this sequence will NOT be
48  // affected. The subsequent callbacks will not be invoked until the
49  // future is actually DISCARDED.
50  template <typename T>
51  Future<T> add(const lambda::function<Future<T>()>& callback);
52 
53 private:
54  // Not copyable, not assignable.
55  Sequence(const Sequence&);
56  Sequence& operator=(const Sequence&);
57 
58  SequenceProcess* process;
59 };

62 class SequenceProcess : public Process<SequenceProcess>
63 {
64 public:
65  SequenceProcess() : last(Nothing()) {}
66 
67  template <typename T>
68  Future<T> add(const lambda::function<Future<T>()>& callback)
69  {
70  // This is the future that is used to notify the next callback
71  // (denoted by 'N' in the following graph).
72  Owned<Promise<Nothing> > notifier(new Promise<Nothing>());
73 
74  // This is the future that will be returned to the user (denoted
75  // by 'F' in the following graph).
76  Owned<Promise<T> > promise(new Promise<T>());
77 
78  // We use a graph to show how we hook these futures. Each box in
79  // the graph represents a future. As mentioned above, 'F' denotes
80  // a future that will be returned to the user, and 'N' denotes a
81  // future that is used for notifying the next callback. Each arrow
82  // represents a "notification" relation. We will explain in detail
83  // what "notification" means in the following.
84  //
85  // 'last' 'last' 'last'
86  // | | |
87  // v v v
88  // +---+ +---+ +---+ +---+ +---+ +---+
89  // | N | | N |--+ | N | | N |--+ | N |--+ | N |
90  // +---+ +---+ | +---+ +---+ | +---+ | +---+
91  // ==> | ^ ==> | ^ | ^
92  // | | | | | |
93  // | +---+ | +---+ | +---+
94  // +-->| F | +-->| F | +-->| F |
95  // +---+ +---+ +---+
96  //
97  // Initial => Added one callback => Added two callbacks.
98 
99  // Setup the "notification" from 'F' to 'N' so that when a
100  // callback is done, signal the notifier ('N').
101  promise->future().onAny(lambda::bind(&completed, notifier));
102 
103  // Setup the "notification" from previous 'N' to 'F' so that when
104  // a notifier ('N') is set (indicating the previous callback has
105  // completed), invoke the next callback ('F') in the sequence.
106  last.onAny(lambda::bind(&notified<T>, promise, callback));
107 
108  // In the following, we setup the hooks so that if this sequence
109  // process is being terminated, all pending callbacks will be
110  // discarded. We use weak futures here to avoid cyclic dependencies.
111 
112  // Discard the future associated with this notifier.
113  notifier->future().onDiscard(
114  lambda::bind(
115  &internal::discard<T>,
116  WeakFuture<T>(promise->future())));
117 
118  // Discard the notifier associated with the previous future.
119  notifier->future().onDiscard(
120  lambda::bind(
121  &internal::discard<Nothing>,
122  WeakFuture<Nothing>(last)));
123 
124  // Update the 'last'.
125  last = notifier->future();
126 
127  return promise->future();
128  }
129 
130 protected:
131  virtual void finalize()
132  {
133  last.discard();
134 
135  // TODO(jieyu): Do we need to wait for the future of the last
136  // callback to be in DISCARDED state?
137  }
138 
139 private:
140  // Invoked when a callback is done.
141  static void completed(Owned<Promise<Nothing> > notifier)
142  {
143  notifier->set(Nothing());
144  }
145 
146  // Invoked when a notifier is set.
147  template <typename T>
148  static void notified(
149  Owned<Promise<T> > promise,
150  const lambda::function<Future<T>()>& callback)
151  {
152  if (promise->future().hasDiscard()) {
153  // The user has shown the intention to discard this callback
154  // (i.e., by calling future.discard()). As a result, we will
155  // just skip this callback.
156  promise->discard();
157  } else {
158  promise->associate(callback());
159  }
160  }
161 
162  Future<Nothing> last;
163 };
