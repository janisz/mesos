49 // An image store abstraction that "stores" images. It serves as a
50 // read-through cache (cache misses are fetched remotely and
51 // transparently) for images.
52 class Store
53 {
54 public:

56  const Flags& flags);
57 
58  virtual ~Store() {}
59 
60  virtual process::Future<Nothing> recover() = 0;

62  // Get the specified image (and all its recursive dependencies) as a
63  // list of rootfs layers in the topological order (dependencies go
64  // before dependents in the list). The images required to build this
65  // list are either retrieved from the local cache or fetched
66  // remotely.
67  //
68  // NOTE: The returned list should not have duplicates. e.g., in the
69  // following scenario the result should be [C, B, D, A] (B before D
70  // in this example is decided by the order in which A specifies its
71  // dependencies).
72  //
73  // A --> B --> C
74  // | ^
75  // |---> D ----|
76  //
77  // The returned future fails if the requested image or any of its
78  // dependencies cannot be found or failed to be fetched.
79  virtual process::Future<ImageInfo> get(const Image& image) = 0;
80 };
