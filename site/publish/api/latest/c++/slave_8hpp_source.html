735  // Sends a message to the connected executor.
736  template <typename Message>
737  void send(const Message& message)
738  {
739  if (state == REGISTERING || state == TERMINATED) {
740  LOG(WARNING) << "Attempting to send message to disconnected"
741  << " executor " << *this << " in state " << state;
742  }
743
