50 // Deserializes a string message into a protobuf message based on the
51 // HTTP content type.
52 template <typename Message>
53 Try<Message> deserialize(
54  ContentType contentType,
55  const std::string& body)
56 {
57  switch (contentType) {
58  case ContentType::PROTOBUF: {
59  Message message;
60  if (!message.ParseFromString(body)) {
61  return Error("Failed to parse body into a protobuf object");
62  }
63  return message;
64  }
65  case ContentType::JSON: {
66  Try<JSON::Value> value = JSON::parse(body);
67  if (value.isError()) {
68  return Error("Failed to parse body into JSON: " + value.error());
69  }
70 
71  return ::protobuf::parse<Message>(value.get());
72  }
73  }
74 
75  UNREACHABLE();
76 }
