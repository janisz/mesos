91 // When the same variable is listed multiple times,
92 // uses only the last value.
93 template <>
94 inline Try<hashmap<std::string, std::string>> parse(const std::string& value)
95 {
96  // Convert from string or file to JSON.
97  Try<JSON::Object> json = parse<JSON::Object>(value);
98  if (json.isError()) {
99  return Error(json.error());
100  }
101 
102  // Convert from JSON to Hashmap.
103  hashmap<std::string, std::string> map;
104  foreachpair (const std::string& key,
105  const JSON::Value& value,
106  json.get().values) {
107  if (!value.is<JSON::String>()) {
108  return Error(
109  "The value of key '" + key + "' in '" + stringify(json.get()) + "'"
110  " is not a string");
111  }
112 
113  map[key] = value.as<JSON::String>().value;
114  }
115  return map;
116 }
