178 // Structure describing the per-process mounts as found in
179 // /proc/[pid]/mountinfo. In particular, entries in this table specify
180 // the propagation properties of mounts, information not present in
181 // the MountTable or /etc/fstab. Entry order is preserved when
182 // parsing /proc/[pid]/mountinfo.
183 struct MountInfoTable {
184  // Structure describing an individual /proc/[pid]/mountinfo entry.
185  // See the /proc/[pid]/mountinfo section in 'man proc' for further
186  // details on each field.
187  struct Entry {
188  static Try<Entry> parse(const std::string& s);
189 
190  Entry() {}
191 
192  // Returns the ID of the peer group in which this mount resides.
193  // Returns none if this mount is not a shared mount.
194  Option<int> shared() const;
195 
196  // Returns the ID of the peer group in which this mount's master
197  // resides in. Returns none if this mount is not a slave mount.
198  Option<int> master() const;
199 
200  int id; // mountinfo[1]: mount ID.
201  int parent; // mountinfo[2]: parent ID.
202  dev_t devno; // mountinfo[3]: st_dev.
203 
204  std::string root; // mountinfo[4]: root of the mount.
205  std::string target; // mountinfo[5]: mount point.
206 
207  // Filesystem independent (VFS) options, e.g., "rw,noatime".
208  std::string vfsOptions; // mountinfo[6]: per-mount options.
209  // Filesystem dependent options, e.g., "rw,memory" for a memory
210  // cgroup filesystem.
211  std::string fsOptions; // mountinfo[11]: per-block options.
212 
213  // Current possible optional fields include shared:X, master:X,
214  // propagate_from:X, unbindable.
215  std::string optionalFields; // mountinfo[7]: optional fields.
216 
217  // mountinfo[8] is a separator.
218 
219  std::string type; // mountinfo[9]: filesystem type.
220  std::string source; // mountinfo[10]: source dev, other.
221  };
