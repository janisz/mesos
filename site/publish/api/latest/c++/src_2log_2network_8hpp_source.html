186  process::Future<size_t> watch(size_t size, Network::WatchMode mode)
187  {
188  if (satisfied(size, mode)) {
189  return pids.size();
190  }
191 
192  Watch* watch = new Watch(size, mode);
193  watches.push_back(watch);
194 
195  // TODO(jieyu): Consider deleting 'watch' if the returned future
196  // is discarded by the user.
197  return watch->promise.future();
198  }

200  // Sends a request to each of the groups members and returns a set
201  // of futures that represent their responses.
202  template <typename Req, typename Res>
203  std::set<process::Future<Res> > broadcast(
204  const Protocol<Req, Res>& protocol,
205  const Req& req,
206  const std::set<process::UPID>& filter)
207  {
208  std::set<process::Future<Res> > futures;
209  typename std::set<process::UPID>::const_iterator iterator;
210  for (iterator = pids.begin(); iterator != pids.end(); ++iterator) {
211  const process::UPID& pid = *iterator;
212  if (filter.count(pid) == 0) {
213  futures.insert(protocol(pid, req));
214  }
215  }
216  return futures;
217  }
218 
219  template <typename M>
220  Nothing broadcast(
221  const M& m,
222  const std::set<process::UPID>& filter)
223  {
224  std::set<process::UPID>::const_iterator iterator;
225  for (iterator = pids.begin(); iterator != pids.end(); ++iterator) {
226  const process::UPID& pid = *iterator;
227  if (filter.count(pid) == 0) {
228  process::post(pid, m);
229  }
230  }
231  return Nothing();
232  }
233 
234 protected:
235  virtual void finalize()
236  {
237  foreach (Watch* watch, watches) {
238  watch->promise.fail("Network is being terminated");
239  delete watch;
240  }
241  watches.clear();
242  }

247  Watch(size_t _size, Network::WatchMode _mode)

259  // Notifies the change of the network.
260  void update()
261  {
262  const size_t size = watches.size();
263  for (size_t i = 0; i < size; i++) {
264  Watch* watch = watches.front();
265  watches.pop_front();
266 
267  if (satisfied(watch->size, watch->mode)) {
268  watch->promise.set(pids.size());
269  delete watch;
270  } else {
271  watches.push_back(watch);
272  }
273  }
274  }
275 
276  // Returns true if the current size of the network satisfies the
277  // constraint specified by 'size' and 'mode'.
278  bool satisfied(size_t size, Network::WatchMode mode)
279  {
280  switch (mode) {
281  case Network::EQUAL_TO:
282  return pids.size() == size;
283  case Network::NOT_EQUAL_TO:
284  return pids.size() != size;
285  case Network::LESS_THAN:
286  return pids.size() < size;
287  case Network::LESS_THAN_OR_EQUAL_TO:
288  return pids.size() <= size;
289  case Network::GREATER_THAN:
290  return pids.size() > size;
291  case Network::GREATER_THAN_OR_EQUAL_TO:
292  return pids.size() >= size;
293  default:
294  LOG(FATAL) << "Invalid watch mode";
295  UNREACHABLE();
296  }
297  }
298 
299  std::set<process::UPID> pids;
300  std::list<Watch*> watches;
301 };

400 inline void ZooKeeperNetwork::watched(
401  const process::Future<std::set<zookeeper::Group::Membership> >&)
402 {
403  if (memberships.isFailed()) {
404  // We can't do much here, we could try creating another Group but
405  // that might just continue indefinitely, so we fail early
406  // instead. Note that Group handles all retryable/recoverable
407  // ZooKeeper errors internally.
408  LOG(FATAL) << "Failed to watch ZooKeeper group: " << memberships.failure();
409  }
410 
411  CHECK_READY(memberships); // Not expecting Group to discard futures.
412 
413  LOG(INFO) << "ZooKeeper group memberships changed";
414 
415  // Get data for each membership in order to convert them to PIDs.
416  std::list<process::Future<Option<std::string> > > futures;
417 
418  foreach (const zookeeper::Group::Membership& membership, memberships.get()) {
419  futures.push_back(group.data(membership));
420  }
421 
422  process::collect(futures)
423  .after(Seconds(5),
424  [](process::Future<std::list<Option<std::string>>> datas) {
425  // Handling time outs when collecting membership
426  // data. For now, a timeout is treated as a failure.
427  datas.discard();
428  return process::Failure("Timed out");
429  })
430  .onAny(executor.defer(lambda::bind(&This::collected, this, lambda::_1)));
431 }

434 inline void ZooKeeperNetwork::collected(
435  const process::Future<std::list<Option<std::string> > >& datas)
436 {
437  if (datas.isFailed()) {
438  LOG(WARNING) << "Failed to get data for ZooKeeper group members: "
439  << datas.failure();
440 
441  // Try again later assuming empty group. Note that this does not
442  // remove any of the current group members.
443  watch(std::set<zookeeper::Group::Membership>());
444  return;
445  }
446 
447  CHECK_READY(datas); // Not expecting collect to discard futures.
448 
449  std::set<process::UPID> pids;
450 
451  foreach (const Option<std::string>& data, datas.get()) {
452  // Data could be None if the membership is gone before its
453  // content can be read.
454  if (data.isSome()) {
455  process::UPID pid(data.get());
456  CHECK(pid) << "Failed to parse '" << data.get() << "'";
457  pids.insert(pid);
458  }
459  }
460 
461  LOG(INFO) << "ZooKeeper group PIDs: " << stringify(pids);
462 
463  // Update the network. We make sure that the PIDs from the base set
464  // are always in the network.
465  set(pids | base);
466 
467  watch(memberships.get());
468 }
