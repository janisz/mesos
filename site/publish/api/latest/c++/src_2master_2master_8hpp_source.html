164  void addTask(Task* task)
165  {
166  const TaskID& taskId = task->task_id();
167  const FrameworkID& frameworkId = task->framework_id();
168 
169  CHECK(!tasks[frameworkId].contains(taskId))
170  << "Duplicate task " << taskId << " of framework " << frameworkId;
171 
172  tasks[frameworkId][taskId] = task;
173 
174  if (!protobuf::isTerminalState(task->state())) {
175  usedResources[frameworkId] += task->resources();
176  }
177 
178  LOG(INFO) << "Adding task " << taskId
179  << " with resources " << task->resources()
180  << " on agent " << id << " (" << info.hostname() << ")";

183  // Notification of task termination, for resource accounting.
184  // TODO(bmahler): This is a hack for performance. We need to
185  // maintain resource counters because computing task resources
186  // functionally for all tasks is expensive, for now.
187  void taskTerminated(Task* task)
188  {
189  const TaskID& taskId = task->task_id();
190  const FrameworkID& frameworkId = task->framework_id();
191 
192  CHECK(protobuf::isTerminalState(task->state()));
193  CHECK(tasks[frameworkId].contains(taskId))
194  << "Unknown task " << taskId << " of framework " << frameworkId;
195 
196  usedResources[frameworkId] -= task->resources();
197  if (!tasks.contains(frameworkId) && !executors.contains(frameworkId)) {
198  usedResources.erase(frameworkId);
199  }
200  }

202  void removeTask(Task* task)
203  {
204  const TaskID& taskId = task->task_id();
205  const FrameworkID& frameworkId = task->framework_id();
206 
207  CHECK(tasks[frameworkId].contains(taskId))
208  << "Unknown task " << taskId << " of framework " << frameworkId;
209 
210  if (!protobuf::isTerminalState(task->state())) {
211  usedResources[frameworkId] -= task->resources();
212  if (!tasks.contains(frameworkId) && !executors.contains(frameworkId)) {
213  usedResources.erase(frameworkId);
214  }
215  }
216 
217  tasks[frameworkId].erase(taskId);
218  if (tasks[frameworkId].empty()) {
219  tasks.erase(frameworkId);
220  }
221 
222  killedTasks.remove(frameworkId, taskId);
223  }

277  {
278  CHECK(hasExecutor(frameworkId, executorId))

284  executors[frameworkId].erase(executorId);
285  if (executors[frameworkId].empty()) {
286  executors.erase(frameworkId);

968  private:
969  // Heuristically tries to determine whether a quota request could
970  // reasonably be satisfied given the current cluster capacity. The
971  // goal is to determine whether a user may accidentally request an
972  // amount of resources that would prevent frameworks without quota
973  // from getting any offers. A force flag will allow users to bypass
974  // this check.
975  //
976  // The heuristic tests whether the total quota, including the new
977  // request, does not exceed the sum of non-static cluster resources,
978  // i.e. the following inequality holds:
979  // total - statically reserved >= total quota + quota request
980  //
981  // Please be advised that:
982  // * It is up to an allocator how to satisfy quota (for example,
983  // what resources to account towards quota, as well as which
984  // resources to consider allocatable for quota).
985  // * Even if there are enough resources at the moment of this check,
986  // agents may terminate at any time, rendering the cluster under
987  // quota.
988  Option<Error> capacityHeuristic(
989  const mesos::quota::QuotaInfo& request) const;
990 
991  // We always want to rescind offers after the capacity heuristic. The
992  // reason for this is the race between the allocator and the master:
993  // it can happen that there are not enough free resources at the
994  // allocator's disposal when it is notified about the quota request,
995  // but at this point it's too late to rescind.
996  //
997  // While rescinding, we adhere to the following rules:
998  // * Rescind at least as many resources as there are in the quota request.
999  // * Rescind all offers from an agent in order to make the potential
1000  // offer bigger, which increases the chances that a quota'ed framework
1001  // will be able to use the offer.
1002  // * Rescind offers from at least `numF` agents to make it possible
1003  // (but not guaranteed, due to fair sharing) that each framework in
1004  // the role for which quota is set gets an offer (`numF` is the
1005  // number of frameworks in the quota'ed role). Though this is not
1006  // strictly necessary, we think this will increase the debugability
1007  // and will improve user experience.
1008  //
1009  // TODO(alexr): Consider removing this function once offer management
1010  // (including rescinding) is moved to allocator.
