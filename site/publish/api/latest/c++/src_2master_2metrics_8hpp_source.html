83  // Counters for messages from all frameworks of this principal.
84  // Note: We only count messages from active scheduler
85  // *instances* while they are *registered*. i.e., messages
86  // prior to the completion of (re)registration
87  // (AuthenticateMessage and (Re)RegisterFrameworkMessage) and
88  // messages from an inactive scheduler instance (after the
89  // framework has failed over) are not counted.
90 
91  // Framework messages received (before processing).
92  process::metrics::Counter messages_received;
93 
94  // Framework messages processed.
95  // NOTE: This doesn't include dropped messages. Processing of
96  // a message may be throttled by a RateLimiter if one is
97  // configured for this principal. Also due to Master's
98  // asynchronous nature, this doesn't necessarily mean the work
99  // requested by this message has finished.
100  process::metrics::Counter messages_processed;
101 
102  explicit Frameworks(const std::string& principal)
103  : messages_received("frameworks/" + principal + "/messages_received"),
104  messages_processed("frameworks/" + principal + "/messages_processed")

107  process::metrics::add(messages_processed);
108  }
109 
110  ~Frameworks()
