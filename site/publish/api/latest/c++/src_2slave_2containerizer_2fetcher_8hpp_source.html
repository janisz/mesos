44 // Argument passing to and invocation of the external fetcher program.
45 // Bookkeeping of executor files that are cached after downloading from
46 // a URI by the fetcher program. Cache eviction. There has to be exactly
47 // one fetcher with a distinct cache directory per active slave. This
48 // means that the cache directory can only be fixed after the slave ID
49 // has been determined by registration or recovery. Downloads to cache
50 // files are separated on a per-user basis. The cache must only be used
51 // for URIs for which the expected download size can be determined and
52 // trusted before downloading. If there is any problem using the cache
53 // for any given URI, the fetch procedure automatically reverts to
54 // fetching directly into the sandbox directory.
55 class Fetcher
56 {
57 public:
58  // Extracts the basename from a URI. For example, "d.txt" from
59  // "htpp://1.2.3.4:5050/a/b/c/d.txt". The current implementation
60  // only works for fairly regular-shaped URIs with a "/" and a proper
61  // file name at the end.
62  static Try<std::string> basename(const std::string& uri);
63 
64  // Some checks to make sure using the URI value in shell commands
65  // is safe.
66  // TODO(benh): These should be pushed into the scheduler driver and
67  // reported to the user.
68  static Try<Nothing> validateUri(const std::string& uri);
69 
70  // Checks to make sure the URI 'output_file' is valid.
71  static Try<Nothing> validateOutputFile(const std::string& path);
72 
73  // Determines if the given URI refers to a local file system path
74  // and prepends frameworksHome if it is a relative path. Fails if
75  // frameworksHome is empty and a local path is indicated.
76  static Result<std::string> uriToLocalPath(
77  const std::string& uri,
78  const Option<std::string>& frameworksHome);
79 
80  static bool isNetUri(const std::string& uri);
81 
82  Fetcher();
83 
84  // This is only public for tests.
85  Fetcher(const process::Owned<FetcherProcess>& process);
86 
87  virtual ~Fetcher();
88 
89  // TODO(bernd-mesos): Inject these parameters at Fetcher creation time.
90  // Then also inject the fetcher into the slave at creation time. Then
91  // it will be possible to make this an instance method instead of a
92  // static one for the slave to call during startup or recovery.
93  static Try<Nothing> recover(const SlaveID& slaveId, const Flags& flags);
94 
95  // Download the URIs specified in the command info and place the
96  // resulting files into the given sandbox directory. Chmod said files
97  // to the user if given. Send stdout and stderr output to files
98  // "stdout" and "stderr" in the given directory. Extract archives and/or
99  // use the cache if so instructed by the given CommandInfo::URI items.
100  process::Future<Nothing> fetch(
101  const ContainerID& containerId,
102  const CommandInfo& commandInfo,
103  const std::string& sandboxDirectory,
104  const Option<std::string>& user,
105  const SlaveID& slaveId,
106  const Flags& flags);
107 
108  // Best effort to kill the fetcher subprocess associated with the
109  // indicated container. Do nothing if no such subprocess exists.
110  void kill(const ContainerID& containerId);
111 
112 private:
113  process::Owned<FetcherProcess> process;
