42 // Provides an abstraction for removing files and directories after
43 // some point at which they are no longer considered necessary to keep
44 // around. The intent with this abstraction is to also easily enable
45 // implementations that may actually copy files and directories to
46 // "more" permanent storage (or provide any other hooks that might be
47 // useful, e.g., emailing users some time before their files are
48 // scheduled for removal).
49 class GarbageCollector
50 {
51 public:
52  GarbageCollector();
53  virtual ~GarbageCollector();
54 
55  // Schedules the specified path for removal after the specified
56  // duration of time has elapsed. If the path is already scheduled,
57  // this will reschedule the removal operation, and induce a discard
58  // on the previous future.
59  // The future will become ready when the path has been removed.
60  // The future will fail if the path did not exist, or on error.
61  // The future will be discarded if the path was unscheduled, or
62  // was rescheduled.
63  // Note that you currently cannot discard a returned future, instead
64  // you must call unschedule.
65  virtual process::Future<Nothing> schedule(
66  const Duration& d,
67  const std::string& path);
68 
69  // Unschedules the specified path for removal.
70  // The future will be true if the path has been unscheduled.
71  // The future will be false if the path is not scheduled for
72  // removal, or the path has already being removed.
73  // Note that you currently cannot discard a returned future.
74  virtual process::Future<bool> unschedule(const std::string& path);
75 
76  // Deletes all the directories, whose scheduled garbage collection time
77  // is within the next 'd' duration of time.
78  virtual void prune(const Duration& d);
79 
80 private:
81  GarbageCollectorProcess* process;
82 };
