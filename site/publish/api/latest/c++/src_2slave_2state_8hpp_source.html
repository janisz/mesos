63 // This function performs recovery from the state stored at 'rootDir'.
64 // If the 'strict' flag is set, any errors encountered while
65 // recovering a state are considered fatal and hence the recovery is
66 // short-circuited and returns an error. There might be orphaned
67 // executors that need to be manually cleaned up. If 'strict' flag is
68 // not set, any errors encountered are considered non-fatal and the
69 // recovery continues by recovering as much of the state as possible,
70 // while increasing the 'errors' count. Note that 'errors' on a struct
71 // includes the 'errors' encountered recursively. In other words,
72 // 'State.errors' is the sum total of all recovery errors. If the
73 // machine has rebooted since the last slave run, None is returned.
74 Result<State> recover(const std::string& rootDir, bool strict);

115 // Thin wrapper to checkpoint data to disk and perform the necessary
116 // error checking. It checkpoints an instance of T at the given path.
117 // We can checkpoint anything as long as T is supported by
118 // internal::checkpoint. Currently the list of supported Ts are:
119 // - std::string
120 // - google::protobuf::Message
121 // - google::protobuf::RepeatedPtrField<T>
122 // - mesos::Resources
123 //
124 // NOTE: We provide atomic (all-or-nothing) semantics here by always
125 // writing to a temporary file first then using os::rename to atomically
126 // move it to the desired path.
127 template <typename T>
128 Try<Nothing> checkpoint(const std::string& path, const T& t)
129 {
130  // Create the base directory.
131  std::string base = Path(path).dirname();
132 
133  Try<Nothing> mkdir = os::mkdir(base);
134  if (mkdir.isError()) {
135  return Error("Failed to create directory '" + base + "': " + mkdir.error());
136  }
137 
138  // NOTE: We create the temporary file at 'base/XXXXXX' to make sure
139  // rename below does not cross devices (MESOS-2319).
140  //
141  // TODO(jieyu): It's possible that the temporary file becomes
142  // dangling if slave crashes or restarts while checkpointing.
143  // Consider adding a way to garbage collect them.
144  Try<std::string> temp = os::mktemp(path::join(base, "XXXXXX"));
145  if (temp.isError()) {
146  return Error("Failed to create temporary file: " + temp.error());
147  }
148 
149  // Now checkpoint the instance of T to the temporary file.
150  Try<Nothing> checkpoint = internal::checkpoint(temp.get(), t);
151  if (checkpoint.isError()) {
152  // Try removing the temporary file on error.
153  os::rm(temp.get());
154 
155  return Error("Failed to write temporary file '" + temp.get() +
156  "': " + checkpoint.error());
157  }
158 
159  // Rename the temporary file to the path.
160  Try<Nothing> rename = os::rename(temp.get(), path);
161  if (rename.isError()) {
162  // Try removing the temporary file on error.
163  os::rm(temp.get());
164 
165  return Error("Failed to rename '" + temp.get() + "' to '" +
166  path + "': " + rename.error());
167  }
168 
169  return Nothing();
170 }
