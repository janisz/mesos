799 // .WillOnce(LaunchTasks(EXECUTOR, TASKS, CPUS, MEM, ROLE));

803 {

806  int numTasks = tasks;
807 
808  int launched = 0;
809  for (size_t i = 0; i < offers.size(); i++) {
810  const Offer& offer = offers[i];

817  Resources remaining = offer.resources();
818
