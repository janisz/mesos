24 // Initialize Windows socket stack.
25 inline bool wsa_initialize()
26 {
27  // Initialize WinSock (request version 2.2).
28  WORD requestedVersion = MAKEWORD(2, 2);
29  WSADATA data;
30 
31  const int result = ::WSAStartup(requestedVersion, &data);
32  if (result != 0) {
33  const int error = ::WSAGetLastError();
34  LOG(ERROR) << "Could not initialize WinSock, error code : " << error;
35  return false;
36  }
37 
38  // Check that the WinSock version we got back is 2.2 or higher.
39  // The high-order byte specifies the minor version number.
40  if (LOBYTE(data.wVersion) < 2 ||
41  (LOBYTE(data.wVersion) == 2 &&
42  HIBYTE(data.wVersion) != 2)) {
43  LOG(ERROR) << "Incorrect WinSock version found : " << LOBYTE(data.wVersion)
44  << "." << HIBYTE(data.wVersion);
45 
46  // WinSock was initialized, we just didn't like the version, so we need to
47  // clean up.
48  if (::WSACleanup() != 0) {
49  const int error = ::WSAGetLastError();
50  LOG(ERROR) << "Could not cleanup WinSock, error code : " << error;
51  }
52 
53  return false;
54  }
55 
56  return true;
57 }

60 inline bool wsa_cleanup()
61 {
62  // Cleanup WinSock. Wait for any outstanding socket operations to complete
63  // before exiting. Retry for a maximum of 10 times at 1 second intervals.
64  int retriesLeft = 10;
65 
66  while (retriesLeft > 0) {
67  const int result = ::WSACleanup();
68  if (result != 0) {
69  const int error = ::WSAGetLastError();
70  // Make it idempotent.
71  if (error == WSANOTINITIALISED) {
72  return false;
73  }
74 
75  // Wait for any blocking calls to complete and retry after 1 second.
76  if (error == WSAEINPROGRESS) {
77  LOG(ERROR) << "Waiting for outstanding WinSock calls to complete.";
78  ::Sleep(1000);
79  retriesLeft--;
80  } else {
81  LOG(ERROR) << "Could not cleanup WinSock, error code : " << error;
82  return false;
83  }
84  }
85  break;
86  }
87  if (retriesLeft == 0) {
88  return false;
89  }
90 
91  return true;
92 }
