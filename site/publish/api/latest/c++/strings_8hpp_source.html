40 inline std::string remove(
41  const std::string& from,
42  const std::string& substring,
43  Mode mode = ANY)
44 {
45  std::string result = from;
46 
47  if (mode == PREFIX) {
48  if (from.find(substring) == 0) {
49  result = from.substr(substring.size());
50  }
51  } else if (mode == SUFFIX) {
52  if (from.rfind(substring) == from.size() - substring.size()) {
53  result = from.substr(0, from.size() - substring.size());
54  }
55  } else {
56  size_t index;
57  while ((index = result.find(substring)) != std::string::npos) {
58  result = result.erase(index, substring.size());
59  }
60  }
61 
62  return result;
63 }

66 inline std::string trim(
67  const std::string& from,
68  Mode mode = ANY,
69  const std::string& chars = WHITESPACE)
70 {
71  size_t start = 0;
72  Option<size_t> end = None();
73 
74  if (mode == ANY) {
75  start = from.find_first_not_of(chars);
76  end = from.find_last_not_of(chars);
77  } else if (mode == PREFIX) {
78  start = from.find_first_not_of(chars);
79  } else if (mode == SUFFIX) {
80  end = from.find_last_not_of(chars);
81  }
82 
83  // Bail early if 'from' contains only characters in 'chars'.
84  if (start == std::string::npos) {
85  return "";
86  }
87 
88  // Calculate the length of the substring, defaulting to the "end" of
89  // string if there were no characters to remove from the suffix.
90  size_t length = std::string::npos;
91 
92  // Found characters to trim at the end.
93  if (end.isSome() && end.get() != std::string::npos) {
94  length = end.get() + 1 - start;
95  }
96 
97  return from.substr(start, length);
98 }

112 inline std::string replace(

115  const std::string& to)
116 {

118  size_t index = 0;

121  return result;
122  }
123

125  result.replace(index, from.length(), to);

127  }

132 // Tokenizes the string using the delimiters.
133 // Empty tokens will not be included in the result.
134 // Optionally, maximum number of tokens to be returned
135 // can be specified.
136 inline std::vector<std::string> tokenize(
137  const std::string& s,
138  const std::string& delims,
139  const Option<size_t>& maxTokens = None())
140 {
141  size_t offset = 0;
142  std::vector<std::string> tokens;
143 
144  while (maxTokens.isNone() || maxTokens.get() > 0) {
145  size_t nonDelim = s.find_first_not_of(delims, offset);
146 
147  if (nonDelim == std::string::npos) {
148  break; // Nothing left
149  }
150 
151  size_t delim = s.find_first_of(delims, nonDelim);
152 
153  // Finish tokenizing if this is the last token,
154  // or we've found enough tokens.
155  if (delim == std::string::npos ||
156  (maxTokens.isSome() && tokens.size() == maxTokens.get() - 1)) {
157  tokens.push_back(s.substr(nonDelim));
158  break;
159  }
160 
161  tokens.push_back(s.substr(nonDelim, delim - nonDelim));
162  offset = delim;
163  }
164 
165  return tokens;
166 }

169 // Splits the string using the provided delimiters.
170 // The string is split each time at the first character
171 // that matches any of the characters specified in delims.
172 // Empty tokens are allowed in the result.
173 // Optionally, maximum number of tokens to be returned
174 // can be specified.
175 inline std::vector<std::string> split(
176  const std::string& s,
177  const std::string& delims,
178  const Option<size_t>& maxTokens = None())
179 {
180  size_t offset = 0;
181  std::vector<std::string> tokens;
182 
183  while (maxTokens.isNone() || maxTokens.get() > 0) {
184  size_t next = s.find_first_of(delims, offset);
185 
186  // Finish splitting if this is the last token,
187  // or we've found enough tokens.
188  if (next == std::string::npos ||
189  (maxTokens.isSome() && tokens.size() == maxTokens.get() - 1)) {
190  tokens.push_back(s.substr(offset));
191  break;
192  }
193 
194  tokens.push_back(s.substr(offset, next - offset));
195  offset = next + 1;
196  }
197 
198  return tokens;
199 }

202 // Returns a map of strings to strings based on calling tokenize
203 // twice. All non-pairs are discarded. For example:
204 //
205 // pairs("foo=1;bar=2;baz;foo=3;bam=1=2", ";&", "=")
206 //
207 // Would return a map with the following:
208 // bar: ["2"]
209 // foo: ["1", "3"]
210 inline std::map<std::string, std::vector<std::string> > pairs(
211  const std::string& s,
212  const std::string& delims1,
213  const std::string& delims2)
214 {
215  std::map<std::string, std::vector<std::string> > result;
216 
217  const std::vector<std::string> tokens = tokenize(s, delims1);
218  foreach (const std::string& token, tokens) {
219  const std::vector<std::string> pairs = tokenize(token, delims2);
220  if (pairs.size() == 2) {
221  result[pairs[0]].push_back(pairs[1]);
222  }
223  }
224 
225  return result;
226 }
