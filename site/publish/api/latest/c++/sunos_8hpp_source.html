40 inline Result<Process> process(pid_t pid)
41 {
42  std::string fn = "/proc/" + stringify(pid) + "/status";
43  struct pstatus pstatus;
44 
45  Try<int> fd = os::open(fn, O_RDONLY);
46  if (fd.isError()) {
47  return ErrnoError("Cannot open " + fn);
48  }
49 
50  if (::read(fd.get(), &pstatus, sizeof(pstatus)) != sizeof(pstatus)) {
51  os::close(fd.get());
52  return Error("Cannot read from " + fn);
53  }
54  os::close(fd.get());

88 // Reads from /proc and returns a list of all running processes.
89 inline Try<std::set<pid_t> > pids()
90 {
91  std::set<pid_t> pids;
92 
93  Try<std::list<std::string> > entries = os::ls("/proc");
94  if (entries.isError()) {
95  return Error("Failed to list files in /proc: " + entries.error());
96  }
97 
98  foreach (const std::string& entry, entries.get()) {
99  Try<pid_t> pid = numify<pid_t>(entry);
100  if (pid.isSome()) {
101  pids.insert(pid.get()); // Ignore entries that can't be numified.
102  }
103  }
104 
105  if (!pids.empty()) {
106  return pids;
107  }
108 
109  return Error("Failed to determine pids from /proc");
110 }
