29 // Gets symlink data for a given path, if it exists.
30 //
31 // This turns out to be a very complicated task on Windows. The gist of it is
32 // that we know that symlinks on Windows are implemented with the Reparse Point
33 // API, and so the process is a matter of:
34 //
35 // 1. Checking whether the attributes for the file/folder specified by the
36 // path have the reparse point bit set; all symlinks are implemented with
37 // reparse points, so this bit should be on all symlinks.
38 // 2. Opening a file/folder handle for that path, instructing it specifically
39 // to open a handle for the symlink (if the path points at a symlink) and
40 // *not* the file the symlink points at (as is the default). Note that
41 // file and folder handles are different, so we have a function that
42 // chooses appropriately.
43 // 3. Using `DeviceIoControl` to obtain information about the handle for this
44 // reparse point, which we can then query to figure out if it's a reparse
45 // point that is owned by the symlink filesystem filter driver.
46 // 4. If it is, then we report that this path does point at a symlink.
47 //
48 // NOTE: it may be helpful to consult the documentation for each of these
49 // functions, as they give you sources that justify the arguments to the
50 // obscure APIs we call to get this all working.
51 inline Try<SymbolicLink> query_symbolic_link_data(const std::string& path)
52 {
53  // Convert to absolute path because Windows APIs expect it.
54  const Result<std::string> absolute_path = os::realpath(path);
55  if (!absolute_path.isSome()) {
56  return Error(absolute_path.error());
57  }
58 
59  // Windows has no built-in way to tell whether a path points at a symbolic
60  // link; but, we know that symbolic links are implemented with reparse
61  // points, so we begin by checking that.
62  Try<bool> is_reparse_point =
63  reparse_point_attribute_set(absolute_path.get());
64 
65  if (is_reparse_point.isError()) {
66  return Error(is_reparse_point.error());
67  } else if (!is_reparse_point.get()) {
68  return Error(
69  "Reparse point attribute is not set for path '" + absolute_path.get() +
70  "', and therefore it is not a symbolic link");
71  }
72 
73  const Try<SharedHandle> symlink_handle =
74  get_handle_no_follow(absolute_path.get());
75 
76  if (symlink_handle.isError()) {
77  return Error(symlink_handle.error());
78  }
79 
80  // Finally, retrieve symlink data for the handle, if any.
81  return get_symbolic_link_data(symlink_handle.get().get_handle());
82 }
