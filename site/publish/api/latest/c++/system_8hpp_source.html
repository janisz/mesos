161  // HTTP endpoints.
162  Future<http::Response> stats(const http::Request& request)
163  {
164  JSON::Object object;
165  Try<os::Load> load = os::loadavg();
166  if (load.isSome()) {
167  object.values["avg_load_1min"] = load.get().one;
168  object.values["avg_load_5min"] = load.get().five;
169  object.values["avg_load_15min"] = load.get().fifteen;
170  }
171 
172  Try<long> cpus = os::cpus();
173  if (cpus.isSome()) {
174  object.values["cpus_total"] = cpus.get();
175  }
176 
177  Try<os::Memory> memory = os::memory();
178  if (memory.isSome()) {
179  object.values["mem_total_bytes"] = memory.get().total.bytes();
180  object.values["mem_free_bytes"] = memory.get().free.bytes();
181  }
182 
183  return http::OK(object, request.url.query.get("jsonp"));
184  }
