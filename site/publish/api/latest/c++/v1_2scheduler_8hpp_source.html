44 // Interface to Mesos for a scheduler. Abstracts master detection
45 // (connection and disconnection).
46 //
47 // Expects three callbacks, 'connected', 'disconnected', and
48 // 'received' which will get invoked _serially_ when it's determined
49 // that we've connected (i.e., detected master), disconnected
50 // (i.e, detected no master), or received events from the master.
51 // The library reconnects with the master upon a disconnection.
52 //
53 // NOTE: All calls and events are dropped while disconnected.
54 class Mesos
55 {
56 public:
57  // The credential will be used for authenticating with the master. Currently,
58  // only HTTP basic authentication is supported.
59  Mesos(const std::string& master,
60  ContentType contentType,
61  const std::function<void()>& connected,
62  const std::function<void()>& disconnected,
63  const std::function<void(const std::queue<Event>&)>& received,
64  const Option<Credential>& credential);
65

68

70  Mesos& operator=(const Mesos& other) = delete;
71 
72  virtual ~Mesos();
73 
74  // Attempts to send a call to the master.
75  //
76  // The scheduler should only invoke this method once it has received the
77  // 'connected' callback. Otherwise, all calls would be dropped while
78  // disconnected.
79  //
80  // Some local validation of calls is performed which may generate
81  // events without ever being sent to the master. This includes when
82  // calls are sent but no master is currently detected (i.e., we're
83  // disconnected).
84  virtual void send(const Call& call);
85 
86  // Force a reconnection with the master.
87  //
88  // In the case of a one-way network partition, the connection between the
89  // scheduler and master might not necessarily break. If the scheduler detects
90  // a partition, due to lack of `HEARTBEAT` events (e.g., 5) within a time
91  // window, it can explicitly ask the library to force a reconnection with
92  // the master.
93  //
94  // This call would be ignored if the scheduler is already disconnected with
95  // the master (e.g., no new master has been elected). Otherwise, the scheduler
96  // would get a 'disconnected' callback followed by a 'connected' callback.
97  virtual void reconnect();
98 
99 protected:
100  // NOTE: This constructor is used for testing.
101  Mesos(
102  const std::string& master,
103  ContentType contentType,
104  const std::function<void()>& connected,
105  const std::function<void()>& disconnected,
106  const std::function<void(const std::queue<Event>&)>& received,
107  const Option<Credential>& credential,
108  const Option<std::shared_ptr<mesos::master::detector::MasterDetector>>&
109  detector);
110 
111  // Stops the library so that:
112  // - No more calls can be sent to the master.
113  // - No more callbacks can be made to the scheduler. In some cases, there
114  // may be one additional callback if the library was in the middle of
115  // processing an event.
116  //
117  // NOTE: This is used for testing.
118  virtual void stop();
119 
120 private:
121  MesosProcess* process;
122 };
