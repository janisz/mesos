24 // A useful type that can be used to represent a Try that has failed. This is a
25 // lot like `ErrnoError`, except instead of wrapping an error coming from the C
26 // standard libraries, it wraps an error coming from the Windows APIs.

44  }

46  DWORD allocate_message_buffer =
47  FORMAT_MESSAGE_ALLOCATE_BUFFER |
48  FORMAT_MESSAGE_FROM_SYSTEM |
49  FORMAT_MESSAGE_IGNORE_INSERTS;
50 
51  DWORD default_language = MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT);

53  // This following function `FormatMessage` is a lot like `strerror`, except
54  // it pretty-prints errors from the Windows API instead of from the C
55  // standard library. Basically, the semantics are: we pass in `errorCode`,
56  // and it allocates room for a pretty-printed error message at
57  // `message_buffer`, and then dumps said pretty-printed error message at
58  // that address, in our `default_language`.
59  //
60  // The 5th actual parameter (namely `(LPSTR)&message_buffer`), may look
61  // strange to readers of this code. It is copied directly out of the
62  // documentation[1], and is unfortunately required to get the
63  // pretty-printed error message. The short story is:
64  //
65  // * The flag `FORMAT_MESSAGE_ALLOCATE_BUFFER` tells `FormatMessage` to
66  // point `message_buffer` (which is an `LPSTR` a.k.a. `char*`) at a
67  // string error message. But, `message_buffer` to point a `char*` at a
68  // different place, `FormatMessage` would need the address
69  // `&message_buffer` so that it could change where it is pointing.
70  // * So, to solve this problem, the API writers decided that when you
71  // pass that flag in, `FormatMessage` will treat the 5th parameter not
72  // as `LPSTR` (which is what the type is in the function signagure),
73  // but as `LPSTR*` a.k.a. `char**`, which (assuming you've casted the
74  // parameter correctly) allows it to allocate the message on your
75  // behalf, and change `message_buffer` to point at this new error
76  // string.
77  // * This is why we need this strange cast that you see below.
78  //
79  // Finally, and this is important: it is up to the user to free the memory
80  // with `LocalFree`! The line below where we do this comes directly from
81  // the documentation as well.
82  //
83  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms679351(v=vs.85).aspx
84  LPSTR message_buffer;
85  size_t size = FormatMessage(
86  allocate_message_buffer,
87  nullptr, // Ignored.
88  errorCode,
89  default_language,
90  (LPSTR) &message_buffer, // See comment above about quirky cast.
91  0, // Ignored.
92  nullptr); // Ignored.
93 
94  std::string message(message_buffer, size);
95 
96  // Required per documentation above.
97  LocalFree(message_buffer);
98 
99  return message;
100  }
101 };
