26 inline bool exists(const std::string& path)
27 {
28  Result<std::string> absolutePath = os::realpath(path);
29 
30  if (!absolutePath.isSome()) {

34  // NOTE: `GetFileAttributes` redirects to either `GetFileAttributesA`
35  // (ASCII) or `GetFileAttributesW` (for `wchar`s). It returns
36  // `INVALID_FILE_ATTRIBUTES` if the file could not be opened for any reason.
37  // Checking for one of two 'not found' error codes (`ERROR_FILE_NOT_FOUND` or
38  // `ERROR_PATH_NOT_FOUND`) is a reliable test for whether the file or
39  // directory exists. See also [1] for more information on this technique.
40  //
41  // [1] http://blogs.msdn.com/b/oldnewthing/archive/2007/10/23/5612082.aspx
42  DWORD attributes = GetFileAttributes(absolutePath.get().c_str());
43 
44  if (attributes == INVALID_FILE_ATTRIBUTES) {
45  DWORD error = GetLastError();
46  if (error == ERROR_FILE_NOT_FOUND || error == ERROR_PATH_NOT_FOUND) {
47  return false;
48  }
49  }
50 
51  return true;
52 }
