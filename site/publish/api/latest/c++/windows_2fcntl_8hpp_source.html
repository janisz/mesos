43 inline Try<Nothing> nonblock(int fd)
44 {
45  if (net::is_socket(fd)) {
46  const u_long non_block_mode = 1;
47  u_long mode = non_block_mode;
48 
49  int result = ioctlsocket(fd, FIONBIO, &mode);
50  if (result != NO_ERROR) {
51  return WindowsSocketError();
52  }
53  } else {
54  // Extract handle from file descriptor.
55  HANDLE handle = reinterpret_cast<HANDLE>(::_get_osfhandle(fd));
56  if (handle == INVALID_HANDLE_VALUE) {
57  return WindowsError("Failed to get `HANDLE` for file descriptor");
58  }
59 
60  if (GetFileType(handle) == FILE_TYPE_PIPE) {
61  DWORD pipe_mode = PIPE_NOWAIT;
62  if (SetNamedPipeHandleState(handle, &pipe_mode, nullptr, nullptr)) {
63  return WindowsError();
64  }
65  }
66  }
67 
68  return Nothing();
69 }
