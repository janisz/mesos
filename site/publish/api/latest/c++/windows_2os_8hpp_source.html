133 // Overload of os::pids for filtering by groups and sessions. A group / session
134 // id of 0 will fitler on the group / session ID of the calling process.
135 // NOTE: Windows does not have the concept of a process group, so we need to
136 // enumerate all processes.
137 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
138 {
139  DWORD max_items = 4096;
140  DWORD bytes_returned;
141  std::vector<pid_t> processes;
142  size_t size_in_bytes;
143 
144  // Attempt to populate `processes` with PIDs. We repeatedly call
145  // `EnumProcesses` with increasingly large arrays until it "succeeds" at
146  // populating the array with PIDs. The criteria for determining when
147  // `EnumProcesses` has succeeded are:
148  // (1) the return value is nonzero.
149  // (2) the `bytes_returned` is less than the number of bytes in the array.
150  do {
151  // TODO(alexnaparu): Set a limit to the memory that can be used.
152  processes.resize(max_items);
153  size_in_bytes = processes.size() * sizeof(pid_t);
154  BOOL result = ::EnumProcesses(processes.data(), size_in_bytes,
155  &bytes_returned);
156 
157  if (!result) {
158  return WindowsError("os::pids: Call to `EnumProcesses` failed");
159  }
160 
161  max_items *= 2;
162  } while (bytes_returned >= size_in_bytes);
163 
164  return std::set<pid_t>(processes.begin(), processes.end());
165 }

208 // Suspends execution of the calling process until a child specified by `pid`
209 // has changed state. Unlike the POSIX standard function `::waitpid`, this
210 // function does not use -1 and 0 to signify errors and nonblocking return.
211 // Instead, we return `Result<pid_t>`:
212 // * In case of error, we return `Error` rather than -1. For example, we
213 // would return an `Error` in case of `EINVAL`.
214 // * In case of nonblocking return, we return `None` rather than 0. For
215 // example, if we pass `WNOHANG` in the `options`, we would expect 0 to be
216 // returned in the case that children specified by `pid` exist, but have
217 // not changed state yet. In this case we return `None` instead.
218 //
219 // NOTE: There are important differences between the POSIX and Windows
220 // implementations of this function:
221 // * On POSIX, `pid_t` is a signed number, but on Windows, PIDs are `DWORD`,
222 // which is `unsigned long`. Thus, if we use `DWORD` to represent the `pid`
223 // argument, passing -1 as the `pid` would (on most modern servers)
224 // silently convert to a really large `pid`. This is undesirable.
225 // * Since it is important to be able to detect -1 has been passed to
226 // `os::waitpid`, as a matter of practicality, we choose to:
227 // (1) Use `long` to represent the `pid` argument.
228 // (2) Disable using any value <= 0 for `pid` on Windows.
229 // * This decision is pragmatic. The reasoning is:
230 // (1) The Windows code paths call `os::waitpid` in only a handful of
231 // places, and in none of these conditions do we need `-1` as a value.
232 // (2) Since PIDs virtually never take on values outside the range of
233 // vanilla signed `long` it is likely that an accidental conversion
234 // will never happen.
235 // (3) Even though it is not formalized in the C specification, the
236 // implementation of `long` on the vast majority of production servers
237 // is 2's complement, so we expect that when we accidentally do
238 // implicitly convert from `unsigned long` to `long`, we will "wrap
239 // around" to negative values. And since we've disabled the negative
240 // `pid` in the Windows implementation, we should error out.
241 // * Finally, on Windows, we currently do not check that the process we are
242 // attempting to await is a child process.
243 inline Result<pid_t> waitpid(long pid, int* status, int options)
244 {
245  const bool wait_for_child = (options & WNOHANG) == 0;
246 
247  // NOTE: Windows does not implement pids <= 0.
248  if (pid <= 0) {
249  errno = ENOSYS;
250  return ErrnoError(
251  "os::waitpid: Value of pid is '" + stringify(pid) +
252  "'; the Windows implementation currently does not allow values <= 0");
253  } else if (options != 0 && options != WNOHANG) {
254  // NOTE: We only support `options == 0` or `options == WNOHANG`. On Windows
255  // no flags other than `WNOHANG` are supported.
256  errno = ENOSYS;
257  return ErrnoError(
258  "os::waitpid: Only flag `WNOHANG` is implemented on Windows");
259  }
260 
261  // TODO(hausdorff): Check that `pid` is one of the child processes. If not,
262  // set `errno` to `ECHILD` and return -1.
263 
264  // Open the child process as a safe `SharedHandle`.
265  const HANDLE process = ::OpenProcess(
266  PROCESS_QUERY_INFORMATION | SYNCHRONIZE,
267  FALSE,
268  static_cast<DWORD>(pid));
269 
270  if (process == nullptr) {
271  return WindowsError("os::waitpid: Failed to open process for pid '" +
272  stringify(pid) + "'");
273  }
274 
275  SharedHandle scoped_process(process, ::CloseHandle);
276 
277  // If `WNOHANG` flag is set, don't wait. Otherwise, wait for child to
278  // terminate.
279  const DWORD wait_time = wait_for_child ? INFINITE : 0;
280  const DWORD wait_results = ::WaitForSingleObject(
281  scoped_process.get(),
282  wait_time);
283 
284  // Verify our wait exited correctly.
285  const bool state_signaled = wait_results == WAIT_OBJECT_0;
286  if (options == 0 && !state_signaled) {
287  // If `WNOHANG` is not set, then we should have stopped waiting only for a
288  // state change in `scoped_process`.
289  errno = ECHILD;
290  return WindowsError(
291  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
292  "'. `::WaitForSingleObject` should have waited for child process to " +
293  "exit, but returned code '" + stringify(wait_results) +
294  "' instead");
295  } else if (wait_for_child && !state_signaled &&
296  wait_results != WAIT_TIMEOUT) {
297  // If `WNOHANG` is set, then a successful wait should report either a
298  // timeout (since we set the time to wait to `0`), or a successful state
299  // change of `scoped_process`. Anything else is an error.
300  errno = ECHILD;
301  return WindowsError(
302  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
303  "'. `ENOHANG` flag was passed in, so `::WaitForSingleObject` should " +
304  "have either returned `WAIT_OBJECT_0` or `WAIT_TIMEOUT` (the " +
305  "timeout was set to 0, because we are not waiting for the child), " +
306  "but instead returned code '" + stringify(wait_results) + "'");
307  }
308 
309  if (!wait_for_child && wait_results == WAIT_TIMEOUT) {
310  // Success. `ENOHANG` was set and we got a timeout, so return `None` (POSIX
311  // `::waitpid` would return 0 here).
312  return None();
313  }
314 
315  // Attempt to retrieve exit code from child process. Store that exit code in
316  // the `status` variable if it's `nullptr`.
317  DWORD child_exit_code = 0;
318  if (!::GetExitCodeProcess(scoped_process.get(), &child_exit_code)) {
319  errno = ECHILD;
320  return WindowsError(
321  "os::waitpid: Successfully waited on child process with pid '" +
322  std::to_string(pid) + "', but could not retrieve exit code");
323  }
324 
325  if (status != nullptr) {
326  *status = child_exit_code;
327  }
328 
329  // Success. Return pid of the child process for which the status is reported.
330  return pid;
331 }

334 inline std::string hstrerror(int err)
335 {
336  char buffer[1024];
337  DWORD format_error = 0;
338 
339  // NOTE: Per the Linux documentation[1], `h_errno` can have only one of the
340  // following errors.
341  switch (err) {
342  case WSAHOST_NOT_FOUND:
343  case WSANO_DATA:
344  case WSANO_RECOVERY:
345  case WSATRY_AGAIN: {
346  format_error = ::FormatMessage(
347  FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
348  nullptr,
349  err,
350  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
351  buffer,
352  sizeof(buffer),
353  nullptr);
354  break;
355  }
356  default: {
357  return "Unknown resolver error";
358  }
359  }
360 
361  if (format_error == 0) {
362  // If call to `FormatMessage` fails, then we choose to output the error
363  // code rather than call `FormatMessage` again.
364  return "os::hstrerror: Call to `FormatMessage` failed with error code" +
365  std::to_string(GetLastError());
366  } else {
367  return buffer;
368  }
369 }

525 inline Result<PROCESSENTRY32> process_entry(pid_t pid)
526 {
527  // Get a snapshot of the processes in the system. NOTE: We should not check
528  // whether the handle is `nullptr`, because this API will always return
529  // `INVALID_HANDLE_VALUE` on error.
530  HANDLE snapshot_handle = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, pid);
531  if (snapshot_handle == INVALID_HANDLE_VALUE) {
532  return WindowsError(
533  "os::process_entry: Call to `CreateToolhelp32Snapshot` failed");
534  }
535 
536  SharedHandle safe_snapshot_handle(snapshot_handle, ::CloseHandle);
537 
538  // Initialize process entry.
539  PROCESSENTRY32 process_entry;
540  ZeroMemory(&process_entry, sizeof(PROCESSENTRY32));
541  process_entry.dwSize = sizeof(PROCESSENTRY32);
542 
543  // Get first process so that we can loop through process entries until we
544  // find the one we care about.
545  SetLastError(ERROR_SUCCESS);
546  bool has_next = Process32First(safe_snapshot_handle.get(), &process_entry);
547  if (!has_next) {
548  // No first process was found. We should never be here; it is arguable we
549  // should return `None`, since we won't find the PID we're looking for, but
550  // we elect to return `Error` because something terrible has probably
551  // happened.
552  if (GetLastError() != ERROR_SUCCESS) {
553  return WindowsError("os::process_entry: Call to `Process32First` failed");
554  } else {
555  return Error("os::process_entry: Call to `Process32First` failed");
556  }
557  }
558 
559  // Loop through processes until we find the one we're looking for.
560  while (has_next) {
561  if (process_entry.th32ProcessID == pid) {
562  // Process found.
563  return process_entry;
564  }
565 
566  has_next = Process32Next(safe_snapshot_handle.get(), &process_entry);
567  if (!has_next) {
568  DWORD last_error = GetLastError();
569  if (last_error != ERROR_NO_MORE_FILES && last_error != ERROR_SUCCESS) {
570  return WindowsError(
571  "os::process_entry: Call to `Process32Next` failed");
572  }
573  }
574  }
575 
576  return None();
577 }

580 // Generate a `Process` object for the process associated with `pid`. If
581 // process is not found, we return `None`; error is reserved for the case where
582 // something went wrong.
583 inline Result<Process> process(pid_t pid)
584 {
585  // Find process with pid.
586  Result<PROCESSENTRY32> entry = process_entry(pid);
587 
588  if (entry.isError()) {
589  return WindowsError(entry.error());
590  } else if (entry.isNone()) {
591  return None();
592  }
593 
594  HANDLE process_handle = ::OpenProcess(
595  PROCESS_QUERY_LIMITED_INFORMATION | PROCESS_VM_READ,
596  false,
597  pid);
598 
599  if (process_handle == INVALID_HANDLE_VALUE) {
600  return WindowsError("os::process: Call to `OpenProcess` failed");
601  }
602 
603  SharedHandle safe_process_handle(process_handle, ::CloseHandle);
604 
605  // Get Windows Working set size (Resident set size in linux).
606  PROCESS_MEMORY_COUNTERS proc_mem_counters;
607  BOOL get_process_memory_info = ::GetProcessMemoryInfo(
608  safe_process_handle.get_handle(),
609  &proc_mem_counters,
610  sizeof(proc_mem_counters));
611 
612  if (!get_process_memory_info) {
613  return WindowsError("os::process: Call to `GetProcessMemoryInfo` failed");
614  }
615 
616  // Get session Id.
617  pid_t session_id;
618  BOOL process_id_to_session_id = ::ProcessIdToSessionId(pid, &session_id);
619 
620  if (!process_id_to_session_id) {
621  return WindowsError("os::process: Call to `ProcessIdToSessionId` failed");
622  }
623 
624  // Get Process CPU time.
625  FILETIME create_filetime, exit_filetime, kernel_filetime, user_filetime;
626  BOOL get_process_times = ::GetProcessTimes(
627  safe_process_handle.get_handle(),
628  &create_filetime,
629  &exit_filetime,
630  &kernel_filetime,
631  &user_filetime);
632 
633  if (!get_process_times) {
634  return WindowsError("os::process: Call to `GetProcessTimes` failed");
635  }
636 
637  // Get utime and stime.
638  ULARGE_INTEGER lKernelTime, lUserTime; // In 100 nanoseconds.
639  lKernelTime.HighPart = kernel_filetime.dwHighDateTime;
640  lKernelTime.LowPart = kernel_filetime.dwLowDateTime;
641  lUserTime.HighPart = user_filetime.dwHighDateTime;
642  lUserTime.LowPart = user_filetime.dwLowDateTime;
643 
644  Try<Duration> utime = Nanoseconds(lKernelTime.QuadPart * 100);
645  Try<Duration> stime = Nanoseconds(lUserTime.QuadPart * 100);
646 
647  return Process(
648  pid,
649  entry.get().th32ParentProcessID, // Parent process id.
650  0, // Group id.
651  session_id,
652  Bytes(proc_mem_counters.WorkingSetSize),
653  utime.isSome() ? utime.get() : Option<Duration>::none(),
654  stime.isSome() ? stime.get() : Option<Duration>::none(),
655  entry.get().szExeFile, // Executable filename.
656  false); // Is not zombie process.
657 }

666 // `create_job` function creates a job object whose name is derived
667 // from the `pid` and associates the process with the job object.
668 // Every process started by the `pid` process which is part of the job
669 // object becomes part of the job object. The job name should match
670 // the name used in `kill_job`.
671 inline Try<HANDLE> create_job(pid_t pid)
672 {
673  Try<std::string> alpha_pid = strings::internal::format("MESOS_JOB_%X", pid);
674  if (alpha_pid.isError()) {
675  return Error(alpha_pid.error());
676  }
677 
678  HANDLE process_handle = ::OpenProcess(
679  PROCESS_SET_QUOTA | PROCESS_TERMINATE,
680  false,
681  pid);
682 
683  if (process_handle == INVALID_HANDLE_VALUE) {
684  return WindowsError("os::create_job: Call to `OpenProcess` failed");
685  }
686 
687  SharedHandle safe_process_handle(process_handle, ::CloseHandle);
688 
689  HANDLE job_handle = ::CreateJobObject(nullptr, alpha_pid.get().c_str());
690 
691  if (job_handle == nullptr) {
692  return WindowsError("os::create_job: Call to `CreateJobObject` failed");
693  }
694 
695  JOBOBJECT_EXTENDED_LIMIT_INFORMATION jeli = { { 0 }, 0 };
696 
697  jeli.BasicLimitInformation.LimitFlags = JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE;
698 
699  // The job object will be terminated when the job handle closes. This allows
700  // the job tree to be terminated in case of errors by closing the handle.
701  ::SetInformationJobObject(
702  job_handle,
703  JobObjectExtendedLimitInformation,
704  &jeli,
705  sizeof(jeli));
706 
707  if (::AssignProcessToJobObject(
708  job_handle,
709  safe_process_handle.get_handle()) == 0) {
710  return WindowsError(
711  "os::create_job: Call to `AssignProcessToJobObject` failed");
712  };
713 
714  return job_handle;
715 }

718 // `kill_job` function assumes the process identified by `pid`
719 // is associated with a job object whose name is derived from it.
720 // Every process started by the `pid` process which is part of the job
721 // object becomes part of the job object. Destroying the task
722 // will close all such processes.
723 inline Try<Nothing> kill_job(pid_t pid)
724 {
725  Try<std::string> alpha_pid = strings::internal::format("MESOS_JOB_%X", pid);
726  if (alpha_pid.isError()) {
727  return Error(alpha_pid.error());
728  }
729 
730  HANDLE job_handle = ::OpenJobObject(
731  JOB_OBJECT_TERMINATE,
732  FALSE,
733  alpha_pid.get().c_str());
734 
735  if (job_handle == nullptr) {
736  return WindowsError("os::kill_job: Call to `OpenJobObject` failed");
737  }
738 
739  SharedHandle safe_job_handle(job_handle, ::CloseHandle);
740 
741  BOOL result = ::TerminateJobObject(safe_job_handle.get_handle(), 1);
742  if (result == 0) {
743  return WindowsError();
744  }
745

776  securityAttr.bInheritHandle = TRUE;
777  securityAttr.lpSecurityDescriptor = nullptr;
778

780  HANDLE write_handle;
781 
782  const BOOL result = ::CreatePipe(
783  &read_handle,
784  &write_handle,
785  &securityAttr,
786  0);
787

789  reinterpret_cast<intptr_t>(read_handle),

797  if (pipe[1] == -1) {
