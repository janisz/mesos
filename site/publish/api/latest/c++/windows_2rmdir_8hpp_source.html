30 // Recursive version of `RemoveDirectory`. NOTE: unlike `rmdir`, this requires
31 // Windows-formatted paths, and therefore should be in the `internal` namespace.
32 inline Try<Nothing> recursive_remove_directory(
33  const std::string& path, bool removeRoot)
34 {
35  // Appending a slash here if the path doesn't already have one simplifies
36  // path join logic later, because (unlike Unix) Windows doesn't like double
37  // slashes in paths.
38  std::string current_path;
39 
40  if (!strings::endsWith(path, "\\")) {
41  current_path = path + "\\";
42  } else {
43  current_path = path;
44  }
45 
46  // Get first file matching pattern `X:\path\to\wherever\*`.
47  WIN32_FIND_DATA found;
48  const std::string search_pattern = current_path + "*";
49  const SharedHandle search_handle(
50  FindFirstFile(search_pattern.c_str(), &found),
51  FindClose);
52 
53  if (search_handle.get() == INVALID_HANDLE_VALUE) {
54  return WindowsError(
55  "`os::internal::recursive_remove_directory` failed when searching "
56  "for files with pattern '" + search_pattern + "'");
57  }
58 
59  do {
60  // NOTE: do-while is appropriate here because folder is guaranteed to have
61  // at least a file called `.` (and probably also one called `..`).
62  const std::string current_file(found.cFileName);
63 
64  const bool is_current_directory = current_file.compare(".") == 0;
65  const bool is_parent_directory = current_file.compare("..") == 0;
66 
67  // Don't try to delete `.` and `..` files in directory.
68  if (is_current_directory || is_parent_directory) {
69  continue;
70  }
71 
72  // Path to remove.
73  const std::string current_absolute_path = current_path + current_file;
74 
75  const bool is_directory = os::stat::isdir(current_absolute_path);
76 
77  // Delete current path, whether it's a directory, file, or symlink.
78  if (is_directory) {
79  Try<Nothing> removed =
80  recursive_remove_directory(current_absolute_path, removeRoot);
81 
82  if (removed.isError()) {
83  return Error(removed.error());
84  }
85  } else {
86  // NOTE: this also handles symbolic links.
87  if (::remove(current_absolute_path.c_str()) != 0) {
88  return WindowsError(
89  "`os::internal::recursive_remove_directory` attempted to delete "
90  "file '" + current_absolute_path + "', but failed");
91  }
92  }
93  } while (FindNextFile(search_handle.get(), &found));
94 
95  // Finally, remove current directory unless `removeRoot` is disabled.
96  if (removeRoot && ::_rmdir(current_path.c_str()) == -1) {
97  return ErrnoError(
98  "`os::internal::recursive_remove_directory` attempted to delete file "
99  "'" + current_path + "', but failed");
100  }
101 
102  return Nothing();
103 }

108 // By default, recursively deletes a directory akin to: 'rm -r'. If
109 // `recursive` is false, it deletes a directory akin to: 'rmdir'. In
110 // recursive mode, `removeRoot` can be set to false to enable removing
111 // all the files and directories beneath the given root directory, but
112 // not the root directory itself.
113 // Note that this function expects an absolute path.
114 inline Try<Nothing> rmdir(
115  const std::string& directory,
116  bool recursive = true,
117  bool removeRoot = true)
118 {
119  // Canonicalize the path to Windows style for the call to
120  // `recursive_remove_directory`.
121  Result<std::string> root = os::realpath(directory);
122 
123  if (root.isError()) {
124  return Error(root.error());
125  } else if (root.isNone()) {
126  return Error(
127  "Argument to `os::rmdir` is not a valid directory or file: '" +
128  directory + "'");
129  }
130 
131  if (!recursive) {
132  if (::_rmdir(directory.c_str()) < 0) {
133  return ErrnoError();
134  } else {
135  return Nothing();
136  }
137  } else {
138  return os::internal::recursive_remove_directory(root.get(), removeRoot);
139  }
140 }
