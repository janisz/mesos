61 template <typename... T>
62 Try<std::string> shell(const std::string& fmt, const T&... t)
63 {
64  const Try<std::string> command = strings::internal::format(fmt, t...);
65  if (command.isError()) {
66  return Error(command.error());
67  }
68 
69  FILE* file;
70  std::ostringstream stdoutstr;
71 
72  if ((file = _popen(command.get().c_str(), "r")) == nullptr) {
73  return Error("Failed to run '" + command.get() + "'");
74  }
75 
76  char line[1024];
77  // NOTE(vinod): Ideally the if and while loops should be interchanged. But
78  // we get a broken pipe error if we don't read the output and simply close.
79  while (fgets(line, sizeof(line), file) != nullptr) {
80  stdoutstr << line;
81  }
82 
83  if (ferror(file) != 0) {
84  _pclose(file); // Ignoring result since we already have an error.
85  return Error("Error reading output of '" + command.get() + "'");
86  }
87 
88  int status;
89  if ((status = _pclose(file)) == -1) {
90  return Error("Failed to get status of '" + command.get() + "'");
91  }
92 
93  return stdoutstr.str();
94 }

114 // Concatenates multiple command-line arguments and escapes the values.
115 // If `arg` is not specified (or takes the value `0`), the function will
116 // scan `argv` until a `nullptr` is encountered.

118 {

120  unsigned long index = 0;
