76 // Returns the size in Bytes of a given file system entry. When
77 // applied to a symbolic link with `follow` set to
78 // `DO_NOT_FOLLOW_SYMLINK`, this will return the length of the entry
79 // name (strlen).
80 inline Try<Bytes> size(
81  const std::string& path,
82  const FollowSymlink follow = FOLLOW_SYMLINK)
83 {
84  switch (follow) {
85  case DO_NOT_FOLLOW_SYMLINK: {
86  Try<::internal::windows::SymbolicLink> symlink =
87  ::internal::windows::query_symbolic_link_data(path);
88 
89  if (symlink.isError()) {
90  return Error(symlink.error());
91  } else {
92  return Bytes(symlink.get().substitute_name.length());
93  }
94  break;
95  }
96  case FOLLOW_SYMLINK: {
97  struct _stat s;
98 
99  if (::_stat(path.c_str(), &s) < 0) {
100  return ErrnoError("Error invoking stat for '" + path + "'");
101  } else {
102  return Bytes(s.st_size);
103  }
104  break;
105  }
106  }
107 
108  UNREACHABLE();
109 }

112 inline Try<long> mtime(const std::string& path)
113 {
114  Try<::internal::windows::SymbolicLink> symlink =
115  ::internal::windows::query_symbolic_link_data(path);
116 
117  if (symlink.isSome()) {
118  return Error(
119  "Requested mtime for '" + path +
120  "', but symbolic links don't have an mtime on Windows");
121  }
122 
123  struct _stat s;
124 
125  if (::_stat(path.c_str(), &s) < 0) {
126  return ErrnoError("Error invoking stat for '" + path + "'");
127  }
128 
129  // To be safe, we assert that `st_mtime` is represented as `__int64`. To
130  // conform to the POSIX, we also cast `st_mtime` to `long`; we choose to make
131  // this conversion explicit because we expect the truncation to not cause
132  // information loss.
133  static_assert(
134  std::is_same<__int64, __time64_t>::value,
135  "Mesos assumes `__time64_t` is represented as `__int64`");
136  return static_cast<long>(s.st_mtime);
137 }
