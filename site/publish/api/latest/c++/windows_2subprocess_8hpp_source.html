45 // This function will invoke `os::close` on all specified file
46 // descriptors that are valid (i.e., not `None` and >= 0).
47 inline void close(
48  const InputFileDescriptors& stdinfds,
49  const OutputFileDescriptors& stdoutfds,
50  const OutputFileDescriptors& stderrfds)
51 {
52  HANDLE fds[6] = {
53  stdinfds.read, stdinfds.write.getOrElse(INVALID_HANDLE_VALUE),
54  stdoutfds.read.getOrElse(INVALID_HANDLE_VALUE), stdoutfds.write,
55  stderrfds.read.getOrElse(INVALID_HANDLE_VALUE), stderrfds.write
56  };
57 
58  foreach (HANDLE fd, fds) {
59  if (fd != INVALID_HANDLE_VALUE) {
60  os::close(fd);
61  }
62  }
63 }

66 // Creates a null-terminated array of null-terminated strings that will be
67 // passed to `CreateProcess` as the `lpEnvironment` argument, as described by
68 // MSDN[1]. This array needs to be sorted in alphabetical order, but the `map`
69 // already takes care of that. Note that this function does not handle Unicode
70 // environments, so it should not be used in conjunction with the
71 // `CREATE_UNICODE_ENVIRONMENT` flag.
72 //
73 // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682425(v=vs.85).aspx
74 inline Option<string> createProcessEnvironment(
75  const Option<map<string, string>>& env)
76 {
77  if (env.isNone() || (env.isSome() && env.get().size() == 0)) {
78  return None();
79  }
80 
81  string environmentString;
82  foreachpair (const string& key, const string& value, env.get()) {
83  environmentString += key + '=' + value + '\0';
84  }
85 
86  return environmentString;
87 }

90 inline Try<PROCESS_INFORMATION> createChildProcess(
91  const string& path,
92  const vector<string>& argv,
93  const Option<map<string, string>>& environment,
94  const InputFileDescriptors stdinfds,
95  const OutputFileDescriptors stdoutfds,
96  const OutputFileDescriptors stderrfds)
97 {
98  // Construct the environment that will be passed to `CreateProcess`.
99  Option<string> environmentString = createProcessEnvironment(environment);
100  const char* processEnvironment = environmentString.isNone()
101  ? nullptr
102  : environmentString.get().c_str();
103 
104  PROCESS_INFORMATION processInfo;
105  STARTUPINFO startupInfo;
106 
107  ::ZeroMemory(&processInfo, sizeof(PROCESS_INFORMATION));
108  ::ZeroMemory(&startupInfo, sizeof(STARTUPINFO));
109 
110  // Hook up the `stdin`/`stdout`/`stderr` pipes and use the
111  // `STARTF_USESTDHANDLES` flag to instruct the child to use them[1]. A more
112  // user-friendly example can be found in [2].
113  //
114  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms686331(v=vs.85).aspx
115  // [2] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682499(v=vs.85).aspx
116  startupInfo.cb = sizeof(STARTUPINFO);
117  startupInfo.hStdInput = stdinfds.read;
118  startupInfo.hStdOutput = stdoutfds.write;
119  startupInfo.hStdError = stderrfds.write;
120  startupInfo.dwFlags |= STARTF_USESTDHANDLES;
121 
122  // Build command to pass to `::CreateProcess`.
123  //
124  // NOTE: We are expecting that components of `argv` that need to be quoted
125  // (for example, paths with spaces in them like `C:\"Program Files"\foo.exe`)
126  // to have been already quoted correctly before we generate `command`.
127  // Incorrectly-quoted command arguments will probably lead the child process
128  // to terminate with an error. See also NOTE on `process::subprocess`.
129  string command = strings::join(" ", argv);
130 
131  // Escape the quotes in `command`.
132  //
133  // TODO(dpravat): Add tests cases that cover this functionality. See
134  // MESOS-5418.
135  command = strings::replace(command, "\"", "\\\"");
136 
137  // NOTE: If Mesos is built against the ANSI version of this function, the
138  // environment is limited to 32,767 characters. See[1].
139  //
140  // TODO(hausdorff): Figure out how to map the `path` and `args` arguments of
141  // this function into a call to `::CreateProcess` that is more general
142  // purpose. In particular, on POSIX, we expect that calls to `subprocess` can
143  // be called with relative `path` (e.g., it could simply be `sh`). However,
144  // on Windows, unlike the calls to (e.g.) `exec`, `::CreateProcess` will
145  // expect that this argument be a fully-qualified path. In the end, we'd like
146  // the calls to `subprocess` to have similar command formats to minimize
147  // confusion and mistakes.
148  //
149  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682425(v=vs.85).aspx
150  BOOL createProcessResult = CreateProcess(
151  nullptr,
152  (LPSTR)command.data(),
153  nullptr, // Default security attributes.
154  nullptr, // Default primary thread security attributes.
155  TRUE, // Inherited parent process handles.
156  0, // Normal thread priority.
157  (LPVOID)processEnvironment,
158  nullptr, // Use parent's current directory.
159  &startupInfo, // STARTUPINFO pointer.
160  &processInfo); // PROCESS_INFORMATION pointer.
161 
162  if (!createProcessResult) {
163  return WindowsError("createChildProcess: failed to call 'CreateProcess'");
164  }
165 
166  return processInfo;
167 }
