32 // Write out the string to the file at the current fd position.
33 inline Try<Nothing> write(int fd, const std::string& message)
34 {
35  size_t offset = 0;
36 
37  while (offset < message.length()) {
38  ssize_t length =
39  os::write(fd, message.data() + offset, message.length() - offset);
40 
41 #ifdef __WINDOWS__
42  int error = WSAGetLastError();
43 #else
44  int error = errno;
45 #endif // __WINDOWS__
46 
47  if (length < 0) {
48  // TODO(benh): Handle a non-blocking fd? (EAGAIN, EWOULDBLOCK).
49  if (net::is_restartable_error(error)) {
50  continue;
51  }
52  return ErrnoError();
53  }
54 
55  offset += length;
56  }
57 
58  return Nothing();
59 }

62 // A wrapper function that wraps the above write() with
63 // open and closing the file.
64 inline Try<Nothing> write(const std::string& path, const std::string& message)
65 {
66  Try<int> fd = os::open(
67  path,
68  O_WRONLY | O_CREAT | O_TRUNC | O_CLOEXEC,
69  S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
70 
71  if (fd.isError()) {
72  return ErrnoError("Failed to open file '" + path + "'");
73  }
74 
75  Try<Nothing> result = write(fd.get(), message);
76 
77  // We ignore the return value of close(). This is because users
78  // calling this function are interested in the return value of
79  // write(). Also an unsuccessful close() doesn't affect the write.
80  os::close(fd.get());
81 
82  return result;
83 }
